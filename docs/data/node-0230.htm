Child actions


public class HomeController : Controller

{

    [ChildActionOnly]                               // this action method can be used for child actions only

    public ActionResult MyChildAction(int id = 0)

    {

        return PartialView(id);                     // the child action method usually renders a partial view

    }

}


<div>

    @Html.Action("MyChildAction")                   <!-- insert the result of a child action into a view            -->

    @Html.Action("MyChildAction", "Home")           <!-- call a child action of a specific controller               -->

    @Html.Action("MyChildAction", new { id = 1 })   <!-- pass input parameters to the child action                  -->

</div>

Helper methods (external)


public static class CustomHelpers                   // implement a class with extension methods

{

    public static MvcHtmlString BoldParagraph(

                        this HtmlHelper html,       // this method is an extension to the HtmlHelper class

                        string text)                // pass parameters as required

    {

        html.RouteCollection                        // routing information

        html.ViewBag                                // dynamic data

        html.ViewContext.Controller                 // the controller processing this requests

        html.ViewContext.HttpContext                // HTTP request and response

        html.ViewContext.RouteData                  // routing information

        html.ViewContext.View                       // the view that invoked the helper method


        TagBuilder bold = new TagBuilder("b");      // use the TagBuilder to construct HTML elements

        bold.SetInnerText(html.Encode(text));       // use htmlEncode to mask special HTML characters


        TagBuilder paragraph = new TagBuilder("p");

        paragraph.AddCssClass("myCssClass");

        paragraph.InnerHtml += bold.ToString();


        return new MvcHtmlString(                   // return the result as plain HTML text (it contains HTML markup

                        paragraph.ToString());      // and special characters will not be masked)

    }


    public static string Hi(this HtmlHelper html)   // return a string for non-html output

    {

        return "<b>Hello world!</b>";               // will be converted to "&lt;b&gt;Hello world!&lt;/b&gt;"

    }   

}


@using MVC.Infrastructure                           <!-- import the extension class' namespace into the view        -->

@Html.BoldParagraph("Hello world!")                 <!-- call the extension method -->
