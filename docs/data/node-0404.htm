public class MyServlet extends HttpServlet {        // use GenericServlet as base class for non-http servlets


    public String getServletInfo() {return "Info";} // optionally provide a short description of the servlet


    public void init() {...}                        // override these methods if initialisation/cleanup is needed during the

    public void destroy() {...}                     // instantiation or destruction of servlets


    public void service(HttpServletRequest request, // override this method for non-http servlets

                        HttpServletResponse response)

                        throws ServletException,

                               IOException {...}


    public void doGet(HttpServletRequest request,   // for HTTP servlets, override some of those methods instead:

                      HttpServletResponse response) // doGet, doPost, doPut, doDelete, doHead, doOptions and doTrace

                      throws ServletException,

                             IOException {


        this.getServletContext().log("...");        // log via the servlet context


        request.getCookies()                        // obtain request details#

        request.getSession()

        request.getHeader("ACCEPT")

        request.getQueryString()

        ...


        request.getInputStream()                    // retreive request payload (body) data                  

        request.getReader()                   


        response.setContentType("text/html");       // configure the response object

        response.setStatus(200);

        response.addCookie(new Cookie("k", "v"));

        PrintWriter pw = response.getWriter();      // ... and write the resulting payload (body) data

        pw.print("<b>Hello world!</b>");

        pw.close();


        response.sendRedirect("/home");             // or redirect the browser

    }

}
