Struct


Person = Struct.new(:name, :country)      # create a class with constructor and accessors


class Person                              # can optionally extend the class with additional methods

  def to_s

    "#{name} (#{country})"                # attributes MUST be accessed without "@"

  end

end


fred = Person.new("David", "Australia")   # the struct automatically provides a constructor



OpenStruct


require 'ostruct'                           # standard library


person = OpenStruct.new                     # does not create a class, but an object with dynamic properties

person.name = "David"                       # can assign any property on-the-fly

person.country = "Australia"


person.age == nil                           # accessing undefined properties returns nil
