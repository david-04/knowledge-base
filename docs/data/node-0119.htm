JavaScript API


$(".btn").click(function () {                       // this requires JavaScript

    $(this).button("loading")                       // depreacted: switch the button to the loading state

           .delay(1000)

           .queue(function () {

               $(this).button("reset");             // revert it back to the normal state

               $(this).dequeue();

           });

});


$("#myButton").button("toggle");                    // toggle between selected/pushed and unselected

$("#myButton").button("loading");                   // deprecated: switch to the loading state or back

$("#myButton").button("reset");                     // revert to the non-loading normal state

$("#myButton").button("complete");                  // set the button to any custom state
