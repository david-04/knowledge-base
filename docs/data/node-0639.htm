formatter.Serialize(myStream, myObject)                       ' serialise the given object into the specified stream

myObject = CType(formatter.Deserialize(myStream), SomeClass)  ' returns an untyped object that requires casting

Custom mapping (not for XmlSerializer)


<Serializable>

Public Class SomeClass

    Implements ISerializable                                  ' implement the interface ISerializable


    Public Property Id As Integer


    Public Sub GetObjectData(info As SerializationInfo,       ' called by the serialiser to retrieve the object data

                             context As StreamingContext) _

        Implements ISerializable.GetObjectData

        info.AddValue("MyIdField", Me.Id)                     ' only fields mapped here are serialised

    End Sub


    Public Sub New(info As SerializationInfo,                 ' special construtor for deserialisation

                   context As StreamingContext)

        Me.Id = info.GetInt32("MyIdField")                    ' manually retrieve the values

    End Sub


End Class
