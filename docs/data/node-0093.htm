Value pipes


import { Pipe,

         PipeTransform } from "@angular/core";


@Pipe({name: "quoteIt"})                            // give this pipe an external (symbolic) name

export class QuotePipe implements PipeTransform {


    transform(stdin: string, quote: string) {       // the pipe receives its input data and optional config parameters (quote)

        if (quote) return quote + stdin + quote;    // transform the input data and beware of non-provded config parameters

        else return stdin;

    }

}

Collection pipes


import { Pipe,

         PipeTransform } from "@angular/core";

import { Person        } from "./data-model"


@Pipe({ name: "adultsOnly"})                        // this pipe is "pure" it is only invoked if the whole input array

export class AdultPipe implements PipeTransform {   // is replaced by another one; it will not update the view if existing

                                                    // elements within an array are modified or if new elements are pushed

    transform(stdin: Person[],                      // into an array

              minAge: number): Person[]{

        minAge = minAge || 18;

        if (stdin != null) {

            return stdin.filter(

                person => (minAge <= person.age)

             );

        } else {

            return stdin;

        }

    }

}


@Pipe({                                             // derive an "impure" pipe: it will be executed during every change

    name: "adultsOnlyImpure",                       // detection cycle; it detects modified and added/removed array elements,

    pure: false                                     // but comes with a heavy performance penalty

})

export class AdultImpurePipe extends AdultPipe { }
