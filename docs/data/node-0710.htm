Boolean


xs:boolean                                          // can be true, false, 0 or 1



Numeric


xs:unsignedByte                                     //                          0 ... 255

xs:unsignedShort                                    //                          0 ... 65,535

xs:unsignedInt                                      //                          0 ... 429,4967,295

xs:unsignedLong                                     //                          0 ... 1,844,674,407,370,955,1615


xs:nonNegativeInteger                               //                          0 ... oo

xs:positiveInteger                                  //                          1 ... oo

xs:negativeInteger                                  //                        -oo ... -1

xs:nonPositiveInteger                               //                        -oo ...  0


xs:byte                                             //                       -128 ... 127

xs:short                                            //                    -32,768 ... 32,767

xs:int                                              //             -2,147,483,648 ... 2,147,483,647

xs:long                                             // -9,223,372,036,854,775,808 ... 9,223,372,036,854,775,807

xs:integer                                          //                        -oo ... oo


xs:decimal                                          // decimal number without any restrictions (no exponential format)

xs:double                                           // 8-byte float (supports the exponential format 2.5E+10, NaN, INF and -INF)

xs:float                                            // 4-byte float (supports the exponential format 2.5E+10, NaN, INF and -INF)



Strings


xs:string                                           // unicode string

xs:normalizedString                                 // a string with the restriction whiteSpace="replace"


xs:anyURI                                           // URL or URN (uniform resource name)

xs:base64Binary                                     // base64 encoded binary data

xs:hexBinary                                        // binary data (each byte is represented as [0-9A-Z][0-9A-Z])

xs:language                                         // en, en-AU, custom languages via X prefix, e.g. X-Saesisch


xs:ENTITIES                                         // whitespace separated list of entity names

xs:ENTITY                                           // named entity

xs:ID                                               // id must be unique amongst all xs:ID elements

xs:IDREF                                            // must refer to an ID set elsewhere in the document

xs:IDREFS                                           // whitespace-separated list of IDREFs

xs:Name                                             // a valid XML (element) name (including a namespace prefix)

xs:NCName                                           // non-colonized name (an XML element name without namespace prefix)

xs:NMTOKEN                                          // a valid XML token name

xs:NMTOKENS                                         // a list of XML token names separated by blanks

xs:NOTATION                                         // can't be instantiated directly, must be derived

xs:QName                                            // qualified name (with a namespace prefix)

xs:token                                            // normalised string without [\t\r\n], without leading/trailing blanks and without "  +"



Date and time


xs:date                                             // YYYY-MM-DD with optional timezone (2016-12-24+08:00 or 2016-12-24Z for UTC)

xs:dateTime                                         // YYYY-MM-DDThh:mm:ss with optional timezone (2016-12-24T18:45:00+8:00, Z for UTC)

xs:time                                             // ±hh:mm:ss.xxx (with timezone: 15:41:00-05:00 or 15:41:00Z)


xs:duration                                         // ±PnYnMnDTnHnMnS (e.g. P3M2D = 3 months + 2 days, zeros can be omitted)


xs:gDay                                             // day of month, format: ---DD (e.g. ---31; with timezone: ---05+08:00 or ---05Z for UTC)

xs:gMonth                                           // month, format: --MM-- (e.g. --12--; with timezone: --12--+08:00 --12--Z for UTC)

xs:gMonthDay                                        // day of the month: --MM-DD (e.g. --12-24; with timezone: --12-24+08:00 or --12-24Z for UTC)

xs:gYear                                            // year: YYYY (e.g. 2016; with timezone: 2016+08:00 or 2016Z for UTC)

xs:gYearMonth                                       // year+month: YYYY-MM (e.g. 2016-12; with timezone: 2016-12+08:00 or 2016-12Z for UTC)
