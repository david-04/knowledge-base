public class MyPerson

{

    [Display(Name = "Full name")]

    public string name { get; set; }                // properties must be public for the model-binder to work

}


[MetadataType(typeof(MyAnnotatedClass))]            // for generated classes: instruct MVC to obtain metadata from

public partial class MyGeneratedClass               // another class (the generated class must be partial)

{

    public string name { get; set; }

}

public partial class MyAnnotatedClass               // MVC uses this class to overlay the original one above

{

    [Display(Name = "Full name")]

    public string name { get; set; }                // include only properties that need annotations

}

Controllers


public class HomeController : Controller

{

    [HttpGet]                                       // add a GET action method to show the form

    public ActionResult Index()                    

    {

        return View(new Person());                  // pass in an empty (unpopulated) view model

    }


    [HttpPost]                                      // add a POST action method to handle the postbacks

    public ActionResult Index(Person person)        // automatic model binding passes in deserialised form data

    {

        if (!ModelState.IsValid)

        {

            return View(person);                    // if there are validation errors: render the form again

        }

        else

        {

            return Redirect("/Home/Thank-you");     // otherwise process the data and redirect to a GET URL

        }

    }

}
