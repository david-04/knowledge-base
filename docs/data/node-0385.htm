Images


<img src         = "my-image.jpg"                   <!-- an image (URL)                                                        -->

     longdesc    = "my-image-description.html"      <!-- URL with a document containing a long description of the image        -->

     width       = "100"                            <!-- image width in pixels                                                 -->

     height      = "100"                            <!-- image height in pixels                                                -->

     alt         = "alternate text"                 <!-- alternate text (when the image can't be loaded or displayed)          -->

     crossorigin = "{anonymous|use-credentials}"    <!-- images from third-party sites to be used with canvases                -->

     ismap       />                                 <!-- specifies an image as a server-side image-map                         -->



Data URIs


> base64 -w 0 my-image.jpg                          <!-- convert an image file into base64                                     -->


data:image/jpeg;base64,R0lGODlhEAA...               <!-- JPEG                                                                  -->

data:image/png;base64,R0lGODlhEAA...                <!-- PNG                                                                   -->

data:image/gif;base64,R0lGODlhEAA...                <!-- GIF                                                                   -->

data:image/x-icon;base64,R0lGODlhEAA...             <!-- ICO                                                                   -->



Captions


<figure>                                            <!-- a figure with a caption                                               -->

    <img src="image.jpg" alt="my image">            <!-- alt can be omitted if a figcaption is provided                        -->

    <figcaption>caption</figcaption>                <!-- the caption to be displayed below the image                           -->

</figure>



Image maps (server-side)


<a href= "./click.php">

    <img src="my-image.jpg" ismap />                <!-- will link to ./click.php?10,35 (adding the coordinates to the link)   -->

</a>



Image maps (client-side)


<img src="my-image.jpg" usemap="#myImageMap">       <!-- attach an image map to an image                                       -->


<map name="myImageMap">                             <!-- the image map must have a name                                        -->

    <area shape    ="{default|rect|circle|poly}"    <!-- default denotes the whole image area                                  -->

          coords   = "..."                          <!-- rect: x1,y1,x2,y2, circle: x,y,radius, poly: x1,y1,x2,y2,...xn,yn     -->

          href     = "http://google.com.au/"        <!-- the link associated with this area of the image                       -->

          hreflang = "en"                           <!-- natural language of the linked resource                               -->

          alt      = "alternate text"               <!-- link should be downloaded (optional file name as suggestion)          -->

          download = "save-as-name.txt"             <!-- force download and (optionally) suggest a save-as filename            -->

          type     = "text/html"                    <!-- the MIME type of the linked resource                                  -->

          media    = "print and (resolution:50dpi)" <!-- media-query indicating which media the link is optimised for          -->

          target   = "myFrameName"                  <!-- or set to _blank|_parent|_self|_top                                   -->

          rel      = "alternate"                    <!-- alternate presentation of the document (e.g. a translation)           -->

          rel      = "author"                       <!-- link to the document's author                                         -->

          rel      = "bookmark"                     <!-- permanent URL of the current document (used for bookmarking)          -->

          rel      = "help"                         <!-- link to a help document                                               -->

          rel      = "license"                      <!-- link to copyright information                                         -->

          rel      = "next"                         <!-- next document in the series (i.e. next page)                          -->

          rel      = "nofollow"                     <!-- unendorsed document (e.g. advert), crawlers should not follow         -->

          rel      = "noreferrer"                   <!-- browser should not send a referrer when following this link           -->

          rel      = "prefetch"                     <!-- target document should be cached in advance                           -->

          rel      = "prev"                         <!-- previous document in the series (i.e. previous page)                  -->

          rel      = "search"                       <!-- link to a tool that can be used to search the current document        -->

          rel      = "tag"/>

</map>
