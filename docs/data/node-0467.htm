Common event properties


event.bubbles                                       // indicates if this is a bubbling event

event.cancelable                                    // event can be cancelled

event.currentTarget                                 // the element to which the event handler has been attached

event.defaultPrevented                              // DOM Level 3: preventDefault() has been called before

event.detail                                        // additional event detail information

event.eventPhase                                    // 1=capturing phase, 2=at target, 3=bubbling

event.preventDefault()                              // cancel out the event's default behaviour (e.g. browsing when clicking on a link)

event.stopImmediatePropagation()                    // DOM Level 3: stop further capturing/bubbling as well as other chained handlers

event.stopPropagation()                             // stop further capturing/bubbling

event.target                                        // the innermost element that initiated the event

event.trusted                                       // DOM Level 3: was the event raised by the browser (or by JavaScript code?)

event.type                                          // e.g. "click"

event.view                                          // the window object in which the event occurred


event.cancelBubble                                  // IE only: set to true to cancel propagation

event.returnValue                                   // IE only: set to false to cancel the default behaviour

event.srcElement                                    // IE only: the innermost element that initiated the event

onmousedown                                         // any mouse key (right or left) pressed down

onmouseup                                           // mouse key was released

onmouseenter                                        // mouse moved into the screen area covered by the element (non-bubbling)

onmouseleave                                        // mouse moved out of the area occupied by the element (non-bubbling)

onmouseover                                         // mouse moved out of the area occupied by the element (bubbling)

onmouseout                                          // mouse moved into the screen area covered by the element (bubbling)

onmousemove                                         // mouse is being moved around


event.client{X|Y}                                   // location of the mouse position in the viewport

event.page{X|Y}                                     // location of the mouse position on the page/document

event.screen{X|Y}                                   // location of the mouse position on the screen

event.{shift|ctrl|alt|meta}Key                      // indicates if the Shift/Ctrl/Alt/Meta key was being pressed

event.relatedTarget                                 // only mouse{over|out}: the element where the mouse comes from or goes to (IE <= 8: fromElement and toElement)

event.button                                        // only mouse{up|down}: 0 (primary button), 1 (middle button), 2 (secondary button) (not in IE <= 8)

event.detail                                        // counts the clicks (mousedown+mouseup) at the very same (pixel) location

Keyboard and text events


onkeydown                                           // a key was (or is being) pressed

onkeypress                                          // deprecated: key press that generates one or more characters

onkeyup                                             // a key was released

ononinput                                           // replaces onkeypress (does not fire in IE <= 9 when deleting characters)

onselect                                            // user has selected text

oncontextmenu                                       // context menu (call event.preventDefault() to not show the system's context menu)


event.keyCode                                       // only key{up|down}: ASCII code (number) of the lowercase character pressed (IE, Chrome and Safari return 186 for ";")

event.charCode                                      // only keypress: ASCII code of the character pressed (not reliable for ctrl-keys, is often 0)

event.key                                           // DOM L3: the character pressed as a string (e.g. "x", "X", "Shift" or "Down")

event.char                                          // DOM L3: the same as event.key but is set to null when a non-character (control) key is pressed (not in IE <= 9)

event.keyIdentifer                                  // Safari 5 + Chrome only: the same as event. key but "U+0000" unicode character for control keys

event.location                                      // DOM L3: 0=default keyboard, 1=left (e.g. left Alt key), 2=right, 3=numberic,

                                                    // 4=mobile (virtual keyboard), 5=joystick

                                                    // (only IE 9+; Safari 5 + Chrome use keyLocation but only set it to 0 or 3)

event.getModifierState("Shift")                     // DOM L3: check if the modifier key (Shift, Control, Alt, AltGraph, Meta) was pressed

event.{shift|ctrl|alt|meta}Key                      // indicates if the Shift/Ctrl/Alt/Meta key was being pressed

Drag-and-drop


ondragstart                                         // HTML5: a drag operation has started (fired on the dragged element)

ondrag                                              // fired repeatedly as long as the element is being dragged

ondragend                                           // the drag operation has ended (maybe without dropping the element)


ondragenter                                         // the element was dragged to a valid target (fired on the potential drop-target)

ondragover                                          // fired repeatedly while the element is above a valid target

ondragleave                                         // the element was dragged out of the valid target

ondrop                                              // the element was dropped on a valid target


event.target                                        // the element being dragged

event.dataTransfer.setData("type", "data")          // called in dragstart, type={Text|url} changed to {text/plain|text/uri-list} in HTML5 (not in IE <= 10)

event.dataTransfer.getData("type")                  // retrieve the data with the given mime type

event.dataTransfer.clearData("type")                // remove the data with the given mime type

event.dataTransfer.types                            // list of all MIME types for which data is present

event.dataTransfer.dropEffect                       // set in ondragenter to change the cursor style (values: none, move, copy, link)

event.dataTransfer.effectAllowed                    // set in ondragstart to: "uninitialized", "none", "copy", "link", "move", "copyLink", "copyMove", "linkMove", "all"

event.dataTransfer.setDragImage(element, x, y)      // set an HTML <img> element as the drag-icon with the cursor at image's (x,y) position


element.addEventListener("dragover",                // make any element a valid drag target

    function () { event.preventDefault(); })

element.addEventListener("dragenter",                                       

    function () { event.preventDefault(); })

element.addEventListener("drop",

    function () { event.preventDefault(); })


<div draggable="true" />                            <!-- allow any element to be dragged (dragstart must call event.dataTransfer.setData) -->




Drag-and-drop example


dragSourceDomElement.addEventListener(              // enable an element to be dragged away

    "mousedown",

    function (event) {

        if (typeof this.style.WebkitUserDrag

            !== "undefined") {

            this.style.WebkitUserDrag = "element";

        }

        if (typeof this.draggable !== "undefined") {

            this.draggable = true;

        }

        if (typeof this.dragDrop === "function") {

            this.dragDrop();

        }

    }

);


dragSourceDomElement.addEventListener(              // copy content to the clipboard when the dragging starts

    "dragstart",

    function (event) {

        event.stopPropagation();

        var dataTransfer = event.dataTransfer;

        dataTransfer.effectAllowed = "copyMove";

        try {

            dataTransfer.setData("text/html",     

                                 $(this).html());

            dataTransfer.setData("text/plain",

                                 $(this).html());

        }

        catch (error) {

            dataTransfer.setData("Text", html);     // IE compatibilitiy

        }

    }

);


dragSourceDomElement.addEventListener(              // remove the dragged content when the dragging ends

    "dragend",                                      // will also be fired if the drag-and-drop did not hit a droppable target

    function (event) {

        this.parentNode.removeChild(this);

    }

);



File-drop example


dropTargetDomElement.addEventListener(              // something is begin dragged onto a potential target

    "dragenter",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

    }

);


dropTargetDomElement.addEventListener(              // called for each and every mouse move within the potential drop target

    "dragover",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

    }

);


dropTargetDomElement.addEventListener(              // mouse leaves the drop target

    "dragleave",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

    }

);


dropTargetDomElement.addEventListener(              // retrieve the clipboard data during the acutal drop

    "drop",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

        var dataTransfer = event.dataTransfer;

        try {

            this.innerHTML =

                dataTransfer.getData("text/html");

        }

        catch (error) {

            this.innerHTML =

                dataTransfer.getData("Text");

        }

    }

);



HTML5 drop events (files)


dropTargetDomElement.addEventListener(              // add support for dropping files

    "dragenter",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

    }

);


dropTargetDomElement.addEventListener(              // called for each and every mouse move within the potential target

    "dragover",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

    }

);


dropTargetDomElement.addEventListener(              // the mouse leaves the drop target

    "dragleave",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

    }

);


dropTargetDomElement.addEventListener(              // the actual drop operations

    "drop",

    function (event) {

        event.preventDefault();

        event.stopPropagation();

        var dataTransfer = event.dataTransfer;

        var html = "";

        for (var i = 0;                             // process each file separately

             i < dataTransfer.files.length; i++) {

            file = dataTransfer.files[i];

            if (!file.name && file.fileName) {      // unify the file's name and size

                file.name = file.fileName;

            }

            if (!file.size && file.fileSize) {

                file.size = file.fileSize;

            }

            var mimeType = file.type;               // the MIME type (like image/png) is passed in as well


            if (typeof FormData !== 'undefined') {  // upload the files (requires HTML5 FormData)

                var form = new FormData();

                form.append('file[]',               // (virtual) name of the form field (use [] to indicate an array)

                            file, file.name);

                xhr = new XMLHttpRequest();

                xhr.open('POST', '/path');

                xhr.send(form);

            }


            var reader = new FileReader();          // load the file contents; browser check via: if(FileReader) {...}

            reader.onload = function (event) {

                var contents = event.target.result;

            };

            reader.readAsText(file);                // also: readAsArrayBuffer(), readAsDataURL()

        }

    }

);
