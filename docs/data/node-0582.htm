Variables


$myProperty: background-color;                      /* define a global variable                                             */

$myProperty: color !default;                        /* assign a value only if the variable is undefined or null             */


body {

    $color      : red;                              /* define a local variable (confined to the rule's scope)               */

    $width      : 50% !global;                      /* define a global variable (accessible outside the rule's scope)       */

    #{$property}: $color;                           /* reference variables; #{$variable} inserts the value without quotes   */

}                                                   /* and prevents futher expansion (like arithmetic operations)           */



Data types


numbers                                             /* 1.2, 13, 10px                                                        */

strings                                             /* "foo", 'bar', baz, "the value is #{$myVariable}"                     */

colors                                              /* blue, #04a3f9, rgba(255, 0, 0, 0.5)                                  */

booleans                                            /* true, false                                                          */

nulls                                               /* null                                                                 */

lists                                               /* (1.5em 1em 0 2em) or (Helvetica, Arial, sans-serif)                  */

maps                                                /* (key1:  value1, key2: value2)                                        */



Lists


$myList: 1px solid black;                           /* can be assigned to a space-separated list of values                  */

$myList: "Arial", "Times New Roman", sans-serif;    /* ... or can use the comma as a separator                              */

$myList: 1px solid, "Arial" sans-serif;             /* ... or can use blanks and commas at the same time                    */

$myList: ("Arial", "Times New Roman"  sans-serif);  /* ... and might additionally be wrapped into parentheses               */

$myList: (1px 2px) (10px 12px);                     /* ... list of lists                                                    */

$myList: ();                                        /* define an empty list                                                 */

$myList: 100%,;                                     /* a list with only ont item                                            */

$myList: 1px 2px,;                                  /* a list with only ont item that is a list itself                      */





nth($myList, 2);                                    /* extract the second element from the list                             */

join($myList, $myOtherList);                        /* concatenate two lists                                                */

append($myList, 1px 100%);                          /* append an element (or list) to a list                                */



Maps


$myMap: (key1:value1, key2:value2);                 /* define a key-value map                                               */





map-get($myMap, key1);                              /* retrieve a single key's value from the map                           */

map-merge($myMap1, $myMap2);                        /* merge two maps into one                                              */

inspect($myMap);                                    /* create a JSON-like debug-friendly dump                               */



Colors


$myColour: #010203;                                 /* define a colour                                                      */

$myColour: rgba(255, 0, 0, 0.75);

$myColour: hsl(0, 100%, 50%);                       /* or: hsl($hue: 0, $saturation: 100%, $lightness: 50%);                */





$myColour1 * 2 + $myColour2;                        /* colours can be used for arithmetic operations                        */

opacify($myColour, 0.3);                            /* add 0.3 to the opacity                                               */

transparentize($myColour, 0.3);                     /* subtract 0.3 from the opacity                                        */

ie-hex-str($myColour)                               /* convert into IE-specific hex format with opacity (#FFFFFFFF)         */
