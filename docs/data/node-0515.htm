Request object


request.headers['user-agent']                       // key-value-hashtable of HTTP request headers (key names are always lowercase)

request.acceptedLanguages                           // array of natural languages accepted by the client

request.accepts(['text/html', 'text/plain'])        // convenience method to check the HTTP accept header

request.xhr                                         // boolean indicating if this is an Ajax request


request.ip                                          // client IP address


request.url                                         // path+querystring (without protocol/domain/port)

request.originalUrl                                 // same as url but retains its value even if url is rewritten internally

request.protocol                                    // http or https

request.secure                                      // boolean indicating if this is a TLS connection

request.host                                        // the server hostname used by the client (can be spoofed)

request.path                                        // path without protocol/domain/port/querystring

request.route                                       // information about the currently used route

request.query                                       // key-value-hashtable of all query string parameters


request.params                                      // array with named route paramters

request.param('name')                               // ambiguous: retrieves a named rout parameter or get/post parameters (avoid)

request.cookies                                     // object containing the cookies sent by the client

request.signedCookies


request.body                                        // use a middleware to parse the body into an object, can have the following MIME types

                                                    // application/x-www-form-urlencoded  => POST form

                                                    // multipart/form-data                => POST file upload

                                                    // application/json                   => Ajax request



Response


app.disable('x-powered-by');                        // don't ad the X-Powered-By response header


response.status(404)                                // set the response status code

                                                    // 200              ok

                                                    // 301,302,303,307  redirect (303 enforces a GET request)

                                                    // 400,401,404      bad request, unauthorized, file not found

                                                    // 500              internal server error

response.set('Content-Type', 'text/plain')          // set a response header

response.type('text/html')                          // set the content type

response.type('txt')                                // types without string are interpreted as file extensions and mapped to a type automatically


response.redirect('/contact')                       // redirect (defaults to status code 302 - found)

response.redirect(301, '/contact')                  // recirect with "301 - moved permanently"


response.cookie('name', 'value', [options])         // set a cookie (requires middleware support)

response.clearCookie('name', [options])             // delete a cookie


response.send('<html><body>...')                    // send body data (default content type is text/html)

response.send(jsonObject)                           // send body data as serialised JSON (sets the content type automatially)

response.send(200, '<html><body>...')               // send body data with a selected HTTP response code

response.json(jsonObject)                           // send a JSON object as body data (sets the content type automatically)

response.json(200, jsonObject)

response.jsonp(200, jsonObject)                     // send a JSONP object as body data

response.format({                                   // send the data that best matches the client's accept header

    'text/plain': 'hi  there',

    'text/html': '<b>hi there</b>'

})                            



response.attachment('filename')                     // browser hint: please download this file (filename is will be the suggested name)

response.download(                                  // send the given file as body data and advice the browser to download the file (rather than viewing it online)

    './video.mp4',                                  // the file to be downloaded

    ['filename'],                                   // browser hint: use this filename when saving the download locally

    [callback]                                      // callback function once the download is complete

)    

res.sendFile(                                       // send the given file as body data (but do not instruct the browser to download/save it)

    './styles/default.css',

    [options],

    [callback]

)


response.locals                                     // default context for the rendering views (used to pass data to them)

response.render('my-view', [locals], callback)      // render the given view (options can be an object that overrides some parts of response.locals)
