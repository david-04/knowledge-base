Custom dialogs (AWT)


class MyDialog extends Dialog {                     // custom dialog window

   

    public MyDialog(Frame parent) {                 // dialogs must be attached to a parent window, frame or dialog

       

        super(parent, "My dialog", false);          // initialise the dialog as non-modal (use "true" for modal)

       

        this.setLayout(new FlowLayout());           // set the layout manager

        this.add(new Label("placeholder"));         // ... and add components as required


        this.addWindowListener(new WindowAdapter(){ // handle the closing of the dialog

            public void windowClosing(

                               WindowEvent e) {

                dispose();

            }

        });

    }

}


MyDialog myDialog = new MyDialog(this);             // create and display a dialog

myDialog.setVisible(true);
