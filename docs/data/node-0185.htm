@SpringBootApplication                              // this is the main application

public class Application {


    public static void main(String[] args) {

        SpringApplication.run(                      // tell the Spring Boot framework to start the application

                Application.class, args

        );

    }


    @Bean                                           // this method returns an object that is stored/managed by the container

    public CommandLineRunner

        commandLineRunner(ApplicationContext ctx) {


        return args -> {

            String[] beanNames = ctx.getBeanDefinitionNames();  // pre-bundled beans

            Arrays.sort(beanNames);

            for (String beanName : beanNames) {

                System.out.println(beanName);

            }

        };

    }

}
