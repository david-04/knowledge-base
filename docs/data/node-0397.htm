CustomerType.java


public class CustomerType {                         // each POJO class must have a default constructor + getters + setters


    private int id;

    private String description;

   

    public CustomerType() {}                       

    public CustomerType(int id, String description) {

        this.id          = id;

        this.description = description;

    }

                                                    // must have getters and setters for each property

    public int    getId() { return id; }

    public void   setId(int id) { this.id = id; }

    public String getDescription() { return description; }

    public void   setDescription(String description) { this.description = description; }

}



Order.java


public class Order {


    private int id;

    private Customer customer;

    private Date orderDate;

    private int items;

    private double price;

   

    public Order() {}

    public Order(int id, Customer customer, Date orderDate, int items, double price) {

        super();

        this.id        = id;

        this.customer  = customer;

        this.orderDate = orderDate;

        this.items     = items;

        this.price     = price;

    }

   

    public int      getId() { return id; }

    public void     setId(int id) { this.id = id; }

    public Date     getOrderDate() { return orderDate; }

    public void     setOrderDate(Date orderDate) {this.orderDate = orderDate; }

    public int      getItems() { return items; }

    public void     setItems(int items) { this.items = items; }

    public double   getPrice() { return price; }

    public void     setPrice(double price) { this.price = price; }

   

    public Customer getCustomer() { return customer; }

    public void     setCustomer(Customer customer) { this.customer = customer; }

}



Customer.java


public class Customer {

   

    private int id;

    private String firstName;

    private String lastName;

    private CustomerType customerType;

    private Set<Order> orders;

   

    public Customer() {}

    public Customer(int id, String firstName, String lastName, CustomerType customerType, Set<Order> orders) {

        this.id           = id;

        this.firstName    = firstName;

        this.lastName     = lastName;

        this.customerType = customerType;

        this.orders       = orders;

    }

   

    public int          getId() {return id;}

    public void         setId(int id) { this.id = id; }

    public String       getFirstName() { return firstName; }

    public void         setFirstName(String firstName) { this.firstName = firstName; }

    public String       getLastName() { return lastName; }

    public void         setLastName(String lastName) { this.lastName = lastName; }


    public CustomerType getCustomerType() { return customerType; }

    public void         setCustomerType(CustomerType customerType) { this.customerType = customerType; }

    public Set<Order>   getOrders() { return orders; }

    public void         setOrders(Set<Order> orders) { this.orders = orders; }

}

    <id name="id" type="int">

      <generator class="assigned" />

    </id>

    <property generated="never" lazy="false" name="firstName"

                       type="java.lang.String">

      <column name="FIRSTNAME" />

    </property>

    <property generated="never" lazy="false" name="lastName"

                       type="java.lang.String">

      <column name="LASTNAME" />

    </property>

                       name="customerType">

      <column name="CUSTOMERTYPEID" />

    </many-to-one>

    <set name="orders" sort="unsorted">

      <key column="CustomerId" />

      <one-to-many class="myPackage.Order" />

    </set>

  </class>

</hibernate-mapping>
