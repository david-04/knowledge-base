Strings and numbers


SET         name "David"                            // add a key

SET         name "David" EX 60                      // set a key that expires in 60 seconds

SET         name "David" PX 1000                    // set a key that expires in 1000 milliseconds

SET         name "David" NX                         // only set the key if it does not already exist

SET         name "David" XX                         // only set the key if it does already exist


SETEX       name 60 "David"                         // set a key and attach the expiration time in one atomic operation


MSET        name "David" age 38                     // add multiple keys


GET         name                                    // get a key

MGET        name age                                // get multiple keys at once


INCR        age                                     // increment the selected key's value by 1

INCRBY      age 5                                   // increment the selected key's value by the specified integer amount

INCRBYFLOAT age 5.5                                 // increment the selected key's value by the specified decimal amount


DECR        age                                     // decrement the selected key's value by 1

DECRBY      age 5                                   // decrement the selected key's value by the specified integer amount


DEL         myKey1 myKey2                           // delete the selected keys



Node


var redis = require("redis");                       // npm install redis

var client = redis.createClient();                  // connect to the database

// ...                                              // submit commands as reuquired

client.quit();                                      // close the connection


client.set("myKey", "Hello world!");                // store a value


client.get("myKey", redis.print);                   // retrieve and print a value


client.get("myKey", function (err, replies) {       // retrieve a value with a custom callback function

    console.log(replies[0]);

});


client.mget(                                        // retrieve multiple values at once

    ["myKey", "myOtherKey"],

    function (err, replies) {

        console.log(replies.length);

        console.log(replies[0]);           

        console.log(replies[1]);                    // the replies element for unmatched keys is set to null

    }

);


client.incr       ("myKey"     );                   // increment the key's value by 1

client.incrby     ("myKey", 5  );                   // increment the key's value by the given integer amount

client.incrbyfloat("myKey", 5.5);                   // increment the key's value by the given decimal amount


client.decr       ("myKey"     );                   // decrement the key's value by 1

client.decrby     ("myKey", 5  );                    // decrement the key's value by the given integer amount
