Box shadows


box-shadow:      2px 2px                            /* horizontal and vertical offset                            */

                 10px                               /* blur radius                                               */

                 0px                                /* optional: increase the border size before blurring        */

                 red                                /* optional: colour                                          */

                 inset;                             /* optional: cast the shadow into the inside of the element  */


box-shadow:      0 3px 8px 0 #ccc, 0 3px 8px 0 red; /* draw multiple shadows at once                             */



Text shadows


text-shadow:     {none                              /* disable text shadows                                      */

                 |2px 2px 1px black                 /* x_offset, y_offset, blur_radius, colour (optional)        */

                 |4px 4px 2px, -4px -4px 1px}       /* can pass multiple shadows at once                         */
