Declaring commands


<Window ...>

    <Window.CommandBindings>                                          <!-- commands are available in the whole window               -->

        <CommandBinding Command    = "ApplicationCommands.Copy"/>     <!-- standard command handled completely by the system        -->

        <CommandBinding Command    = "ApplicationCommands.Undo"       <!-- commands with a customised implementation                -->

                        CanExecute = "UndoCmdCanExecute"              <!-- called to determine if the command can be executed       -->

                        Executed   = "UndoCmdExecuted"/>              <!-- called to actually execute the command                   -->



<Button Command                   = "ApplicationCommands.Cut"         <!-- attach a commands to buttons, menu items, ...            -->

        CommandTarget             = "{Binding ElementName=textBox}"   <!-- redirect the command execution to another element        -->

        CommandManager.CanExecute = "CutCommandCanExecute"/>          <!-- attach custom delegates to the command  -->



Class MainWindow

    Public Sub New()

        InitializeComponent()

        Dim Binding As New CommandBinding(ApplicationCommands.Help)   ' create a new binding for the specific command

        AddHandler Binding.CanExecute, AddressOf CanHelpExecute       ' optionally register the event delegates

        AddHandler Binding.Executed, AddressOf HelpExecuted

        Me.CommandBindings.Add(Binding)                               ' register the command binding

    End Sub

End Class



Implementing commands


Private Sub CanExecute(sender As Object,                              ' indicate If the command could be executed right now

                       ea As CanExecuteRoutedEventArgs) 

    ea.CanExecute = True

End Sub


Private Sub Executed(sender As Object,                                ' perform the command sepcific actions

                     ea As ExecutedRoutedEventArgs)     

    ea.Handled = True                                                 ' prevent the event from further propagation

End Sub
