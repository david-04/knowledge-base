Formatting


color:           {red|transparent};                 /* set the font colour (or make it transparent)                  */


text-shadow:     {none                              /* disable text shadows                                          */

                 |2px 2px 1px black                 /* x_offset, y_offset, blur_radius, colour (optional)            */

                 |4px 4px 2px, -4px -4px 1px}       /* can pass multiple shadows at once                             */


text-decoration: {none|underline|overline           /* add under-/overlines or strike-throughs (or remove them)      */

                 |line-through};                    /* can pass multiple values (space-separated) at once            */


font-style:      {normal|italic|oblique};           /* oblique is similar to italic                                  */


font-weight:     {normal|bold                       /* absolute font-weigth                                          */

                 |lighter|bolder                    /* relative font-weight                                          */

                 |100...900};                       /* 100 (thin) ... 400 (normal) ... 900 (heavy)                   */


font-size:       {120%|2em|larger|smaller           /* relative font size                                            */

                 |9px|2rem|xx-small|x-small|small   /* absolute font size                                            */

                 |medium|large|x-large|xx-large};


line-height:     {1.2|1.2em|120%|normal};           /* 1.2 == 1.2em == 120%                                          */



Transformations


text-transform:  {none|uppercase|lowercase          /* change the text content's capitalisation                      */

                 |capitalize}


font-variant:    {normal|small-caps};               /* small-caps shows capital letters only                         */



Fonts


font-family:     "MyFont",                          /* switch to a particular font                                   */

                 serif,sans-serif,cursive,          /* ... or select a predefined one                                */

                 fantasy,monospace;                 /* (pass multiple fonts in the preferred order/priority)         */


font:            italic                             /* shortcut: font-style                                          */

                 small-caps                         /*           font-variant                                        */

                 bold                               /*           font-weight                                         */

                 1.2em/1.4em                        /*           font-size/line-height                               */

                 "Arial", serif                     /*           font-family (can include multiple fonts)            */


font:            {caption|icon|menu|message-box     /* use a predefined font style                                   */

                 |small-caption|status-bar};



Alignment


direction:       {ltr|rtl};                         /* text direction (is usually set via the HTML "dir" attribute)  */


direction:       {left|right|center|justify         /* horizontal text alignment                                     */

                 |start|end};                       /* (start=left for ltr and right for rtl)                        */


text-indent:     1.25em;                            /* indent the first line (only) */


letter-spacing:  {normal|5px};                      /* set the space between single characters                       */


word-spacing:    {normal|5px};                      /* set the space between distinct words                          */



Wrapping


white-space:     {normal                            /* white space is collapsed and wrapped (default)                */

                 |pre                               /* neither collapse nor break lines                              */

                 |nowrap                            /* collapse but don't break lines                                */

                 |pre-wrap                          /* don't collapse but break lines                                */

                 |pre-line};                        /* collapse and wrap                                             */

                                                   

overflow-wrap:   {normal|break-word};               /* break words that are longer than the whole line               */
