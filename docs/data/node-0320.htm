Output parameters


static void Add(int x1, int x2, out int sum)

{

    sum = x1 + x2;

}


Add(1, 2, out sum);



Call by reference


static void Square(ref int x)

{

    x *= 2;

}


Square(ref x);



Variable argument list


static string Concat(params Object[] values)

{

    StringBuilder result = new StringBuilder();

    foreach (Object obj in values)

        result.Append(obj.ToString()).Append(" ");

    return result.ToString();

}


Concat(1, 2, "abc");



Optional and named parameters


static string Greet(string firstName, string lastName = "")

{

    return "Hello " + firstName + " " + lastName;

}


Greet(firstName: "David")

Greet(lastName:  "Hofmann", firstName: "David")
