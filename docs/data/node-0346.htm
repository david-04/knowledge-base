Transitions


                                                    /* transitions work with pseudo-elements (:hover) and when          */

                                                    /* dynamically assigning classes to elements                        */


transition-duration:         1.0s;                  /* duration of the transition from start to end                     */


transition-delay:            0.5s;                  /* wait the given duration before starting the transition           */

transition-delay:           -0.25s;                 /* wait for 1/4s and then jump right to the 1/4s keyframe           */


transition-timing-function: {linear                 /* constant speed from start to end                                 */

                            |ease|ease-in-out       /* accelerate and decelerate                                        */

                            |ease-in|ease-out       /* accelerate (or decelerate) only                                  */

                            |step-start             /* jump immediately to the end                                      */

                            |step-end};             /* freeze and jump to the end at the end of the transition duration */


transition-property:        color,transform;        /* properties to animate (default: all supported properties)        */

                                                    /* all                      font-size           outline-color       */

                                                    /* background-color         font-weight         outline-width       */

                                                    /* background-position      height              padding-bottom      */

                                                    /* border-bottom-color      left                padding-left        */

                                                    /* border-bottom-width      letter-spacing      padding-right       */

                                                    /* border-left-color        line-height         padding-top         */

                                                    /* border-left-width        margin-bottom       right               */

                                                    /* border-right-color       margin-left         text-indent         */

                                                    /* border-right-width       margin-right        text-shadow         */

                                                    /* border-spacing           margin-top          top                 */

                                                    /* border-top-color         max-height          vertical-align      */

                                                    /* border-top-width         max-width           visibility          */

                                                    /* bottom                   min-height          width               */

                                                    /* clip                     min-width           word-spacing        */

                                                    /* color                    opacity             z-index             */




transition:                 all 1s ease 0.25s;      /* shortcut: property, duration, timing-function, delay             */

transition:                 color 1s, transform 2s; /*           can configure multiple animations at once              */



Keyframe animations


@keyframes myAnimation {   from { opacity: 1.0; }   /* define a named keyframe animation (as a standalone CSS rule)     */

                           50%  { opacity: 0.5; }   /* set as many properties as required                               */

                           to   { opacity: 0.0; }}


animation-name:            {myAnimation|none};      /* attach or remove the animation to/from the target element        */


animation-duration:         1.0s;                   /* duration of the animation (from start to end) */


animation-delay:            0.5s;                   /* wait the given duration before starting the animation            */

animation-delay:           -0.25s;                  /* wait for 1/4s and then jump right to the 1/4s keyframe           */


animation-timing-function: {linear|                 /* constant speed from start to end                                 */

                           |ease|ease-in-out        /* accelerate and decelerate                                        */

                           |ease-in|ease-out        /* accelerate (or decelerate) only                                  */

                           |step-start              /* jump immediately to the end                                      */

                           |step-end};              /* freeze and jump to the end at the end of the transition duration */


animation-iteration-count: {1|2|3|...|infinite};    /* repeat the given number of times (or infinitley)                 */


animation-direction:       {normal                  /* run from start to end and jump back                              */

                           |reverse                 /* run from end to start and jump back                              */

                           |alternate               /* run start -> end -> start -> end -> start -> ...                 */

                           |alternate-reverse};     /* run          end -> start -> end -> start -> end -> ...          */


animation-fill-mode:       {none                    /* before and after the animation the non-animated state is shown   */

                           |forwards                /* at the end the last animation frame is preserved                 */

                           |backwards               /* the first frame is shown before (and until) the animation starts */

                           |both};                  /* combination of forwards+backwards                                */


animation-play-state:      {paused|running};        /* pause or resume the animation                                    */


animation:                 myAnimation              /* animation-name                                                   */

                           2s                       /* animation-duration                                               */

                           ease                     /* animation-timing-function                                        */

                           0.5s                     /* animation-delay                                                  */

                           infinite                 /* animation-iteration-count                                        */

                           alternate                /* animation-direction                                              */

                           forwards                 /* animation-fill-mode                                              */

                           paused;                  /* animation-play-state                                             */
