Menus


ContextMenu                                         // a pop-up menu

MenuBar                                             // a window's main menu

Menu                                                // an entry in the main menu (e.g. File, Edit, Help)


MenuItem                                            // a menu item (e.g. Open, Save As, Close)

CheckMenuItem

RadioMenuItem

SeparatorMenuItem



Menu bar


MenuBar menuBar = new MenuBar();                    // create a menu bar

borderPane.setTop(menuBar);


Menu fileMenu = new Menu("_File");                  // add main menus

menuBar.getMenus().add(fileMenu);


MenuItem fileNew = new MenuItem("_New");            // add menu items

fileMenu.getItems().add(fileNew);


Menu fileRecent = new Menu("_Recent");              // add submenus

fileMenu.getItems().add(fileRecent);


MenuItem recent1 = new MenuItem("1");

fileRecent.getItems().add(recent1);





fileNew.setOnAction((ActionEvent event) ->          // add an event handler

    label.setText("New")

);


MenuItem fileNew = new MenuItem("_New");            // mark one character as the mnemonic

fileNew.setMnemonicParsing(true);                   // disable the special meaning of the underscore


fileNew.setAccelerator(                             // assign an accelerator (shortcut)

        KeyCombination.keyCombination(

                "shortcut+N"));



Context menu


ContextMenu menu = new ContextMenu();               // create and populate a context menu

menu.getItems().addAll(

    new MenuItem("Copy"),

    new MenuItem("Paste")

);


label.setContextMenu(menu);                         // attach the context menu to a single component


scene.setOnContextMenuRequested(                    // a context menu can also be attached to a whole scene

    (ContextMenuEvent ae)

    -> menu.show(scene.getRoot(),                   // ... and must be mounted onto the scene's root pane

                 ae.getScreenX(),

                 ae.getScreenY())

);



Toolbars


ToolBar toolbar = new ToolBar(                      // create a toolbar

    new Button("1"), new Button("2")                // add buttons to it

);

borderPane.setTop(toolbar);                         // add the toobar to the root pane
