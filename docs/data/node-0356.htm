Web fonts


@font-face {                                        /* register a web font                                                */

    font-family: "MyFont";                          /* use as: font-family: "MyFont", "Arial", sans-serif;                */

    src        : url("../fonts/MyFont.eot");        /* this is for IE<9 (all other browsers use the next line instead)    */

    src        : local("MyFont"),                   /* use the local font if it is already installed                      */

                 local("MyOtherFont"),             

                 url("../fonts/MyFont.woff");       /* and otherwise load the given font file                             */

}



Formats


woff ... Web Oben Font Format                       /* expectd to be the future, supported by all browsers (except IE<9)  */

otf .... OpenType Font                              /* only supported by IE but also the only option for IE<9             */

ttf .... TrueType Font                              /* limited support in IE9+                                            */

eot .... Embedded OpenType

svg .... Scalable Vector Graphics                   /* slow and poor quality                                              */


Composition 1: WOFF                                 /* for new browsers                                                   */

               EOT                                  /* for IE < 9                                                         */

               TTF or OTF                           /* for Android and older Non-IE browsers                              */

               SVG (optional)                       /* for older iPads and iPhones                                        */


Composition 2: TTF or OTF                           /* for new browsers                                                   */

               EOT                                  /* for IE < 9                                                         */
