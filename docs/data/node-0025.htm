Hash tables


HSET         myHashtable "name" "David"             // add a key-value pair to a hashtable

HMSET        myHashtable "name" "David" "age" 38    // add multiple key-value pairs to a hashtable

HINCRBY      myHashtable "age" 1                    // increment a hash item's value by the given integer amount

HINCRBYFLOAT myHashtable "age" 1.1                  // increment a hash item's value by the given decimal amount


HGET         myHashtable "name"                     // retrieve a hash item's value

HMGET        myHashtable "name" "age"               // retrieve multiple hash items' values

HGETALL      myHashtable                            // retrieve all hash items (as array [key1, value1, key2, value2, ...])

HKEYS        myHashtable                            // retrieve all hash keys (without value)

HVALS        myHashtable                            // retrieve all hash values (without key)


HDEL         myHashtable "name"                     // delete a hash item



Node


client.hmset(                                       // add hash items

    "myHashtable",

    "name", "David",

    "age", 38

);


client.hincrby("myHashtable", "age", 1);            // increment a hash item's value


client.hgetall(                                     // retrieve all hash items

    "myHashtable",

    function (err, replies) {

        console.log("Name:", replies["name"]);

        console.log("Age:", replies["age"]);

    }

);
