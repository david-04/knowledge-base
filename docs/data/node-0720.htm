Numbers


<xsl:number

    value              = "myElement"                <!-- XPath returning the number to be formatted                        -->

                                                    <!-- if this attribute is not set, then the number formatted is the    -->

                                                    <!-- position of the current node within a selection of other nodes:   -->

    count              = "myReferenceElement"       <!-- count only nodes that match this XPath                            -->

    from               = "myRootElement"            <!-- the position counting will start from this node                   -->

    level              = "{single|multiple|any}"    <!-- single: count only preceding siblings of the current node         -->

                                                    <!-- any: count preceding nodes independent of hierarchies             -->

                                                    <!-- multiple: count nested: 1, 1.1, 1.2, 2, 2.1, 2.2.1, ...           -->

    format             = "{1|01|a|A|i|I}"           <!--  1 => 1 2 3   |   a => a b c   |   i => i ii iii                  -->

                                                    <!-- 01 => 1 2 3   |   A => A B C   |   I => I II III                  -->

    grouping-separator = ","                        <!-- how to separate groups (of thousands)                             -->

    grouping-size      = "3"                        <!-- group size (add a separater after every 3 digits)                 -->

    lang               = "en-au"                    <!-- the language alphabet used for the numbering                      -->

    letter-value       = "{alphabetic|traditional}"

/>





<xsl:decimal-format

    name               = "myNumberFormat"           <!-- symbolic name (used by the format-number() function)              -->

    decimal-separator  = "."

    grouping-separator = ","

    infinity           = "Infinity"

    minus-sign         = "-"

    NaN                = "NaN"

    percent            = "%"

    per-mille          = "‰"

    zero-digit         = "0"

    digit              = "#"                        <!-- placeholder used in format strings                                -->

    pattern-separator  = ";"                        <!-- separates positive and negative sub-formats in a format pattern   -->

/>





<xsl:value-of select="format-number(123)"/>         <!-- format a numeric value                                            -->

<xsl:value-of select="format-number(123, #0.#)"/>   <!-- optionally apply a format string:                                 -->

                                                    <!-- 0 ... a digit including leading and trailing zeros                -->

                                                    <!-- # ... a digit but no leading or trailing zeros                    -->

                                                    <!-- . ... decimal separator                                           -->

                                                    <!-- , ... grouping seprator (groups of thousands)                     -->

                                                    <!-- ; ... positive and negative subformats (e.g. ##0.##; -##0.##)     -->

                                                    <!-- - ... negative prefix                                             -->

                                                    <!-- % ... multiply by 100 and show with a percent sign                -->

                                                    <!-- %. ... multiply by 1,000 and show with per mille sign             -->

                                                    <!-- X ... any other character can be used in the prefix or suffix     -->

                                                    <!-- ' ... quote special characters                                    -->
