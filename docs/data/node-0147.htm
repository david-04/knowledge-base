INSERT


create.insertInto(

    AUTHOR,                                             // the table to insert to

    AUTHOR.ID, AUTHOR.FIRST_NAME, AUTHOR.LAST_NAME      // the fields to populate

)

.values(100, "Celine", "Dion")                          // multipe values can be inserted at once

.values(101, "Sarah", "Brightman")

.execute();


create.insertInto(AUTHOR).defaultValues().execute();    // create a record with default values only


create.insertInto(

    AUTHOR, AUTHOR.ID, AUTHOR.FIRST_NAME

).values(

    100, defaultValue(FIRST_NAME.ID)                    // use the default value for some fields only

)         


create.insertInto(ENDCUSTOMERS)

      .set(ENDCUSTOMERS.ID, 100l)                       // set fields individually

      .set(ENDCUSTOMERS.CITY, "Melbourne")

      .newRecord()                                      // create multiple records in one go

      .set(ENDCUSTOMERS.ID, 100l)

      .set(ENDCUSTOMERS.CITY, "Geelong")

      .execute();


create.insertInto(ENDCUSTOMERS)

      .select(selectFrom(ENDCUSTOMERS).where(...)       // insert from a select

      .execute();


create.insertInto(AUTHOR, AUTHOR.ID, AUTHOR.LAST_NAME)

      .values(3, "Smith")

      .onDuplicateKeyUpdate()                            // separate UPSERT/UPDATE commands for pre-existing records

      .set(AUTHOR.LAST_NAME, "Smith")

      .execute();


create.insertInto(AUTHOR, AUTHOR.ID, AUTHOR.LAST_NAME)

      .values(3, "Smith")

      .onDuplicateKeyIgnore()                            // ignore the whole command if the record already exists

      .execute();


create.insertInto(AUTHOR, AUTHOR.NAME)

      .values("John Smith")

      .returning(AUTHOR.ID)                                       // in the same transaction, select and return the auto-assigned values

      .fetchOne();
