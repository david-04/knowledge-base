Iterations


outer: for (var i = 0; i < 10; i++) {               // label the loop for reference purposes

    for (var j = 0; j < 10; j++) {

        continue outer;                             // break and continue can reference the label

    }

}


for (var propName in window) {                      // iterate over all properties of the object

    document.write(propName);                       // the property name as returned as string (in random order)

}



Switch statement


switch (colour) {                                        // like in C/C++/Java/C# but works for strings as well

    case "Red":

        break;                                     // break required

    case "Green":

        break;

    default:

}


switch (true) {                                    // works without variable as well

    case 10 < variable && variable <= 20:          // access any variable that's in the current scope

        break;

    default:

}
