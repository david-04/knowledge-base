Setup


https://github.com/angular/quickstart               // download the quickstart seed from the Angular web site


> npm install                                       // download the required packages


> npm start                                         // start the development tools which does the following

                                                    // start a local web server to host the application

                                                    // watch for file changes and automatically recompile the application

                                                    // reload the app in the browser after each re-compilation



Folder structure


[root]

|

+-- index.html                                      // the application's main HTML file

|

+-- app                                             // all components that make up the application

|   |

|   +-- app.component.ts                            // the root component (that hosts/embeds all [child] views)

|   +-- app.module.ts                               // the root module that imports all objects required by the application

|   +-- app-routing.module.ts                       // fundamental routing rules (like default URL and 404)

|   +-- main.ts                                     // the main program that assembles and starts the application

|   |

|   +-- feature-1                                   // each feature (area of the application) has its own subfolder

|   |   |

|   |   +-- dashboard.component.ts                  // components are have a code file and optionally HTML and CSS templates

|   |   +-- dashboard.component.html

|   |   +-- dashboard.component.css

|   |   |

|   |   +-- feature-1.module.ts                     // the feature module with optional routing tables

|   |   +-- feature-1-routing.module.ts

|   |   |

|   |   +-- data-model.ts                           // data model classes don't have a special naming convention

|   |   |

|   |   +-- rest.service.ts                         // other items like services, directives, pipes, ...

|   |   +-- currency.pipe.ts                        // other items like services, directives, pipes, ...

|   |   +-- highlight.directive.ts                  // other items like services, directives, pipes, ...

|   |

|   +-- feature-2

|       |

|       +-- [...]

|

+-- e2e                                             // end-to-end test cases

|

+-- node_modules                                    // Angular and supporting libraries
