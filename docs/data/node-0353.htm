Tables


table {


    border:          1px solid black;               /* draw a frame around the table (won't create gridlines)      */


    border-spacing:  3px;                           /* set the spacing between cells                               */


    border-collapse: collapse;                      /* join adjacent cell borders (implies border-spacing:0)       */


    caption-side:    {top|bottom};                  /* where to place the caption                                  */


    empty-cells:     {show|hide};                   /* show or hide empty cells                                    */


    table-layout:    {auto                          /* auto-adjust the column width based on the cells' contents   */

                     |fixed};                       /* use the specified column widths only                        */

}


tr {


    visibility:      collapse;                      /* hide a table row without affecting the table layout         */

}


td {


    border:         1px solid black;                /* draw a border around each cell (gridlines)                  */


    text-align:     {left|right|center|justify};    /* horizontal alignment of the content                         */


    vertical-align: {top|middle|bottom};            /* vertical alignment of the content                           */


    padding:        5px;                            /* eqivalent to cellpadding="5"                                */

}
