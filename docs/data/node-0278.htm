Customising controls through skins


<asp:Button BackColor="#cccccc" ForeColor="#308462" runat="server" />   <!-- set the appareance of the button control     -->

<asp:Button CssClass="MyButton" runat="server" />                       <!-- better: redirect to a CSS class instead      -->

<asp:Button SkinId="MySkin" BackColor="Green" runat="server" />         <!-- define a style with an explicit SkinID       -->


<asp:Button ID="Submit" SkinId="MySkin" runat="server" Text="Submit" /> <!-- explicitely reference the SkinID in the ASPX -->
