Embedded code


<script src   = "javascript.js"                     <!-- file reference (unless code is embedded)                             -->

        type  = "text/javascript"                   <!-- can be left out as this is the default value as well                 -->

        defer = "defer"                             <!-- run in parallel while loading/displaying the page                    -->

        async = "async" >                           <!-- run after loading/displaying the page                                -->

    // <![CDATA[

        ...                                         <!-- for XHTML (only) the code should be wrapped in a CDATA section       -->

    // ]]>

</script>                                           <!-- the SCRIPT element requires a mandatory and explicit closing tag     -->



Loading code and styles dynamically


<script type="text/javascript">                     // load a dynamically selected JavaScript file

    document.write(

        '<script type="text/javascript" '

        +       'src="file.js">'

        +'<\/script>');                             // must be masked to not confuse the parser

</script>


var script = document.createElement("script");

script.type = "text/javascript";

script.src = "client.js";

document.body.appendChild(script);


var link = document.createElement("link");

link.rel = "stylesheet";

link.type = "text/css";

link.href = "styles.css";

var head = document.getElementsByTagName("head")[0];

head.appendChild(link);



Strict mode


"use strict";                                       // execute in strict mode (applicable globally or within a function only)



Dynamic code evaluation


var message = "hello world";

eval("alert(message)");                             // evals can reference variables and functions from outside


eval("function sayHello() { alert('hello'); }");    // evals can define new functions (not in strict mode)

sayHello();


eval("var name = 'David';");                        // evals can define variables (not in strict mode)

alert(name);



Operators


~                                                  // bitwise NOT

& |                                                // bitwise and/or

^                                                  // bitwise XOR

<< >> >>>                                          // left and right-shift (signed and unsigned)

== !=                                                       // equal/not-equal value (conversion is applied)

=== !==                                            // equal/not-equal value AND type (no implicit conversion)
