Setup


PM> Install-Package Microsoft.AspNet.Web.Optimization


<configuration>                                     <!-- in ~/Views/web.config                                          -->

  <system.web.webPages.razor>

    <pages>

      <namespaces>

        <add namespace="System.Web.Optimization"/>  <!-- add the optimization package's namespace to all views          -->



Defining bundles


namespace MVC                                       // must be in the project's root namespace (place in ~/App_Start)

{

    public class BundleConfig

    {

        public static void RegisterBundles(         // define all the bundles

                        BundleCollection bundles)

        {

            bundles.Add(new StyleBundle(            // use StyleBundle for CSS and ScriptBundle for JavaScript

                "~/Content/css"

            ).Include(

                "~/Content/*.css"                   // use wildcards to include all files (order will be random)

            ));


            bundles.Add(new ScriptBundle(

                "~/bundles/javascript"

            ).Include(

                "~/Scripts/jquery-{version}.js",    // use {version} as a placeholder (that MVC will resolve)

                "~/Scripts/jquery.validate.js"

            ));

        }

    }

}


public class MvcApplication : HttpApplication       // reference the bundle definition from the Global.asax class

{

    protected void Application_Start()

    {

        BundleConfig.RegisterBundles(

            BundleTable.Bundles);

    }

}



Including bundles


@Styles.Render("~/Content/css")             <!-- create a <style> element referencing the bundle -->

@Scripts.Render("~/bundles/javascript")     <!-- create a <script> element referening the bundle -->
