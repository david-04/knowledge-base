Detail view controls


<asp:FormView    />                                                    <!-- simple tab format to edit one record            -->

<asp:DetailsView />                                                    <!-- customisable layout to edit one record          -->



DetailsView


<asp:DetailsView ID               = "CustomerDetailsView"              <!-- a regular table in key-value format             -->

                 runat            = "server"

                 DataSourceID     = "SqlDataSourceDetails"             <!-- the data source (if one is used)                -->

                 AutoGenerateRows = "false"                            <!-- auto-generate fields for all entity attributes  -->

                 DataKeyNames     = "CustomerId"                       <!-- the primary key field of the entity             -->

                 AllowPaging      = "True"                             <!-- browsing when multiple results are returned     -->

                 DefaultMode      = "Insert">                          <!-- insert or edit (update)                         -->

    <Fields>

        <asp:BoundField   DataField        = "FirstName"               <!-- define the require fields                       -->

                                 HeaderText       = "First name" />

        <asp:BoundField   DataField        = "LastName"

                          HeaderText       = "Last name" />

        <asp:CommandField ShowInsertButton = "True"                    <!-- add the relevant insert/update/cancel buttons   -->

                          ShowCancelButton = "False" />

    </Fields>

    <PagerSettings Position      = "TopAndBottom"                      <!-- customise the page browsing                     -->

                   FirstPageText = "First page"

                   LastPageText  = "Last page"

                   Mode          = "NextPreviousFirstLast" />

</asp:DetailsView>



FormView


<asp:FormView ID           = "DetailsView1"                            <!-- fully customisable free-form layout             -->

              runat        = "server"

              DataSourceID = "SqlDataSourceDetails"                    <!-- the data source (if one is used)                -->

              DataKeyNames = "CustomerId"                              <!-- the primary key field of the entity             -->

              DefaultMode  = "Insert">                                 <!-- insert or edit (update)                         -->

    <ItemTemplate>                                                     <!-- the template for the read-only view             -->

        <asp:Label ID="FullName" runat="server"

            Text='<%# Eval("LastName")+","+Eval("FirstName") %>' />

    </ItemTemplate>

    <EditItemTemplate>                                                 <!-- the template for editing existing items         -->

        <asp:TextBox ID="LastNameEdit" runat="server"

            Text='<%# Bind("LastName") %>' />,                      

        <asp:TextBox ID="FirstNameEdit" runat="server"

            Text='<%# Bind("FirstName") %>' />

    </EditItemTemplate>

    <InsertItemTemplate>                                               <!-- template for inserting new items                -->

         <!-- ... -->

    </InsertItemTemplate>

</asp:FormView>
