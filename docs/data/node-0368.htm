Elements


*                                                   /* universal selector                                                           */

table                                               /* type selector                                                                */

#myElementId                                        /* ID selector                                                                  */

.myClass                                            /* class selector                                                               */

.myClass.myOtherClass                               /* must have both classes at the same time                                      */


h1,h2,h3                                            /* union - any of the given elements                                            */

div     p                                           /* descendants - <p> can be anywhere within <div>                               */

div   > p                                           /* children - <p> must be a direct child of <div>                               */

table + p                                           /* CSS3: adjacent siblings - <p> must directly follow a <table>                 */

table ~ p                                           /* CSS3: general siblings - <p> must follow a <table> within the same parent    */



Attributes


input[type="text"]                                  /* elements must have the given attribute values                                */


[attribute               ]                          /* must have the given attribute (no matter what its value is)                  */

[attribute  = "value"    ]                          /* the attribute must have the given value                                      */

[attribute != "value"    ]                          /* the attribute is either not set at all or does otherwise have another value  */

[attribute ^= "prefix"   ]                          /* the attribute's value must start with the given prefix                       */

[attribute |= "prefix"   ]                          /* the attribute's value must either be "prefix" or start with "prefix-"        */

[attribute *= "substring"]                          /* the attribute's value must contain the given substring                       */

[attribute ~= "word"     ]                          /* the attribute's value must contain the given word (separated by whitespace)  */

[attribute $= "suffix"   ]                          /* the attribute's value must end with the given suffix                         */

[attribute  =...         ][attribute2=...]          /* element must match all the specified attribute rules                         */



Dynamic state


:active                                             /* element is active (i.e. mouse button is currently pressed down)              */

:focus                                              /* element has the focus                                                        */

:hover                                              /* while the mouse is over the element                                          */

:checked                                            /* must be selected or checked (radio buttons anc check boxes)                  */

:disabled                                           /* must be disabled                                                             */

:enabled                                            /* must be enabled                                                              */


:link                                               /* unvisited link                                                               */

:visited                                            /* visited link                                                                 */

:target                                             /* the <a id="..."> while <a href="#..."> is being pressed                      */



Hierarchy


:root                                               /* the document root element (usually the <html> node)                          */

:empty                                              /* element must not have any children (neither nodes nor text)                  */


:first-child                                        /* the first child of its parent                                                */

:last-child                                         /* the last child of its parent                                                 */

:first-of-type                                      /* the first child of its own type (parents can have many first children)       */

:last-of-type                                       /* the last child of its own type (parents can have many first children)        */


:nth-child(index)                                   /* the index'th child within its parent container (first element == 1)          */

:nth-child(odd)                                     /* every other child (1,3,5...)                                                 */

:nth-child(even)                                    /* every other child (2,4,6...)                                                 */

:nth-last-child(index)                              /* like nth-child but couting backwards from the end (last index == 1)          */


:nth-of-type(index)                                 /* nth-child of the same element/tag type (first index == 1)                    */

:nth-last-of-type(index)                            /* nth-last child of the same element/tag type (last index == 1)                */

:only-child                                         /* element must be its parent's only child                                      */

:only-of-type                                       /* the only child of its own type (parents can have many only children)         */



Assorted


:not(selector)                                      /* must not match the given selector                                            */


:before { content : "\201C"; }                      /* insert content before an element                                             */

:after  { content : "\201D"; }                      /* insert content after an element                                              */


::first-line                                        /* the first line of text (in a paragraph)                                      */

::first-letter                                      /* the fisrt letter (of a paragraph)                                            */


:lang(fr)                                           /* natural language (e.g. <p lang="fr">)                                        */
