JavaScript API


$(".navbar").affix({                                  // configure an affix element

    offset: {

        top: 10

    }

});


onaffix.bs.affix                                      // before an element becomes affixed

onaffixed.bs.affix                                    // after an element has been affixed

onaffix-top.bs.affix                                  // before an element becomes affixed to the top of the screen

onaffixed-top.bs.affix                                // after an element has been affixed to the top of the screen

onaffix-bottom.bs.affix                               // before an element becomes affixed to the bottom of the screen

onaffixed-bottom.bs.affix                             // after an element has been affixed to the bottom of the screen
