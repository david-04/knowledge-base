Usage


Using myObject As New SomeClass()                           ' automatically have Dispose() called at the end of the Using block

    ' ...

End Using



Dim myObject As SomeClass = Nothing                         ' alternative: Dispose() manually

Try                                                         ' wrapped by a try block to ensure that Dispose() is always called

    myObject = New SomeClass()

Finally

    myObject?.Dispose()                                     ' explicitely dispose of the object

End Try



Disposable


Class SomeClass

    Implements IDisposable                                  ' implement IDisposable


    Private hasBeenDisposed As Boolean                      ' only dispose once


    Public Sub Dispose() Implements IDisposable.Dispose

        If Not hasBeenDisposed Then

            ' call properties' and fields' Dispose()

            ' relase and clean up unmanaged resources

        End If

        hasBeenDisposed = True                              ' prevent re-disposal

    End Sub


End Class



Disposable and finalisable


Class SomeClass

    Implements IDisposable                                  ' implement IDisposable


    Private hasBeenDisposed As Boolean                      ' only dispose once


    Protected Overrides Sub Finalize()                      ' destructor inherited from Object, invoked by GC

        Try

            Dispose(False)                                  ' do the actual disposal work

            MyBase.Finalize()                               ' finalise the base class as well

        Catch                                               ' swallow all exceptions, otherwise the application crashes

        End Try

    End Sub


    Public Sub Dispose() Implements IDisposable.Dispose

        Dispose(True)                                       ' do the actual disposal work

        GC.SuppressFinalize(Me)                             ' no need for the GC to call the destructor

    End Sub


    Protected Overridable Sub Dispose(disposing As Boolean)

        If Not hasBeenDisposed Then                         ' only dispose once

            If disposing Then

                ' call properties' and fields' Dispose()

            End If

            ' relase and clean up unmanaged resources

            hasBeenDisposed = True

        End If

    End Sub


End Class
