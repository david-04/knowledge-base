Create actions


function myActionCreator(name) {                    // an action creator builds and returns an action object (but does not dispatch it)

    return {

        type: 'MY_ACTION',                          // must have a property named "type" which classifies the action type

        user: name,                                 // payload data can be incorporated into the action object as required

        data: {id: 1, message: 'Hi'}

    }

}



Dispatch actions


store.dispatch(myActionCreator('David'));           // actions are dispatched to/through the store


const myBoundActionCreator = name =>                // ... or create a bound action creator

    store.dispatch(myActionCreator(name));



Bind action creators


import { bindActionCreators } from 'redux';


var boundCreators = bindActionCreators(             // bind action creators to a store

    myActionCreator, dispatch                       // can pass an object with multiple action creators instead of the single myActionCreators

);                                                  // returns an object (not a single function)


<MyComponent {...boundCreators} />                  // pass the bound action creators down the chain



Generic action creators


function makeActionCreator(type,                    // a generic factory function for action creators

                           ...argNames) {           // receives the property names of the action (excluding the "type" property)

    return function (...args) {                     // returns the create function (vargs, can't be typed in TypeScript)

        let action = { type };

        argNames.forEach((arg, index) => {

            action[argNames[index]] = args[index]

        });

        return action;

    }

}


const ADD_TODO = 'ADD_TODO';

const EDIT_TODO = 'EDIT_TODO';


export const addTodo = makeActionCreator(           // generate the action creators

    ADD_TODO, 'text'

);

export const editTodo = makeActionCreator(

    EDIT_TODO, 'id', 'text'

);
