Timers


timeout  = setTimeout(function () { }, 1000);       // schedule a one-off timer

clearTimeout(timeout);                              // cancel a one-off timer


interval = setInterval(function () { }, 1000);      // schedule a repeating timer

clearInterval(interval);                            // cancel a repeating timer


process.nextTick(function () {  });                 // schedule a one-off timer during the next event loop (i.e. ASAP)



Self-rescheduling timers


(function schedule() {                              // create an anoymous context

    setTimeout(function() {

        console.log('hello world');                 // do the actual work

        schedule();                                 // reschedule the next run by accessing the anoynmous context wrapper function

    }, 1000);

}());
