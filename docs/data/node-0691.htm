Elements and hierarchies


<?xml encoding="utf-8" ?>                           <!-- external DTDs need a text declaration (version="1.0" is optional) -->


<!myElement myElement (child?)>                     <!-- optional                                                          -->

<!myElement myElement (child+)>                     <!-- repeatable                                                        -->

<!myElement myElement (child*)>                     <!-- optional and repeatable                                           -->


<!myElement myElement (child1, child2)>             <!-- sequence (must be in this order)                                  -->

<!myElement myElement (child1 | child2)>            <!-- choice (either or)                                                -->

<!myElement myElement (child1, (child2 | child3)+)> <!-- combined sequence/choice/repetition                               -->


<!myElement myElement EMPTY>                        <!-- no charater content and no children                               -->

<!myElement myElement (#PCDATA)>                    <!-- character data (but no children)                                  -->

<!myElement myElement (#PCDATA | child)*>           <!-- mixed content (character data + children)                         -->

<!myElement myElement ANY>                          <!-- any node that has been declared in the DTD                        -->

                    attribute2 CDATA #IMPLIED>


<!ATTLIST myElement xml:lang NMTOKEN #IMPLIED>      <!-- when using <myElement xml:lang="en-au">, then the attribute must  -->

                                                    <!-- be declared explictely                                            -->



Default values


#REQUIRED                                            <!-- mandatory                                                        -->

#IMPLIED                                             <!-- optional                                                         -->

#FIXED "value"                                       <!-- constant "value" (can't use any other value in the XML)          -->

"default value"                                      <!-- default value (can be overridden in the XML)                     -->
