GridView


<asp:GridView ... DataSourceID           = "MyDataSource"      <!-- the data source (if it is used)                              -->

                  AllowPaging            = "True"              <!-- activate pagination                                          -->

                  PageSize               = "10"                <!-- show 10 records on each page                                 -->

                  AllowSorting           = "True"              <!-- allow sorting by clicking on the table header                -->

                  DataKeyNames           = "CustomerId"        <!-- the entity's primary key field(s)                            -->

                  AutoGenerateColumns    = "False"             <!-- only show columns explicitely defined below                  -->

                  AutoGenerateEditButton = "False"             <!-- add an edit (or insert or update) button automatically       -->

                  EmptyDataText          = "No data found"     <!-- text to display when no data is returned                     -->

                  EnableSortingAndPagingCallbacks="true" />    <!-- perform paging and sorting via Ajax requests                 -->

    <Columns>

        <asp:CommandField ShowEditButton   = "True"            <!-- edit/delete/select buttons (or text links)                   -->

                          ShowDeleteButton = "True"

                          ShowSelectButton = "True" />

        <asp:BoundField   DataField        = "CustomerId"      <!-- regular data fields                                          -->

                          HeaderText       = "Customer No"

                          ReadOnly         = "True"

                          SortExpression   = "CustomerId"/>

        <asp:BoundField   DataField        = "FirstName"

                          HeaderText       = "First name"

                          SortExpression   = "FirstName" />

    </Columns>

    <EmptyDataTemplate>                                        <!-- element to display when no data is returned                  -->

        No data found

    </EmptyDataTemplate>

    <PagerSettings Position      = "TopAndBottom"              <!-- customise the page browsing                                  -->

                   FirstPageText = "First page"

                   LastPageText  = "Last page"

                   Mode          = "NextPreviousFirstLast" />

</asp:GridView>
