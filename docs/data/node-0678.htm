Class MainWindow

    Inherits Window


    Private Property button As Button = Nothing


    Public Sub New(windowTitle As String, height As Integer, width As Integer)


        Me.Title = windowTitle                                                  ' configure the window

        Me.WindowStartupLocation = WindowStartupLocation.CenterScreen

        Me.Height = height

        Me.Width = width


        Me.button = New Button()                                                ' configure a button

        Me.button.Content = "Test button"

        Me.button.Height = 25

        Me.button.Width = 100


        Me.Content = button                                                     ' attach the button to the window


    End Sub


End Class


Class Program

    Inherits Application


    <STAThread>

    Shared Sub Main(args() As String)

        Dim application As New Program()                                        ' create a New application instance

        AddHandler application.Startup, AddressOf AppStartUp                                       ' register startup And shutdown delegates

        AddHandler application.Exit, AddressOf AppExit

        application.Run()                                                       ' fire a startup Event

    End Sub


    Shared Sub AppExit(sender As Object, e As ExitEventArgs)                    ' cleanup actions, save configuration, ...

    End Sub


    Shared Sub AppStartUp(sender As Object, e As StartupEventArgs)

        Dim window As New MainWindow("My WPF App!", 200, 300)                   ' instantiate a New window

        window.Show()

    End Sub


End Class
