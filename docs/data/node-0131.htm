JavaScript API


$("#myModal").modal("show");                          // show the dialog

$("#myModal").modal("hide");                          // hide the dialog

$("#myModal").modal("toggle");                        // toggle between show and hide


$("#myModal").modal({                                 // show the dialog with the given configuration

    backdrop: "static",

    keyboard: false

});


$("#myModal").modal("handleUpdate")                   // realign the dialog (used after programmatically changing its content)


onshow.bs.modal                                       // before the dialog is made visible

onshown.bs.modal                                      // after the dialog has been made visible

onhide.bs.modal                                       // before the dialog is hidden

onhidden.bs.modal                                     // after the dialog has been hidden


$("#myModal").on("show.bs.modal", function(event) {

    var button = $(event.relatedTarget);              // retrieve data from the triggering element (button, link, ...)

    var titleData = button.data("title");

    $(this).find(".modal-title").text(titleData);     // update the dialog's content

});
