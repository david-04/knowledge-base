Drawing


Canvas canvas = new Canvas(100, 100);               // create a canvas component to draw on

pane.getChildren().add(canvas);                     // add the canvas to the container

GraphicsContext gc = canvas.getGraphicsContext2D(); // obtain a graphics context


gc.setStroke(Color.RED);                            // set the fore- and background colour

gc.setFill(Color.BLUE);


gc.strokeLine(x1, y1, x2, y2);                      // draw a line


gc.strokeRect(x, y, w, h);                          // draw a rectangle

gc.fillRect(x, y, w, h);


gc.strokeOval(x, y, w, h);                          // draw a circle

gc.fillOval(x, y, w, h);


gc.strokeText(text, x, y);                          // draw text
