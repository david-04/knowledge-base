Conditional processing


div {


    width: if($myVariable < 10, 50%, 300px);        /* the inline if-function works like the ?: operator in C/C++/Java     */


    @if $myVariable < 10 {                          /* when used as an directive, it also supports else-if branches        */

        color: red;

    } @else if $myVariable < 20 {

        font-weigth: bold;

    } @else {

        text-decoration: underline;

    }

}


@if $myVariable < 10 {                              /* the directive can also be used to produce whole rules               */

    #myElement {

        color: red;

    }

}



Iterations


@for $i from 1 through 3 {                          /* couting loop ("1 through 3" can also be written as "1 to 3")         */

    .myClass#{$i} {

        width: 2em * $i;

    }

}





@each $severity in info, warning, error {           /* iterate through a list of values                                     */

    .#{$severity}Box {

       border: 1px solid black;

    }

}





.myClass {

    @each $k, $v in (color: red, width: 10%) {      /* iterate through the key-value-pairs of a map                         */

        #{$k}: $v;

    }

}





@each $name, $color, $width in (info, black, 50%),  /* iterate through a list of lists                                      */

                               (warn, orange, 50%),

                               (error, red, 50%) {

    .#{$name}Box {

        color: $color;

        width: $width;

      }

}





$i: 1;

@while $i < 5 {                                     /* while-loop                                                           */

    h#{$i} {

        width: 2em * $i;

    }

    $i: $i + 1;

}
