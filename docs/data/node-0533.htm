eval


result = eval "MyClass.new"                 # returns the result of the expression





x = 10

def my_function

  x = 20

  binding                                   # retun the "binding" (i.e. the current scope/context)

end

eval("puts x")                              # return 10 (runs in the global context)

eval("puts x", my_function)                 # return 20 (runs in my_function's context)





my_object = MyClass.new

my_object.instance_eval("puts @property")   # run an eval in the context of a class

my_object.instance_eval { puts @property }  # can also use a code block rather than stringified code
