UDP server


require 'socket'                            # standard library


socket = UDPSocket.new                      # create a server socket

socket.bind(nil, 1234)                      # bind to all IP addresses on port 1234

text, sender = socket.recvfrom(16)          # receive a message of up to 16 bytes

socket.close                                # close the socket



UDP client


require 'socket'                            # standard library


socket = UDPSocket.new                      # create a client socket

socket.send("hello", 0, 'localhost', 1234)  # send a message to a given host

socket.close                                # close the socket
