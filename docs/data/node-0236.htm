Static content


Right click on IIS Express in the system tray       // local files matching the URL are served as static content by default

    => Show All Applications                        // in order to prevent this, follow these steps ...

    => Select the application in question

    => Click the "Config:" link



<add name="UrlRoutingModule-4.0"

     type="System.Web.Routing.UrlRoutingModule"

     preCondition="" />                             <!-- set the preCondition to "" (has some other value by default)         -->



public static void RegisterRoutes(

                        RouteCollection routes)

{

    routes.RouteExistingFiles = true;               // disable static content (all requests will be served through routes)


    routes.MapRoute(                                // create routes to intercept requests for static content

        "MyRouteName",

        "Content/StaticContent.html",               // the URL pattern reflects the local file

        new {controller = "Customer",               // map the route to a controller and action method

             action     = "List"    }

    );


    routes.IgnoreRoute("styles/{filename}.css");    // disable routing and serve static content for those URLs

}
