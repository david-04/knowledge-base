Shell commands


errorlevel = system("command")         # open a shell, run the command and close the shell


print "text" | "command"               # pipe text into shell command

"command"    | getline                 # run a shell command and read its output


print "text" |& "command"              # pipe text into a bidrectional shell command

"command"    |& getline                # read the output of a bidrectional shell command


fflush(["command"])                    # flush all pending input into the command (or all commands if no parameter is passed)


close("command")                       # close the specified command

close("command", "to")                 # bidrectional only: close the pipe feeding input into the command

close("command", "from")               # bidrectional only: close the pipe delivering output from the command




Shell commands


service = "/inet/tcp/0/127.0.0.1/100"  # the format is: /type/protocol/local-port/remote-host/remote-port

                                       # type .......... inet4 (IP4), inet6 (IP6) or inet (use system default)

                                       # protocol ...... tcp or udp

                                       # local-port .... local (outgoing) port (0 for auto-select)

                                       # remote-host ... remote server IP address or host name

                                       # remote-port ... remote server port


print "text" |& service                # send data to the remote service

service      |& getline                # receive data from the remote service

close(service)                         # close the connection
