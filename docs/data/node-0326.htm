class MainWindow : Window

{

    private Button button = null;


    public MainWindow(string windowTitle, int height, int width)             // construct the UIElements manually

    {

        this.Title                 = windowTitle;                            // configure the window

        this.WindowStartupLocation = WindowStartupLocation.CenterScreen;

        this.Height                = height;

        this.Width                 = width;


        this.button                = new Button();                           // configure a button

        this.button.Content        = "Test button";

        this.button.Height         = 25;

        this.button.Width          = 100;


        this.Content               = button;                                 // attach the button to the window

    }

}


class Program : Application

{

    [STAThread]

    static void Main(string[] args)

    {

        Program program  = new Program();                                    // create a new application instance

        program.Startup += AppStartUp;                                       // register startup and shutdown delegates

        program.Exit    += AppExit;

        program.Run();                                                       // fire a startup event

    }


    static void AppExit(object sender, ExitEventArgs e)                      // cleanup actions, save configuration, ...

    {

    }


    static void AppStartUp(object sender, StartupEventArgs e)

    {

        MainWindow main = new MainWindow("My WPF App!", 200, 300);           // instantiate a new window

        main.Show();

    }

}
