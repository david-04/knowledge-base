Layout (master page)


<!DOCTYPE html>                                     <!-- _MyLayout.cshtml (file should start with an underscore and reside  -->

<html>                                              <!-- in ~/Views/Controller or ~Views/Shared                             -->

<head>

    <title>@ViewBag.Title</title>                   <!-- can incorporate (onlay) data that's passed through the ViewBag     -->

</head>

<body>

    @RenderSection("Header")                        <!-- this section must be defined by the view                           -->

    @RenderBody()                                   <!-- the actual view data (that's not in a section) is rendered here    -->

    @RenderSection("OptionalFooter", false)         <!-- this section is only rendered if defined by the view               -->

</body>

</html>



View (content page)


@model MVC.Models.Product                           <!-- the content page can receive model data                            -->


@{

    ViewBag.Title = "My Page Title";                <!-- use the ViewBag to pass data to the layout page                    -->

    Layout = "~/Views/Shared/_MyLayout.cshtml";     <!-- specify which layout page to use                                   -->

}


<h1>@Model.category</h1>                            <!-- all non-sectioned content runs into the layout's RenderBody() call -->


@section Header {

    <p>Welcome!</p>

}
