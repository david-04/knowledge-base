Animations


.show       (milliseconds, ...)                     // increase opacity and size

.hide       (milliseconds, ...)                     // decreate opacity and size

.toggle     (milliseconds, ...)                     // alternate between show() and hide()


.slideDown  (milliseconds, ...)                     // increase the height

.slideUp    (milliseconds, ...)                     // decreate the height

.slideToggle(milliseconds, ...)                     // alternate between slideUp() and slideDown()


.fadeIn     (milliseconds, ...)                     // increase the opacity

.fadeOut    (milliseconds, ...)                     // decreate the opacity

.fadeToggle (milliseconds, ...)                     // decreate the opacity

.fadeTo     (milliseconds, opacity, ...)            // fade to the given opacity (0 ... 1)


.animate({                                          // transition to the given CSS styles

    opacity: 0.5,                                   // from the current opacity to 0.5

    fontSize: "20px"                                // from the current front size to 20px

}, milliseconds, ...);



Settings


.anyAnimationFunction(

    {milliseconds|"slow|normal|fast}"},             // instead of milliseconds the duratino can also be passed as "slow", "normal" or "fast"

    "{linear|swing|...}",                           // easing method (linear and swing are included, the others require jQueryUI)

    function onComplete() { }                       // called when the animation has been completed

);


.anyAnimationFunction({

    duration : 1000,                                // duration im milliseconds

    easing   : "linear",                            // easing method

    queue    : {false|"queueName"},                 // start animation immediately or add it to the queue (call dequeue("queueName") to start it)

    step     : function(now, tween) { },            // called for each animated property of each animated element (use it to modify the Tween object)

    progress : function(anim, progress, msLeft){ }, // called after each step of the animation, only once per animated element

    complete : function() {},                       // called when the animation has been completed

    start    : function(anim){},                    // called when the animation on an element begins

    done     : function(anim, jumpedToTheEnd) { },  // called when the animation on an element completes (its Promise object is resolved)

    fail     : function(anim, jumpedToTheEnd) { },  // called when the animation on an element fails to complete (its Promise object is rejected)

    always   : function(anim,jumpedToTheEnd) { },   // called when the animation on an element completes or stops without completing (its Promise object is either resolved or rejected)

    specialEasing : {                               // for .animate() only: set different easings for different CSS properties

        size:"linear",

        margin:"swing"

    }

});
