Defining an interface


public interface IPerson                                // interface names always start with an I

{

    string FirstName { get; }                           // required properties

    string LastName { get; }

    string FullName();                                  // required methods

}



Implementing an interface


public class Employee : Object, IPerson                 // derived from Object, immplementing IPerson

{

    public string FirstName { get; }

    public string LastName { get; }

    public Employee(string first, string last)

    {

        this.FirstName = first;

        this.LastName = last;

    }

    string IPerson.FullName()                           // if the interface name is used here, then the method can only be called by casting Employee to IPerson

    {

        return this.FirstName + " " + this.LastName;

    }

}
