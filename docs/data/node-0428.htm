Applets (Swing)


public class MyApplet extends JApplet {


    public void init() {

        try {

            SwingUtilities.invokeAndWait(           // initialisation is performed in the Swing thread to ensure that

                    () -> this.initialise());       // events are dispatched properly

        } catch (Exception e) {}

    }

   

    private void initialise() {                     // the applet is constructed just like a regular standalone application

       

        this.setSize(500, 500);

        this.setLayout(new FlowLayout());

       

        JLabel label = new JLabel("My label");

        this.add(label);                            // JDK5 (in JDK<5 use this.getContentPane().add(label))

       

        JButton button = new JButton("1");

        this.add(button);

       

        button.addActionListener(

            (ActionEvent ae) ->

                label.setText("Clicked")

        );       

    }


    public void start()   {...}                     // inject start, suspend and terminate activities

    public void stop()    {...}

    public void destroy() {...}


    public void paint(Graphics graphics) {...}      // paint is usually not overridden by Swing applets

}



Applets (AWT)


public class MyApplet extends    Applet

                      implements Runnable,          // implement Runnable if the applet requires a background thread

                                 MouseListener {    // implement event listeners if the class handles events itself


    public void init() {                            // initialisation (used like a constructor)

        this.addMouseListener(this);                // register for all events

        this.addMouseListener(new MouseAdapter() {  // ... possibly with inlined anonymous classes

            public void mouseClicked(MouseEvent e) {

                status = e.getX() + " " + e.getY();

            }

        });

    }

   

    public void start() {                           // the application is started (or resumed)

        (new Thread(this)).start();

    }


    public void stop() {...}                        // the application is stopped/suspended (e.g. suspend threads)

   

    public void destroy() {...}                     // the application closes (e.g. stop threads)

   

    public void paint(Graphics g) {                 // called when the window content must be restored (use like a destructor)

        g.drawString(this.status, 10, 20);          // print some text onto the canvas

        super.paint(g);                             // it is sometimes necessary to also invoke the base class' paint method

    }

   

    public void update(Graphics g) {                // called when only a part of the window must be restored

        this.paint(g);                              // (the default implementation in Applet just calls paint())

    }


    public void run() {...}                         // the background thread


    public void mouseClicked(MouseEvent e) {        // implementation of event handlers

        this.getGraphics().drawString(

            e.getX() + " " + e.getY(), 150, 150

        );

    }

}



Applet methods (AWT)


applet.getGraphics()                                // get the graphics context

applet.repaint()                                    // request update (or paint) to be called as soon as possible

applet.repaint(200)                                 // if the repaint has not taken place in 200ms, the requets is cancelled

applet.repaint(0, 0, 100, 100)                      // repaint only the top-left 100x100 pixels

applet.setBackground(Color.GRAY);                   // set the fore- or background colour

applet.setForeground(Color.GREEN);

applet.showStatus("Cool stuff!");                   // set the text displayed in the status bar


applet.getParameter("param");                       // get the string value of a parameter passed via the HTML <param> tag (null if not set)


applet.getDocumentBase()                            // get the path of the HTML document

applet.getCodeBase()                                // get the folder of the applet's class file

applet.getAppletContext().showDocument(newUrl)      // navigate the browser to another page
