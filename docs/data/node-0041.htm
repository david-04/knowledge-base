Authentication (server login)


CREATE LOGIN user_name WITH PASSWORD = '***';       -- create a new user account

CREATE LOGIN [domain\user] FROM WINDOWS;            -- create a user account for an existing windows user


ALTER LOGIN ...

DROP LOGIN ...



Authentication (database users)


CREATE USER database_user FOR LOGIN server_login    -- create a database user based on an existing server account

WITH DEFAULT_SCHEMA = my_schema;                    -- optionally set the default schema for the user


CREATE USER database_user  WITH PASSWORD = '***';   -- create a standalone user in a contained database


ALTER USER ...;                                     -- modify or delete a database user

DROP USER ...;



Roles


CREATE SERVER ROLE my_role;                         -- create a new server-wide role

ALTER SERVER ROLE my_role ADD MEMBER my_user;       -- add/remove members from a role

DROP SERVER ROLE my_role;                           -- delete a server-wide role


CREATE ROLE my_role;                                -- create a new database-specific role

ALTER ROLE my_role ADD MEMBER my_user;              -- add/remove members from a role

DROP ROLE my_role;                                  -- delete a database-specific role



Authorisation


GRANT { SELECT|INSERT|UPDATE|DELETE                 -- grant a permission

       |REFERENCES|EXECUTE|CONTROL|ALTER

       |TAKE OWNERSHIP|VIEW DEFINITION

       |CREATE TABLE/PROCEDURE/...}

ON my_object                                        -- for the given object (can also be my_table(col1, col2))

TO user_or_role_1, user_or_role_2                   -- to the given user

WITH GRANT OPTION                                   -- allow the user to grant the permission to a third user

AS principal


DENY permission ON my_object TO my_user             -- deny a permission

CASCADE                                             -- also deny it from any other user that was granted the right by this user

AS principal


REVOKE persmission ON my_object FROM my_user        -- revoke a previously granted permission

CASCADE

AS principal
