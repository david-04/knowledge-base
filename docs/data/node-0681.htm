Binding to UI elements


<TextBox x:Name="myTextBox"/>

<Button Content="{Binding ElementName=myTextBox, Path=Text}"/>    <!-- bind to myTextBox.Text                              -->


<StackPanel DataContext="{Binding ElementName=myTextBox}">        <!-- the DataContext is inherited by all child elements  -->

    <TextBox x:Name="myTextBox"/>

    <Button Content="{Binding Path=Text}"/>                       <!-- only the path needs to be specified                 -->


Binding configuration:


ElementName           = myTextBox          <!-- the root element, can be left out if there is a valid DataContext          -->

Path                  = Text               <!-- the (nested) property of the element containing the relevant data          -->

Mode                  = OneWay             <!-- uni- or bidrectional synchronisation                                       -->

ValidatesOnExceptions = True               <!-- mark the input control red if writing back data raised an exception        -->

UpdateSourceTrigger   = PropertyChanged    <!-- write data back as soon as it changes (not only when losing the focus)     -->



Dependency properties


<TextBox Text="{Binding ElementName=MyWindow, Path=MyInfo}"/>     <!-- bind to a custom property                           -->


Shared ReadOnly Property MyInfoProperty As DependencyProperty =   ' Create the skeleton by typing propdp and pressing TAB

DependencyProperty.Register(

        "MyInfo",                                                 ' unique property name

        GetType(String),                                          ' data type

        GetType(MainWindow)                                       ' the window class owning this property

    )


Public Property MyInfo As String                                  ' accessors only redirect To the DependencyProperty class

    Get

        Return CType(MyBase.GetValue(MyInfoProperty), String)

    End Get

    Set(value As String)

        MyBase.SetValue(MyInfoProperty, value)

    End Set

End Property

    New Point(10, 20),

    New Point(12, 80)

}


Me.MyDataGrid.ItemsSource = points                                ' bind the container to the UI element
