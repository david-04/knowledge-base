Binding programatically


protected void Page_Load(object sender, EventArgs e)

{

    this.GridView.DataSource = this.CurrentCustomer;                               // link the control to the actual object

    this.GridView.DataBind();                                                      // let the binding manager copy all attributes

}



Eval and Bind


<asp:GridView    ... DataSourceID="CustOrdersDataSource" />                        <!-- only a data source needs to be attached       -->

    <asp:Label   ... Text='<%# Eval("LastName")+", "+Eval("FirstName")  %>' />     <!-- Eval supports expressions and function calls  -->

    <asp:Label   ... Text='<%# Eval("LastOrderDate", "{0:mm dd yyyy}" ) %>' />     <!-- Eval supports formatting as well              -->

    <asp:TextBox ... Text='<%# Bind("LastName")                         %>' />     <!-- Bind facilitates a two-way-binding (edit)     -->



Item and BindItem


<asp:GridView    ... ItemType="MyDataModel.Customer" />                            <!-- the data class has to be specified            -->

    <asp:Label   ... Text='<%# Item.LastName + ", " + Item.FirstName    %>' />     <!-- Item is the read-only data element            -->

    <asp:TextBox ... Text='<%# BindItem.LastName                        %>' />     <!-- BindItem supports writing/updating as well    -->



Binding to XML nodes


<% XPathBinder.Eval(Container.DataItem, "employees/employee/Name")      %>         <!-- apply an XPath to the current element         -->

<% XPath("employees/employee/Name")                                     %>         <!-- same as above, just a simpler syntax          -->

<% XPath("employees/employee/HireDate", "{0:mm dd yyyy}")               %>         <!-- XPath results can be formatted                -->

<% XPathBinder.Select(Container.DataItem,"employees/employee")          %>         <!-- retrieve multiple records at once             -->

<% XpathSelect("employees/employee")                                    %>         <!-- same as above, just a simpler syntax          -->
