Variable argument list


Function Concat(ParamArray values() As Object) As String

    Dim result As StringBuilder = New StringBuilder

    For Each obj As Object In values

        result.Append(obj.ToString()).Append(" ")

    Next


    Return result.ToString()

End Function

Optional and named parameters


Function Greet(firstName As String, Optional lastName As String = "") As String

    Return "Hello " & firstName & " " & lastName

End Function
