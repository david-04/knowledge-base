Defining styles and logical resources


<Application ...>

    <Application.Resources>

        <SolidColorBrush x:Key="MyBrush" Color="blue" />           <!-- logical resources like brushes, colours, fonts, ...            -->

        <Style x:Key="MyStyle">                                    <!-- styles combine multiple facets of dofferemt resources          -->

            <Setter Property="Control.FontSize" Value="14"/>

            <Setter Property="Control.Cursor"   Value="Hand"/>

        </Style>



Applying styles and resources


<Button Content="Yes" Background="{StaticResource  MyBrush}"/>     <!-- static resources are applied only once during initalisation    -->

<Button Content="No"  Style     ="{DynamicResource MyStyle}"/>     <!-- dynamic resources are re-applied when changed programmatically -->


Dim myStyle As Style = CType(TryFindResource("MyStyle"), Style)    ' retrieve Object resources by their name

If myStyle IsNot Nothing Then

    Me.myButton.Style = myStyle                                    ' Set an element's style dynamically

End If



Style inheritance


<Style x:Key="MyStyle">                                            <!-- the base style must have an x:Key name                         -->

    <Setter Property = "Control.FontSize" Value ="25"/>

</Style>


<Style x:Key="MyBetterStyle" BasedOn="{StaticResource MyStyle}">

    <Setter Property="Control.Cursor" Value="Hand"/>

</Style>



Default type binding



<Style x:Key="MyStyle" TargetType="Button">                        <!-- can only be applied to buttons                                 -->


<Style x:Key="MyStyle" TargetType="Button">                        <!-- applies to all buttons as a default style                      -->


<Button Content="OK" Style="{x:Null}"/>                            <!-- elements can opt out of the default style                      -->



Context-specific styles


<Style TargetType="TextBox">

    <Style.Triggers>

        <Trigger Property="IsFocused" Value="True">                <!-- only applies when the element has the focus                    -->

            <Setter Property="Background" Value="Yellow"/>

        </Trigger>

        <MultiTrigger>

            <MultiTrigger.Conditions>

                <Condition Property="IsFocused"   Value="True"/>   <!-- only applies when the element has the focus ...                -->

                <Condition Property="IsMouseOver" Value="True"/>   <!-- and if the mouse is over the element                           -->

            </MultiTrigger.Conditions>

            <Setter Property = "Background" Value = "Yellow"/>

        </MultiTrigger>

    </Style.Triggers>

</Style>



Animation styles


<Style x:Key="GrowingButtonStyle" TargetType="Button">

    <Setter Property="Height" Value="40"/>

    <Setter Property="Width" Value="100"/>

    <Style.Triggers>

        <Trigger Property="IsMouseOver" Value="True">

            <Trigger.EnterActions>

                <BeginStoryboard>

                    <Storyboard TargetProperty="Height">

                        <DoubleAnimation From="40" To="200"/>

                    </Storyboard>

                </BeginStoryboard>

            </Trigger.EnterActions>

        </Trigger>

    </Style.Triggers>

</Style>
