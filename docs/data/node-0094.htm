import { NgModule      } from "@angular/core";

import { RouterModule,

         Routes        } from "@angular/router";


const myRoutes: Routes = [                          // routing table

    {

        path      : "",                             // redirect the root path to a default component

        redirectTo: "/people",

        pathMatch : "full"

    },

    {

        path      : "people",                       // map paths to components

        component : ListViewComponent

    },

    {

        path      : "edit/:id",                     // paths can contain placeholders for parameters

        component : EditViewComponent

    }

];


@NgModule({                                         // attach the routing table to the module

    imports: [RouterModule.forRoot(myRoutes)],

    exports: [RouterModule]

})

export class AppRoutingModule { }                   // naming convention: ...RoutingModule in file ...-routing.module.ts






@NgModule({

    imports: [

        MyFeatureRoutingModule,                     // import the routing tables in the application's root module

        AppRoutingModule                            // order matters: import feature modules first

    ]                                               // (AppRoutingModule probably has a "**" => 404 rule)

})

export class AppModule { }





@Component({

    selector: "my-application",

    template: "<router-outlet></router-outlet>",    // the routing target components will be inserted right here

})

export class AppComponent { }



Feature component routing


@NgModule({

    imports: [RouterModule.forChild(myRoutes)],     // register the routes via "forChild" (not "forRoot")

    exports: [RouterModule]                         // (route paths have the same form [full URL] as for root routing)

})

export class MyFeatureRoutingModule { }





@NgModule({

    imports: [MyFeatureRoutingModule]               // import the feature-specific sub-routes into the feature module

})

export class MyFeatureModule { }



Route definitions


{                                                   // order matters: the first rule that matches, will be applied

    path      : "messages",

    component : InboxComponent

} , {

    path      : "users/:id",                        // use placeholders for path parameters

    component : UserDetailComponent

} , {

    path      : "welcome",                          // attach static data to the route; access via:

    component : WelcomeComponent,                   // route.data.subscribe((data:{title:string})=>{this.tile=data.title;});

    data      : { title: "G'Day" }

} , {

    path      : "",

    redirectTo: "/shop",                            // redirect to another path (defined in the routing table)

    pathMatch : "full"                              // default is prefix (the current route must start with the given path)

} , {

    path     : "**",                                // wildcard URL, matches any path

    component: PageNotFoundComponent

} , {

    path: "admin",

    component: AdminComponent,                      // this component requires a <router-outlet> for its child routes

    children: [

        {

            path: "",

            component: AdminOverviewComponent,      // the same as above applies

            children: [

                {

                    path: ":userId",                // if the path contains a user ID, then show the details

                    component: UserDetailComponent

                },

                {

                    path: "",

                    component: UserListComponent    // otherwise show a listing

                }

            ]

        }

    ]

}



Secondary routes


const myRoutes: Routes = [

    {

        path      : "rating",

        component : RatingComponent

        outlet    : "feedback"                      // this is a secondary route, targetting the outlet named "feedback"

    },

];





<router-outlet></router-outlet>                     <!-- the URL-derived component will be shown here                          -->

<router-outlet name="feedback"></router-outlet>     <!-- this outlet is populated through secondary routes                     -->


<a [routerLink]="[{outlets:{feedback:['rating']}}]"><!-- show the rating (secondary) route in the feedback outlet              -->





this.router.navigate([{outlets:{feedback:null}}]);  // clear the feedback outlet
