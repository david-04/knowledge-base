Observables (recommended)


import { Injectable     } from "@angular/core";

import { Headers,

         Http,

         Response,

         RequestOptions } from "@angular/http";

import { Observable     } from "rxjs/Observable";


import { Person         } from "./data-model";


import "rxjs/add/operator/catch";

import "rxjs/add/operator/map";


@Injectable()

export class RestService {


    constructor(private http: Http) { }             // inject the Ajax service


    get(): Observable<Person[]> {                   // service method to retrieve the result list

        return this.http

                   .get("/app/rest.service.json")   // URL to fetch

                   .map(this.extractData)           // retrieve the payload data from the reponse's body

                   .catch(this.handleError);        // error handling

    }


    add(name: string): Observable<Person> {         // service method to add/create a new object

        let headers = new Headers({

            "Content-Type": "application/json"      // mark the body data as JSON

        });

        let options = new RequestOptions({

            headers: headers

        });

        return this.http

                   .post(

                       "/person/add",               // send the request to this URL

                       { name },                    // JSON'fy his object (literal) as body

                       options                      // use these options (HTTP headers in this case)

                    )                   

                    .map(this.extractData)

                    .catch(this.handleError);

    }


    private extractData(res: Response) {            // retrieve the payload data

        let body = res.json();                      // parse the response into a JSON object

        return body || {};                          // unpack if necessary (e.g. return body.data || {})

    }


    private handleError(error: Response | any) {    // error handling

        let errMsg: string;

        if (error instanceof Response) {

            const body = error.json() || "";

            const err = body.error || JSON.stringify(body);

            errMsg = `${error.status} - ${error.statusText || ""} ${err}`;

        } else {

            errMsg = error.message ? error.message : error.toString();

        }

        console.error(errMsg);

        return Observable.throw(errMsg);

    }

}





this.restService.get().subscribe(                   // use the rest service and subscribe to the observable's events

    people => this.people = people,

    error  => this.errorMessage = <any>error

);


this.restService.add(name).subscribe(

    person => this.people.push(person),             // when successful: add the newly created object to the view

    error  => this.errorMessage = <any>error

);



Promises


import "rxjs/add/operator/toPromise";               // convert observables (returned by http.get) into promises


export class RestService {


    get(): Promise<Person[]> {

        return this.http

                   .get("/app/rest.service.json")

                   .toPromise()

                   .then(this.extractData)

                   .catch(this.handleError);

    }


    add(name: string): Promise<Person> {

        let headers = new Headers({

            "Content-Type": "application/json'"

        });

        let options = new RequestOptions({

            headers: headers

        });

        return this.http

                   .post(

                       "/person/add",

                       { name },

                       options

                   )

                   .toPromise()

                   .then(this.extractData)

                   .catch(this.handleError);

    }

}





this.restService.get().then(

        people => this.people = people,

        error => this.errorMessage = <any>error

    );


this.restService.add(name).then(

    person => this.people.push(person),

    error  => this.errorMessage = <any>error

);
