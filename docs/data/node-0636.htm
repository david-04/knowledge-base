Defining attributes


<AttributeUsage(AttributeTargets.Class,         ' limit the atribute to certain elements (bitwise or flag)

                AllowMultiple:=True,            ' can be applied to the same element multiple times

                Inherited:=True)>               ' apply to inherited classes

Public NotInheritable Class MyAttribute

    Inherits Attribute                          ' derived from System.Attribute


    Public Property Message As String


    Public Sub New()                            ' default constructor: [MyAttribute]

    End Sub


    Public Sub New(message As String)           ' another constructor: [MyAttribute("message")]

        Me.Message = message

    End Sub


End Class



Applying attributes


<Assembly: CLSCompliant(True)>                  ' applies to the whole namespace

Namespace MyNamespace


    <Serializable>                              ' applies to the whole class

    Public Class SomeClass


        <Obsolete("Use ... instead")>           ' applies to a single member

        Public Property Id As Integer


    End Class


End Namespace
