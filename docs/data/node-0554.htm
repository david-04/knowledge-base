JSON


require 'json'                              # standard library


json = JSON.generate(hash_or_array)         # convert a hash or array into a stringified json

json = hash_or_array.to_json                # same as above


object = JSON.parse(                        

  json, symbolize_names: true               # parse keys into symbols

)

object[:name]


object = JSON.parse(                        

  json, symbolize_names: false              # parse keys into strings

)

object["name"]



YAML


require 'yaml'                              # stanard libray


yaml = hash_object_or_array.to_yaml         # render an object/array/hash into a YAML string (includes reference to the Ruby classes)

hash_object_or_array = YAML.load(yaml)      # parse a YAML into an object/array/hash (restores class intances)



CSV


require 'csv'                               # standard library


array = CSV.parse(File.read('file.csv'))    # read the whole file in to a 2-dimensional array

array = CSV.read('file.csv')                # same as above

CSV.open('file.csv').each { |columns| ... } # stream each line as an array of columns


CSV.open('file.txt', 'w') do |csv|          # write a CSV file

  array.each { |columns| csv << columns }

end



PStore


require 'pstore'                            # standard library


store = PStore.new("file.pstore")           # open the store (for reading and writing)


store.transaction do                        # write to a store

  store[:people] ||= Array.new              # initialise the propery

  store[:people] << "John"                  # serialise objects

  store[:people] << "Johanna"

end


people = []                                 # reading from a store

store.transaction do

  people = store[:people]

end
