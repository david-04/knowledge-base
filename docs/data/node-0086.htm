Data binding


{{person.name}}                                     <!-- interpolation; insert the property value (masks HTML symbols)          -->

<span [innerHTML]="person.name"></span>             <!-- set the innerHTML (allows HTML code but no <script> tags)              -->


{{person?.name}}                                    <!-- template expressions support the safe-navigation operator              -->


<input [(ngModel)]="person.name" name="name">       <!-- two-way binding ("name" is mandatory if <input> is inside a <form>)    -->

                                                    <!-- requires: import { FormsModule } from "@angular/forms";                -->


<my-component [data]="selectedItem">                <!-- pass data to a child component (child property marked with @Input())   -->

<my-component [(data)]="selectedItem">              <!-- exchange data with a child component via two-way binding               -->


<tagName   attribute  ="static value">              <!-- set to a static (non-evaluated value)                                  -->

<tagName  [attribute] ="user.name">                 <!-- set to an expression that must be evaluated                            -->

<tagName  (attribute) ="myEventHandler()">          <!-- receive and output (an emitted event)                                  -->

<tagName [(attribute)]="user.name">                 <!-- two-way data binding; syntax style: "banana in a box" [()]             -->



HTML attributes


<img [src]="imageUrl">                              <!-- set an HTML attribute (to the component's imageUrl property)           -->

<img bind-src="imageUrl">                           <!-- same as above, alternative syntax                                      -->


<img src="./{{imageName}}.jpg">                     <!-- set an HTML attribute to static text (with possible inpterpolations)   -->


<button [disabled]="disableButton">                 <!-- omits the disabled attribute if the button should be enabled           -->


<td [attr.colspan]="colSpanCount">                  <!-- set an attribute that doesn't have a DOM property (useful for ARIA)    -->

                                                    <!-- this kind of binding is executed only once upon initialisation         -->


CSS styles


<div class="panel" [class.error]="!isValid">        <!-- if isValid evaluates to false, add the CSS class "error"               -->


<div [ngClass]="myClasses">                         <!-- let myClasses = {important : true, error : false};                     -->


<button [style.color]="isValid ? 'green' : 'red'">  <!-- add an inline style to the element                                     -->


<button [style.width.em]="hightlight ? 3 : 1">      <!-- inline styles can have units                                           -->


<div [ngStyle]="myStyles">                          <!-- let myStyles = {"font-size" : "2em", "font-weight" : "bold" };         -->



Template reference variables


<input #phone >                                     <!-- define a symbolic name for a DOM element or an Angular component       -->

<input ref-phone>                                   <!-- same as above, different syntax                                        -->

<button (click)="call(phone.value)">Call</button>   <!-- use the symbolic name like a local variabable                          -->
