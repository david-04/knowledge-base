Audio


<audio src        = "my-song.mp3"                   <!-- mp3 has the best browser support, wav+ogg work occasionally        -->

       preload    = "{auto|metadata|none}"          <!-- preload the file (auto), metadata or no data at all                -->

       mediagroup = "myMediaGroup"                  <!-- all audio/videos in the same group will be played by in-sync       -->

       controls                                     <!-- show controls for the user to start/stop/change the playback       -->

       autoplay                                     <!-- automatically start the playback when the page loads               -->

       loop                                         <!-- play the sound track in a loop                                     -->

       muted >                                      <!-- initially mute the player                                          -->


    <source src   = "my-audio.mp3"                  <!-- add different foramts (don't set the "src" of the <audio> tag)     -->

            type  = "audio/mp3"                     <!-- specify the file/format type                                       -->

            media = "{all|braille|print|...}">      <!-- specify what medium this format is targetted at (also supports     -->


    <source src   = "my-audio.ogg"                  <!-- media queries like media="(max-device-width: 480px)")              -->

            type  = "audio/ogg">


</audio>                                            <!-- must have a separate closing tag                                   -->



Videos


<video src        = "my-song.mp4"                   <!-- mp4 has the best browser support, webm+ogg work occasionally       -->

       poster     = "preview.jpg"                   <!-- preview image to show before the playback is started               -->

       width      = "640"                           <!-- size of the video display on the page                              -->

       height     = "480"

       preload    = "{auto|metadata|none}"          <!-- preload the file (auto), metadata or no data at all                -->

       mediagroup = "myMediaGroup"                  <!-- all audio/videos in the same group will be played by in-sync       -->

       controls                                     <!-- show controls for the user to start/stop/change the playback       -->

       autoplay                                     <!-- automatically start the playback when the page loads               -->

       loop                                         <!-- play the video in a loop                                           -->

       muted >                                      <!-- initially mute the player                                          -->


    <source src   = "my-video.mp4"                  <!-- add a <source> element for each available file format              -->

            type  = "video/mp4"                     <!-- specify the file/format type (can also specify the codec)          -->

            media = "{all|braille|print|...}">      <!-- specify what medium this format is targetted at (also supports     -->


    <source src   = "my-video.webm"                 <!-- media queries like media="(max-device-width: 480px)")              -->

            type  = "video/webm">


    <track src     = "my-subtitles.vtt"             <!-- add a WebVTT subtitle/captions file                                -->

           kind    = "{subtitles|captions|...}"     <!-- kind={description|chapters|metadata} will not be displayed         -->

           srclang = "en"                           <!-- natural language of the subtitles                                  -->

           label   = "English"                      <!-- the title to be presented to the user when chosing a subtitle      -->

           default />                               <!-- make this the default subtitle                                     -->


</video>                                            <!-- must have a separate closing tag                                   -->
