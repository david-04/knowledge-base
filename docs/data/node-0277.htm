Switching between specilised master pages


public partial class Default : System.Web.UI.Page

{

    protected void Page_PreInit(object sender, EventArgs e)

    {

        if (Request.Browser.IsMobileDevice)

            MasterPageFile = "~/MobileMaster.Master";

    }

}



Using specialised pages in a parallel folder structure


public class Global : System.Web.HttpApplication

{

   protected void Session_Start(object sender, EventArgs e)

    {

        HttpRequest httpRequest = HttpContext.Current.Request;

        if (httpRequest.Browser.IsMobileDevice)                             // only switch for mobile clients

        {

            string path = httpRequest.Url.PathAndQuery;

            bool isOnMobilePage = path.StartsWith("/Mobile/",               // only redirect if the non-mobile version has been requestested

                                    StringComparison.OrdinalIgnoreCase);

            if (!isOnMobilePage)

            {

                string redirectTo = "~/Mobile/";

                HttpContext.Current.Response.Redirect(redirectTo);          // redirect the browser

            }

        }

    }

}
