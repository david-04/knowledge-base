Exceptions


try {


    throw 12345;                                    // throw numbers and strings

    throw "Hello world!";

    throw { name: "JavaScript" };                   // throw complex objects

    throw new Error("Something bad happened.");     // throw exceptions of a particular type


} catch (error) {                                   // catch can be left out if there is a finally


    error.message                                   // error description

    error.name                                      // error type


    if (error instanceof Error) { }                 // base type

    if (error instanceof EvalError) { }             // eval() failed

    if (error instanceof RangeError) { }            // array index out of bounds

    if (error instanceof ReferenceError) { }        // referencing an undefined variable

    if (error instanceof SyntaxError) { }           // syntax error in an eval()

    if (error instanceof TypeError) { }             // type mismatch

    if (error instanceof URIError) { }              // malformed URI in encodeURI() or decodeURI()


} finally {

   

    // ...

}

window.onerror = function (msg, url, line) {        // not supported in Opera and Safari
