Defining attributes


[AttributeUsage(AttributeTargets.Class,         // limit the atribute to certain elements (bitwise or flag)

                AllowMultiple = true,           // can be applied to the same element multiple times

                Inherited = true)]              // apply to inherited classes

public sealed class MyAttribute : Attribute     // derived from System.Attribute

{

    public string Message { get; set; }

    public MyAttribute() { }                    // default constructor: [MyAttribute]

    public MyAttribute(string message)          // another constructor: [MyAttribute("message")]

    {

        this.Message = message;

    }

}



Applying attributes


[assembly: CLSCompliant(true)]                  // applies to the whole namespace

namespace MyNamespace

{

    [Serializable]                              // applies to the whole class

    public class MyClass

    {

        [Obsolete("Use ... instead")]           // applies to a single member

        public int id;

    }

}
