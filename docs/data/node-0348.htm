Borders


border:        1px solid red;                       /* values can be passed in any order                                            */

border-color:  red;                                 /* can also be set to currentColor (CSS3)                                       */

border-width:  1px;                                 /* can also be set to thin, medium or thick                                     */

border-style:  solid;                               /* dashed, dotted, solid, double, groove, ridge, inset and outset               */


border-radius: 8px;                                 /* round corners                                                                */

border-radius: 20px / 8px;                          /* elliptical corners (width, height)                                           */

border-radius: 10px 10px 20px 20px;                 /* top+left => top+right => bottom+right => bottom+left                         */

border-radius: 1px 2px 3px 4px / 9px 8px 7px 6px;   /* elliptical corners (width[top+left,top+right].../height[...])                */


border-{top|bottom|left|right} : 1px solid red;     /* style only one side of the frame                                             */

border-{top|bottom|left|right}-color : red;                                

border-{top|bottom|left|right}-width : 1px;                                

border-{top|bottom|left|right}-style : solid;


border-{top|bottom}-{left|right}-radius: 8px;                                


outline:       5px solid red;                       /* show a (additional) border outside of the element                            */

outline-color: black;

outline-style: dotted;

outline-width: 1px;

                                                   



Border images


border:              10px solid transparent;        /* important: must reserve some screen real estate for the border (image)       */


border-image-source: url("my-image.jpg");           /* the image to be used as border                                               */

border-image-slice : 10;                            /* take only the outermost 10 pixels (from each side)                           */

border-image-slice : 10 10 10 10;                   /* slice differently on each side (top, right, bottom, left)                    */

border-image-width : 50%;                           /* width of the border image                                                    */

border-image-width : 10% 20% 30% 40%;               /* different widths per side (top, right, bottom, left)                         */

border-image-repeat: stretch;                       /* repeat, stretch, round (scale tiles to fit), space (distribute tiles to fit) */

border-image-outset: 20px;                          /* add padding between the border and the element itself                        */


border-image:        url("my-image.jpg")            /* border-image-source                                                          */

                     10                             /* border-image-slice                                                           */

                     20                             /* border-image-width                                                           */

                     20                             /* border-image-outset                                                          */

                     space;                         /* border-image-repeat                                                          */


border-image:        url("my-image.jpg") 10 repeat; /* slice 10 pixels from the image and and repeat the tiles                      */
