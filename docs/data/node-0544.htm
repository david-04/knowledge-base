Modules


module Module1                              # module can be structured/used like a (static) class

  def self.my_function                      # all methods must be bound to "self"

    my_helper

  end

  def self.my_helper

    puts "Module 1"

  end

end


Module1.my_function                         # access like a static class





module Module2                              # modules can be structure for import

  def my_function                           # methods must not be bound to "self"

    my_helper

  end

  def my_helper

    puts "Module 2"

  end

end


include Module2                             # import the module into the current scope

my_function                                 # all elements are directly accessible



Mixins


module MyFeatures                           # modules can be structured to be injected into classes

  public                                    # can set visibility (it's public by default)

  def dump                                  # functions are not bound to "self"

    puts self.class.to_s                    # ...but needs to use "self" when calling other methods of the class

  end

end


class MyClass

  include MyFeatures                        # inject the module into the class

end


myObject = MyClass.new

myObject.dump                               # the module's method can now be called
