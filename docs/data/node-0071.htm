Forms and validation


<form name="myForm"                                 <!-- each form must have a name attribute                                           -->

      novalidate                                    <!-- disable browser validation (would interfer with the AngularJS validation)      -->

      ng-submit="addUser(newUser)">                 <!-- event handler to process the form submission                                   -->


    <input  name           = "myTextbox"            <!-- each input element must have a name attribute                                  -->

            type           = "text"                 <!-- can be: checkbox, email, number, radio, text or url                            -->

            ng-readonly    = "isReadOnly()"         <!-- dynamically set or remove the readonly property                                -->

            ng-minlength   = "getMinLength()"       <!-- dynamically set the minimum text length                                        -->

            ng-maxlength   = "getMaxLength()"       <!-- dynamically set the maximum text length                                        -->

            ng-pattern     = "validationPattern"    <!-- validating regular expression (RegExp("^[a-z]"); not for email, url or number) -->

            ng-required    = "isFieldRequired()"    <!-- dynamically set or remove the "required" attribute                             -->

            ng-model       = "userInput.text" />    <!-- data binding                                                                   -->


    <input  name           = "myCheckbox"

            type           = "checkbox"

            ng-checked     = "mustBeChecked()"      <!-- dynamically set the "checked" state of the checkbox                            -->

            ng-true-value  = "getTrueValue()"       <!-- set the value that is saved back to the model when the checkbox is checked     -->

            ng-false-value = "'declined'"/>         <!-- ... can be set dynamically or statically                                       -->


    <button name           = "myButton"

            type           = "submit"

            ng-disabled    = "myForm.$invalid"      <!-- dynamically disable buttons                                                    -->

            value          = "Save" />

</form>



Listbox directives


<form name="myForm" novalidate>

    <select name       = "myDropDownBox"

            ng-model   = "userSelection"

            ng-options = "u.name for u in users">   <!-- auto-generate the options from a collection of the $scope                      -->

        <option value  = ""                         <!-- add this to prevent the automatic addition of: <option value="?"></option>     -->

                ng-selected = "isSelected()">       <!-- dynamically set the selected attribute on an option element                    -->

        </option>              

    </select>

</form>


ng-options="u.name for u in users"                  <!-- data  binding saves the whole "u" object as the selection result in $scope     -->

ng-options="u.id as u.name for u in users"          <!-- data  binding saves only "u.id" as the selection result in $scope              -->

ng-options="u.name group by u.group for u in users" <!-- creates OPTGROUP elements to visually structure/group the listbox items        -->



Validation indicators


$scope.myForm                                       // for each form an object is created within the scope

$scope.myForm.$pristine                             // boolean indicating if the user has NOT interacted with the form yet

$scope.myForm.$dirty                                // boolean indicating if the user has already interacted with the form

$scope.myForm.$valid                                // boolean indicating if all validation rules are satisfied

$scope.myForm.$invalid                              // boolean indicating if there are validation errors

$scope.myForm.$error                                // error details are contained in an $error property (see below)


$scope.myForm.myTextbox                             // for each of the form's elements a property is created

$scope.myForm.myTextbox.${pristine|...|invalid}     // pristine/dirty and valid/invalid flags are the same as for the whole form

$scope.myForm.myTextbox.$error.{email|url|number}   // boolean indicating if the text input does not validate against the type

$scope.myForm.myTextbox.$error.required             // boolean indicating if a mandatory field is still empty

$scope.myForm.myTextbox.$error.minlength            // boolean indicating if the text input is too short

$scope.myForm.myTextbox.$error.maxlength            // boolean indicating if the text input is too long

$scope.myForm.myTextbox.$error.pattern              // boolean indicating if the regular expression pattern does not match the input



Styling


.ng-pristine                                        /* class assigned to untouched forms and input fields                                */

.ng-dirty                                           /* class assigned to modified forms and input fields                                 */

.ng-valid                                           /* class assigned to valid forms and input fields                                    */

.ng-invalid                                         /* class assigned to invalid forms and input fields                                  */

.ng-required                                        /* class assigned to empty mandatory input fields                                    */

.ng-pattern                                         /* class assigned to input fields that don't match the pattern                       */

.ng-minlength                                       /* class assigned to input fields with too little text                               */

.ng-maxlength                                       /* class assigned to input fields to too much text                                   */

.ng-url                                             /* class assigned to input fields with invalid URLs                                  */

.ng-email                                           /* class assigned to input fields with invalid email addresses                       */

.ng-number                                          /* class assigned to input fields with invalid numbers                               */


input.ng-invalid.ng-dirty {color:red;}              /* use them in combination to prevent error styles on pristine elements and forms    */



Showing error messages


<form name="myForm" novalidate>

    <input name="myField" type="email" required

            ng-model="userInput" />

    <div ng-show="myForm.myField.$dirty">           <!-- show errors only if the suer has interacted with the field                     -->

        <p ng-show="myForm.myField.$error.email">   <!-- check the $error object and render the appropriate error message               -->

            Please enter a valid email address

        </p>

        <p ng-show="myForm.myField.$error.required">

            Please enter a value

        </p>

    </div>

    <button type="submit"

            ng-disabled="myForm.$invalid"

            value="Save" />                         <!-- enable the submit button only if the form is valid                             -->

</form>


$scope.getErrorMessage = function (error) {         // add a method to the controler to retrieve the suitable error message

    if (angular.isDefined(error)) {                 // ensure that there is acutally an error

        if (error.required) {                       // check the different error type indicators

            return "Please enter a value";          // ... and create a matching error message

        } else if (error.email) {

            return "Please enter a email address";

        }

    }

}
