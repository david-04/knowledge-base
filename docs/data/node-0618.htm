Enums


enum Season {

    Spring,                                         // 0

    Summer,                                         // 1

    Autumn = 100,                                   // 100

    Winter                                          // 101

}


enum Season {

    Spring = "SPRING",                              // can also initialise as string

    Summer = "SUMMER",                              // however, there's no reverse mapping "SUMMER" => "Summer"

    Autumn = "AUTUMN",

    Winter = "WINTER"

}


enum FileAccess {

    Read      = 1 << 1,                             // computed value based on constants and literals

    Write     = 1 << 2,

    ReadWrite = Read | Write,                       // computed value based on other enumeration values

    Random    = getRandomAccessValue()              // ... or on function calls as well

    Label     = "FILE_ACCESS"                       // ... can also mix with strings

}




Name mapping


type SeasonNames = keyof typeof Season;             // enum names, similar to property names for classes: "Spring" | "Summer" | "Autum" | "Winter"


Season.Spring;                                      // 0 (the value assigned to the enum)

Season["Spring"];                                   // 0 (the value assigned to the enum)

Season[Season.Spring]                               // "Spring" (the nam of the enum)

Season[0]                                           // "Spring" (the nam of the enum)




Static methods


enum Color {

    red = 1,

    green = 2,

    blue = 4,

}


namespace Color {

    export function fromString(name: string) {      // inject a static (factory) method into the enum

        switch (name) {

            case "red": return Color.red;

            case "green": return Color.green;

            case "blue": return Color.blue;

            default: throw new Error("???");

        }

    }

}



Inlining


const enum NodeType {                               // "const" works like the C preprocessor (used for performance reasons):

    Parent,                                         // #define NodeType.Parent 0

    Child                                           // #define NodeType.Child  1

}                                                   // (can't retrieve the name via NodeType[NodeType.Child] at runtime)
