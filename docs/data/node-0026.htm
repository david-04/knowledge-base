Lists


LPUSH  myList "Redis" "MongoDB"                     // prepend these elements at the beginning of the list

RPUSH  myList "Redis" "MongoDB"                     // append these elements at the end of the list


LLEN   myList                                       // get the total number of items in the list


LINDEX myList 0                                     // get the first list item

LINDEX myList -1                                    // get the last list item

LRANGE myList 0 -1                                  // get all items in the selected range (here: all items from start to end)


LPOP   myList                                       // remove and return the first item

RPOP   myList                                       // remove and return the last item
