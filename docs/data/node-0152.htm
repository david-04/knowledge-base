Ajax shortcut methods


jqxhr = $.get(                                      // GET request

    "/url/path",                                    // the URL

    {object|"data"},                                // optional: query string parameters (string will not be url-encoded automatically)

    function onSuccess(data, statusTxt, xhr) { },   // optional: called upon successful completion (statusTxt=="success")

    "{xml|json|script|text|html}"                   // optional: the expected response data type

);


jqxhr = $.getJSON(                                  // GET request for JSON data

    "/url/path",                                    // the URL

    {object|"data"},                                // optional: query string parameters (string will not be url-encoded automatically)

    function onSuccess(data, statusTxt, xhr) { }    // optional: called upon successful completion (statusTxt=="success")

);


jqxhr = $.post(                                     // POST request

    "/url/path",                                    // the URL

    {object|"data"},                                // optional: body paypload data (e.g. $("form").serialize())

    function onSuccess(data, statusTxt, xhr) { },   // optional: called upon successful completion

    "{xml|json|script|text|html}"                   // optional: the expected response data type

);


selection = selection.load(                         // GET a HTML fragment and insert it into the selection

    "/url/path",

    {object|"string"},                              // optional: Query string parameters

    function(responseText, textStatus, xhr) { }     // optional: can access 'this' which corresonds to the currently selected element

);


jqxhr = $.getScript(                                // load and execute JavaScript

    "url",

    function(script,status,xhr) { }                 // optional: executed when the script has been loaded

);

Generic Ajax API


jqxhr = $.ajax("/url/path", {                       // submit an Ajax request


    url           : "http://...",

    method        : '{GET|POST|PUT|ADD|...}',       // the HTTP method

    type          : '{GET|POST|PUT|ADD|...}',       // old name for method (used before jQuery 1.9.0)

    username      : "David",                        // HTTP authentication

    password      : '***',                          // HTTP authentication


    headers       : {'key':'value'},                // HTTP headers

    mimeType      : "application/x-www-form-urlencoded",

    contentType   : "application/json; charset=utf-8",

    processData   : true,                           // encode data into a querystring

    data          : {object|string|array},          // payload data / parameters (e.g. JSON.stringify(data) or  $("form").serialize())

    dataType      : {xml|html|script|json[p]|text}, // data type of the response body

    scriptCharset : '...',                          // characterset for downloaded JavaScript script files

    timeout       : 1000,                           // timeout in milliseconds


    async         : true,

    cache         : false,                          // prevent the browser from caching the response

    crossDomain   : true,                           // allow cross-domain requests

    global        : true,                           // trigger global Ajax event handlers

    ifModified    : false,                          // if true: the request is only successful if the resource was modified since the last request

    isLocal       : true,                           // recognise the current environment as local (file:///)


    beforeSend : function(xhr, options) { },        // set HTTP headers and return true (otherwise it will cancel the request)

    success : function(data, statusText, xhr) { },  // onSuccess event handler

    jsonp : {'onJSONPLoad'|false},                  // name of the JSONP callback function (callback is disabled if set to false)

    jsonpCallback : {'functionName'|function},      // the JSONP callback function

    statusCode : { 404 :  function() { }, ...},     // callback functions for different HTTP response status codes

    error: function(xhr, statusTxt, errString) {},

    complete: function(xhr, statusTxt) { },         // called at the end after onSuccess/onError

});
