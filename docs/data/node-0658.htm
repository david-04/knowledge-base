Syntax


Friend Class Person

    Inherits Object


    Public Property Age As Integer                          ' automatic property


    Private _name As String                                 ' field

    Public Property Name As String                          ' manual property

        Get

            Return _name

        End Get

        Set(value As String)

            _name = value

        End Set

    End Property


    Public Sub New(firstName As String, age As Integer)     ' constructor

        MyBase.New()                                        ' call base class constructor

        Me._name = firstName

        Me.Age = age

    End Sub


End Class
