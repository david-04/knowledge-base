    </fieldset>

</form>

Text fields


<input type           = "{text|password|email|url|  <!-- type of input field/validation                                      -->

                          search|tel|color|hidden}"                     

       autocomplete   = "{on|off}"                  <!-- enable or disable auto-completion                                   -->

       autofocus                                    <!-- set the focus to this field when loading the page                   -->

       disabled                                     <!-- disable this field                                                  -->

       maxlength      = "50"                        <!-- maximum number of characters that the user may type in              -->

       multiple                                     <!-- allow multiple inputs (e.g. emails separated by commas)             -->

       pattern        = "[A-Z]{3}-[0-9]{3}"         <!-- regular expression must match the whole input (i.e. ^...$)          -->

       placeholder    = "(xxx) xxx-xxxx"            <!-- show this placeholder if there is no user input yet                 -->

       readonly                                     <!-- make this field read-only                                           -->

       required                                     <!-- mandatory field, requires user input                                -->

       size           = "10"                        <!-- size of the text field (the actual input can be longer)             -->

       value          = "initial value" />          <!-- initial value to set when loading the page                          -->


<input type           = "{number|range}"            <!-- number is a simple input field, range shows a slider                -->

       min            = "10"

       max            = "25"

       step           = "1" />                      <!-- set for example to 0.1 to allow decimal numbers as well             -->


<input type           = "datetime-local"            <!-- YYYY-MM-DD HH:mm:ss-HH:mm   =>  2016-12-24 18:45-05:00              -->

       type           = "date"                      <!-- YYYY-MM-DD                  =>  2016-12-24                          -->

       type           = "time"                      <!--            HH:mm:ss.ss      =>             18:45:13.07              -->

       type           = "datetime"                  <!-- YYYY-MM-DDTHH:mm:ss         =>  2016-12-24T18:45                    -->

       type           = "month"                     <!-- YYYY-MM                     =>  2016-12                             -->

       type           = "week"                      <!-- YYYY-Www                    =>  2016-W02                            -->

       min            = "2016-12-01"                <!-- must be in the format dictated by the selected "type"               -->

       max            = "2016-12-24"

       step           = "1" />

CSS validation selectors


input:required     {...}                            /* applied only to fields that are marked as mandatory                    */

input:optional     {...}                            /* applied only to fields that are not mandatory                          */

input:valid        {...}                            /* applied only to fields with valid input                                */

input:invalid      {...}                            /* applied only to fields with validation errors                          */

input:in-range     {...}                            /* applied only to fields with values within the min-max-range            */

input:out-of-range {...}                            /* applied only to fields with values outside the min-max-range           */



Custom validators


<{input|textarea} oninput="myValidator(this)" />    <!-- attach an event handler to be called whenever the value changes     -->


function myValidator(input) {

    if (input == "...") {

        input.setCustomValidity("That's  wrong!");  // mark as invalid and set an error message (is displayed by the browser)

    } else {

        input.setCustomValidity("");                // clear any error message if the input is (now) correct

    }

}
