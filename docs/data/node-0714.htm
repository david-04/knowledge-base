Without namespaces


<?xml version="1.0" encoding="utf-8"?>

<xs:schema

    xmlns:xs="http://www.w3.org/2001/XMLSchema">    <!-- do not include any namespace information                        -->


  <xs:complexType name="myType">

    <xs:simpleContent>

      <xs:extension base="xs:string">

        <xs:attribute name="myId" type="xs:int" />

      </xs:extension>

    </xs:simpleContent>

  </xs:complexType>


  <xs:element name="myElement" type="myType"/>


</xs:schema>





<?xml version="1.0" encoding="utf-8"?>

<myElement

    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

    xsi:noNamespaceSchemaLocation="schema.xsd"      <!-- load the schema without referencing any namespace               -->

    checksum="12" />



With namespace


<?xml version="1.0" encoding="utf-8"?>

<xs:schema

    xmlns:xs="http://www.w3.org/2001/XMLSchema"

    targetNamespace      = "my-domain.com"          <!-- the namespace of this XSD's elements and types                  -->

    xmlns                = "my-domain.com"          <!-- this is needed for cross-references within this XSD document    -->

    elementFormDefault   = "qualified"              <!-- the XML must use a namespace prefix for elements                -->

    attributeFormDefault = "unqualified">           <!-- the XML must not use a namespace prefix for attributes          -->


  <xs:complexType name="myType">

    <xs:simpleContent>

      <xs:extension base="xs:string">

        <xs:attribute name="myId" type="xs:int" />

      </xs:extension>

    </xs:simpleContent>

  </xs:complexType>


  <xs:element name="myElement" type="myType"/>      <!-- if xs:schema.xmlns was not set, then the referenced type name   -->

                                                    <!-- would need to be qualified with a namespace prefix              -->

</xs:schema>





<?xml version="1.0" encoding="utf-8"?>

<myElement

    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

    xsi:schemaLocation="my-domain.com schema.xsd"   <!-- import the given XSD file for the specified namespace           -->

    xmlns="my-domain.com"                           <!-- set the default namespace to avoid always prefixing elements    -->

    xmlns:myns="my-domain.com"                      <!-- if attributes must be qualified, a namespace prefix is needed   -->

    myId="12" />                                    <!-- if the XSD said: attributeFormDefault="qualified", then the     -->

                                                    <!-- then the attribute had to be prefixed: myns:myId="12"           -->
