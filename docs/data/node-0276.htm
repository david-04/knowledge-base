Defining content pages


<%@ Page MasterPageFile = "~/MasterPage.Master"                   <!-- base the content page on the given master page -->

         Title          = "HTML page title" ... %>                <!-- override the master page's title  -->

<asp:Content ... ContentPlaceHolderID="head">

</asp:Content>

<asp:Content ... ContentPlaceHolderID="content">                  <!-- use asp:Content to replace the master page's ContentPlaceHolder -->

    <p>This is the page content</p>

</asp:Content>



Accessing master page elements programatically


protected void Page_LoadComplete(object sender, EventArgs e)       // access the master page from Page_LoadComplete as the master's

{                                                                  // Page_Load is invoked only after the content page's Page_Load

    Master.Page.Title = "HTML page title";                         // set the overall HTML head's page title


    TextBox textBox = Master.FindControl("TextBox") as TextBox;    // retrieve a control from the master page

    textBox = (Master as MyMaster).GetTextBox();                   // retrieve a control for which a dedicated getter existis


    textBox = Master.TextBox;                                      // requires <%@ MasterType VirtualPath="~/MyMaster.master" %>

}



Programmatically switching the master page


protected void Page_PreInit(object sender, EventArgs e)            // must be done during the PreInit event

{

    Page.MasterPageFile = "~/MyMaster.master";                     // assign the master page like in the ASPX file

}



Using browser-specific master pages


<%@ Page ... MasterPageFile         = "~/Default.master"           <!-- supported browsers:                                            -->

             Firefox:MasterPageFile = "~/Firefox.master"           <!-- Blackberry   Gateway    iPhone                                 -->

             Safari:MasterPageFile  = "~/Safari.master"            <!-- Chrome       Generic    Opera                                  -->

             Opera:MasterPageFile   = "~/Opera.master"             <!-- Default      IE         Safari                                 -->

             Chrome:MasterPageFile  = "~/Chrome.master" %>         <!-- Firefox      IEMobile   UCBrowser                              -->



Applying a master page project-wide to all content pages


<configuration>                                                    <!-- applies to all content pages (ASP.NET applies some type        -->

  <system.web>                                                     <!-- of detection if it is dealing with a content page or not)      -->

    <pages masterPageFile="~/MyMasterPage.master" />               <!-- can be overridden with the page's MasterPageFile attribute     -->

  </system.web>

  <location path="Admin">                                          <!-- apply the master to all pages in the given folder              -->

    <system.web>

      <pages masterPageFile="~/AdminMasterPage.master" />
