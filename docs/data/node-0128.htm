       value="Toggle" />

</p>

JavaScript API


$("#myCollapsible").collapse("show");                 // show the section

$("#myCollapsible").collapse("hide");                 // hide the section



$("#myCollapsible").collapse("toggle");               // toggle between show and hide

$("#myCollapsible").collapse({toggle: false});


onshow.bs.collapse                                    // before the section is made visible

onshown.bs.collapse                                   // after the section has been made visible

onhide.bs.collapse                                    // before the section is being hidden

onhidden.bs.collapse                                  // after the section has been hidden
