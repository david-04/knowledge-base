public class MyModel                                // the model is a regular C# class with public properties

{

    [Required(ErrorMessage="Name is missing")]      // validation is defined and triggered via annotations

    public string Name { get; set; }                // all properties have public getters and setters

}



App_Start/RouteConfig.cs


public class RouteConfig                            // the routing configuration maps URLs to controllers

{

    public static void RegisterRoutes(

                            RouteCollection routes)

    {

        routes.MapRoute(

            name: "Default",

            url: "{controller}/{action}/{id}",

            defaults: new {                         // when the URL does not contain a controller, action and/or id:

                controller = "Home",                // ... use Controllers/HomeController's

                action     = "Index",               // ... Index method

                id         = UrlParameter.Optional  // ... without passing an ID (???)

            }

        );

    }

}



Controllers/HomeController.cs


public class HomeController : Controller            // the controller

{

   

    [HttpGet]                                       // this method only serves HTTP GET requests

    public ViewResult Index()                       // ... with the routing action "Index"

    {

        ViewBag.title = "Welcome";                  // use the dynamic ViewBag to pass any data to the view


        return View();                              // renders the view Views/Home/Index.cshtml (by convention)

        return View("Thanks");                      // renders the view Views/Home/Thanks.cshtml


    }


    [HttpPost]                                      // this method only serves HTTP POST requests

    public ViewResult Update(MyModel myModel)       // model binding: the request is automatically parsed into a MyModel object

    {

        if (ModelState.IsValid)

        {       

             return View(myModel);                  // render a strongly typed view by passing in an instance of the model

        }

    }

}
