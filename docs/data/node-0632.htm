Required assemblies


MySql.Data.dll



Configuring App.config


<!-- this section is usually already configured in machine.config -->

<system.data>

  <DbProviderFactories>

    <remove invariant="MySql.Data.MySqlClient" />

    <add name="MySQL Data Provider" invariant="MySql.Data.MySqlClient"

         description=".Net Framework Data Provider for MySQL"

         type="MySql.Data.MySqlClient.MySqlClientFactory, MySql.Data, Version=6.9.9.0, Culture=neutral, PublicKeyToken=c5687fc88969c44d" />

  </DbProviderFactories>

</system.data>



Establishing a connecting


Dim connection As DbConnection = DbProviderFactories.GetFactory("MySql.Data.MySqlClient").CreateConnection()

Dim connection As DbConnection = New MySqlConnection()


connection.ConnectionString =

    "Server=127.0.0.1;Database=david;Uid=david;Pwd=test123;"        ' define the connect string in one step


Dim params As New SqlConnectionStringBuilder With                   ' or define the connect string with dedicated attriubtes

{

    .DataSource = "127.0.0.1",

    .InitialCatalog = "david",

    .UserID = "david",

    .Password = "test123",

    .ConnectTimeout = 30

}

connection.ConnectionString = params.ConnectionString


Try                                                                 ' wrap with a try block

    connection.Open()

    ' ...

Finally

    connection.Dispose()                                            ' and dispose manually

End Try


Using connection                                                    ' or let a using block take care of the disposal

    connection.Open()

    ' ...

End Using



SQL statements


Dim command As DbCommand = connection.CreateCommand()               ' simple query

command.CommandText = "select * from customers"

Dim dataReader As DbDataReader = command.ExecuteReader()

While dataReader.Read()

    Console.WriteLine(dataReader("Name"))

End While             


Dim command As DbCommand = connection.CreateCommand()               ' simple update

command.CommandText = "update customers set name='David' ..."

Dim updatedRows As Integer = command.ExecuteNonQuery()


Dim command As DbCommand = connection.CreateCommand()               ' statement with parameters

command.CommandText = "select * from customers where name=@name"

Dim parameter As DbParameter = command.CreateParameter()

parameter.ParameterName = "@name"

parameter.Value = "David"

parameter.DbType = DbType.String

command.Parameters.Add(parameter)

Dim dataReader As DbDataReader = command.ExecuteReader()

While dataReader.Read()

    Console.WriteLine(dataReader("Name"))

End While



Transactions


Dim transaction As DbTransaction = connection.BeginTransaction()    ' start a New transaction


Dim command As DbCommand = connection.CreateCommand()

command.CommandText = "update customers set name='David' ..."

command.Transaction = transaction                                   ' attach transaction to the command

Dim updatedRows As Integer = command.ExecuteNonQuery()


transaction.Commit()                                                ' commit or rollback

transaction.Rollback()



Stored procedures


Dim command As DbCommand = connection.CreateCommand()               ' create a procedure object

command.CommandType = CommandType.StoredProcedure

command.CommandText = "proc_or_func_name"


Dim parameter1 As DbParameter = command.CreateParameter()           ' add an input parameter

parameter1.ParameterName = "name"

parameter1.Value = "David"

parameter1.Direction = ParameterDirection.Input

command.Parameters.Add(parameter1)


Dim parameter2 As DbParameter = command.CreateParameter()           ' add an output parameter

parameter2.ParameterName = "address"

parameter2.Direction = ParameterDirection.Output

command.Parameters.Add(parameter2)


Dim parameter3 As DbParameter = command.CreateParameter()           ' add a parameter for a return value

parameter3.ParameterName = "result"

parameter3.Direction = ParameterDirection.ReturnValue

command.Parameters.Add(parameter3)


command.ExecuteNonQuery()                                           ' execute As "non-query"


Console.WriteLine(command.Parameters("result").Value)               ' retrieve the Return value



Locally cached DataSets


Dim adapter As MySqlDataAdapter = New MySqlDataAdapter(             ' create a data adapter for a particular SELECT statement

    "select * from customers", CType(connection, MySqlConnection))


Dim tableMapping As DataTableMapping =                              ' define name mappings for tables ans columns

    adapter.TableMappings.Add("customers", "Customer")              ' however, it's quite confusing as you frequently

tableMapping.ColumnMappings.Add("name", "Full Name")                ' have to use the unmapped source name anyway


Dim cmdBuilder As New MySqlCommandBuilder(adapter)                  ' automatically create INSERT, UPDATE, ... (only works for tables with primary keys)


adapter.UpdateCommand = New MySqlCommand(                           ' ... or manually create an UPDATE statement

    "update customer set address=@address where name=@name",

    CType(connection, MySqlConnection))

adapter.UpdateCommand.Parameters.Add(                               ' define named parameters

        "@name", MySqlDbType.String, 250, "Full name")

adapter.UpdateCommand.Parameters.Add(

        "@address", MySqlDbType.String, 250, "Address")


Dim dataSet As New DataSet("MySQL Sample DB")                       ' create a DataSet in memory

adapter.Fill(dataSet, "customers")                                  ' let the adapter load the whole table from the database


adapter.Update(dataSet, "customers")                                ' write all changes of the given table back to the database
