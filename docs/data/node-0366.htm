Rule priority

p.error             { color: red !important; }      /* use !important to override settings of more specific rules           */

Visibility


visibility:                  {visible               /* show the element                                                     */

                             |hidden                /* show an empty placeholder only                                       */

                             |collapse};            /* hide a table row without affecting the table layout                  */



Overflow


overflow:                    {visible               /* no clipping, show content outside of the element                     */

                             |hidden                /* clip the overflow and don't show it at all                           */

                             |scroll                /* show a scrollbar                                                     */

                             |auto};                /* show a scrollbar only if clipping occurred                           */

overflow-{x|y}:              {visible|hidden|...};  /* apply the overflow policy to only one direction                      */



Inheriting property values


padding :                    inherit;               /* apply the parent's style (if not inherited automatically)            */



Calculations


width:                       calc(67% - 10px)       /* calculation incorportating relative and absolute measures            */

Image filters


filter: {blur(10px)                                 /* blur with the given radius (default: 0px)                            */

        |brightness(50%)                            /* 0% = black, 100% = original, >100% brighter                          */

        |contrast(50%)                              /* 0% = max contrast [black], 100% = original, >100 less contrast       */

        |drop-shadow(10px 20px 5px 0px red)         /* box shadow (x_offset, y_offset, blur_radius, spread, colour)         */

        |grayscale(50%)                             /* 0% = original, 100% = completeley greyscale                          */

        |hue-rotate(30deg)                          /* 0° = 360° = original                                                 */

        |invert(50%)                                /* 0% = original, 100% = fully inverted                                 */

        |opacity(50%)                               /* 0% = transparent, 100% = opaque                                      */

        |saturate(50%)                              /* 0% = completely unsaturated, 100% = original, >100% super-saturated  */

        |sepia(5%)                                  /* 0% = original, 100% = sepia                                          */

        |url(my-svg-filter.svg#element-id)};        /* URL (and optionally the anchor) of a SVG filter                      */
