Messaging


iFrame.postMessage("message", "*");                 // post a message to iFrame

iFrame.postMessage("message", "http://domain.com"); // post the message only if it was loaded from mydomain.com


function iFrame_onMessage(event) {

    event.data                                      // contains the "message"

    event.origin                                    // the origin URL (or domain?) from where the message came

    event.source                                    // the window that sent the message (only a few properties are accessible)

}
