Grid container


display:                 {grid|inline-grid|subgrid};/* mark as block/inline/nested grid container                            */


grid-template-columns:   10px 20% auto              /* specify the columns (and rows) that form the grid                     */

grid-template-rows:      {10% auto auto 200px       /* specify the columns (auto-columns consume all the remaining space)    */

                         |[start] 10% [main] 100px  /* optionally give symbolic names to the [column separators]             */

                         |repeat(3, 20px)           /* use the repeat-function to create multipe equal column                */

                         |40px 1fr 2fr};            /* fraction size: remaining space is split in 1/3 and 2/3                */


grid-template-areas:     "header header header"     /* give symbolic names names to cells (each row is one string)           */

                         "main   .      sidebar"    /* use the dot (.) for unnamed cells                                     */

                         "footer footer sidebar"    /* reuse the same name to join cells (colspan/rowspan)                   */


grid-column-gap:         10px;                      /* space between the columns/rows (cell-spacing)                         */

grid-row-gap:            10px;


justify-items:           {start|end                 /* horizontal alignment of the cell content (left, right,                */

                         |center|stretch};          /* justified, stretched)                                                 */

align-items:             {start|end                 /* vertical alignment of the cell content (top, bottom,                  */

                         |center|stretch};          /* justified, stretched)                                                 */


justify-content:         {start|end|center|stretch  /* distribute the columns if they don't fill the whole container's size  */

                         |space-around

                         |space-between

                         |space-evenly};

align-content:           {start|end|center|stretch  /* distribute the rows if they don't fill the whole container's size     */

                         |space-around

                         |space-between

                         |space-evenly};

                                                   


grid-auto-columns:       10px;                      /* width of columns outside the grid that have attached child elements   */

grid-auto-rows:          10px;


grid-auto-flow:          {row                       /* when adding unspecified cells: fill left-to-right (line by line)      */

                         |column                    /* fill top-to-bottom (column by column)                                 */

                         |dense};                   /* fill holes                                                            */


grid:                    grid-template-rows         /* shortcut: not supported by any browser yet                            */

                         grid-template-columns ...



Grid children


grid-{row|column}-start: {2                         /* place in the second row/column                                        */

                         |start                     /* start at the line with the symbolic name (grid-template-rows/columns) */

                         |auto};                    /* auto-select the best row/column                                       */


grid-{row|column}-end:   {5                         /* end in the fifth row/column                                           */

                         |span 2                    /* span over two rows/columns                                            */

                         |main                      /* and at the line with the given symbolic name                          */

                         |span main};               /* presumably the same as above                                          */


grid-{row|column}:       myLine / span 2;           /* shortcut for grid-{row|column}-start / grid-{row|column}-end          */


grid-area:               {footer                    /* reference a symoblic name (see grid-template-areas above)             */

                         |1 / main / 1 / span 2};   /* or specify column-start/column-end/row-start/row-end                  */


justify-self:            {start|end|center|stretch};/* horizontally align this child within the container's cell             */

align-self:              {start|end|center|stretch};/* vertically align this child within the container's cell               */
