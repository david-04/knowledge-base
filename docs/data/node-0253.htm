Detecting postbacks


Page.IsPostBack                                                 // check if any kind of postback occurred

null != PreviousPage && PreviousPage.IsCrossPagePostBack        // check if a cross-page postback has occurred

Page.PreviousPage is PlaygroundASP.Default                      // check if the postback comes from Default.aspx



Posting back cross-page


<%@ Page ... Inherits    = "PlaygroundASP.Default" %>           <!-- the main page initiates a cross-page postback             -->

<asp:TextBox ID          = "Name"

             Runat       = "server" />

<asp:Button  Runat       = "server"

             PostBackUrl = "~/Greeting.aspx" />                 <!-- redirect the postback to another page                     -->




<%@ Page ...         Inherits    = "PlaygroundASP.Greeting" %>  <!-- the greeting page receives the cross-page postback        -->

<%@ PreviousPageType VirtualPath = "~/Default.aspx"  %>         <!-- or use the class name as TypeName="PlaygroundASP.Default" -->


TextBox textBox = (TextBox)PreviousPage.FindControl("Name");    // retrieve the control from the originating paet

this.Name.Text = textBox.Text;                                  // copy data to the current page

this.Name.Text = PreviousPage.Name;                             // requires properties to implemented on the originating page



Using cookies


HttpCookie myCookie = new HttpCookie("MyCookie");               // create a new cookie

myCookie.Expires = DateTime.Now.AddMonths(3);                   // set the expiration date

myCookie.Value = "Cookie value";                                // set the value

Response.Cookies.Add(myCookie);                                 // send the cookie to the browser

Request.Cookies.Get("MyCookie").Value                           // retrieve the cookie's value



Using auto-generated sessions


<configuration>

  <system.web>

    <sessionState mode       = "InProc"                         <!-- Off|InProc|StateServer|SQLServer|Custom                   -->

                  cookieless = "UseUri"                         <!-- UseCookies, UseUri                                        -->

                  cookieName = "MyCookie"

                  timeout    = "20" />                          <!-- session times out after 20 minutes                        -->



<%@ Page ... EnableSessionState="True"      %>                  <!-- default: allow read and write access                      -->

<%@ Page ... EnableSessionState="False"     %>                  <!-- disable session state alltogether                         -->

<%@ Page ... EnableSessionState="ReadOnly"  %>                  <!-- read-only access prevents thread-locking                  -->


Session["key"] = "Started at " + DateTime.Now.ToString();       // set a session attribute

this.Label1.Text = (string) Session["key"];                     // retrieve a session attribute


Response.Write(Response.ApplyAppPathModifier("~/Test.aspx"));   // ensure that the session ID is added to the URL (in UseUri mode)



Sharing data within one request


HttpContext.Current.Items["key"] = "data";                      // store date for this HTTP request only

string data = (string) HttpContext.Current.Items["key"];        // retrieve data



Storing global data in the application object


Application.Lock();                                             // lock the global application object

Application["LastRequest"] = DateTime.Now;                      // store data in the application object

DateTime latestRequest = (DateTime)Application["LastRequest"];  // retrieve data

Application.UnLock();                                           // release the lock



Enabling and disabling the ViewState


<configuration><system.web><pages enableViewState="false">      <!-- disable the ViewState for the whole project              -->


<%@ Page ... EnableViewState="False" %>                         <!-- disable the ViewState for a single page                  -->


<asp:Label ... EnableViewState="False" />                       <!-- disable the ViewState for a single control               -->


<configuration><system.web><pages enableViewState="true">       <!-- allow the ViewState in general for the whole project     -->

<%@ Page ... EnableViewState="true" ViewStateMode="Disabled" %> <!-- allow the ViewState per page but disable it by default   -->

<asp:Label ... ViewStateMode="Enabled" />                       <!-- enable the ViewState for selected controls only          -->
