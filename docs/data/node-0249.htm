Adding JavaScript programatically


Page.ClientScript.RegisterClientScriptInclude(              // register an external JS file to be linked to

    "MyJavaScript",                                         // unique name

    "./Scripts/MyJavaScript.js"                             // filename and path (no expansion of ~ here!)

);


Page.ClientScript.RegisterClientScriptBlock(                // register code placed on top of the page

    this.GetType(),                                         // attach this script to the current page

    "MyScript",                                             // give the function a unique name

    @"function hello() { alert('Hello world'); }",          // the javascript code as a string

    true                                                    // encapsulate the code in a <script> tag

);


Page.ClientScript.RegisterStartupScript(                    // register code placed at the bottomo of the page

    this.GetType(),                                         // attach this script to the current page

    "MyScript",                                             // give the function a unique name

    @"function hello() { alert('Hello world'); }",          // the javascript code as a string

    true                                                    // encapsulate the code in a <script> tag

);

Using JavaScript aliases


public class Global : System.Web.HttpApplication

{

    protected void Application_Start(object sender, EventArgs e)

    {

        ScriptManager.ScriptResourceMapping.AddDefinition(  // register jQuery in the global application file

            "jquery",                                       // symbolic name (alias)

            new ScriptResourceDefinition

            {

                Path = "~/Scripts/jquery-3.1.0.min.js"      // file reference to the actual JavaScript file

            }

        );

    }

}


<asp:ScriptManager ID="ScriptManager" runat="server">       <!-- embed a script manager in the page (or in the master)   -->

    <Scripts>

        <asp:ScriptReference Name="jquery" />               <!-- reference the JavaScript's symbolic name from above     -->

    </Scripts>

</asp:ScriptManager>



Budling


PM> Install-Package Microsoft.AspNet.Web.Optimization


public class Global : System.Web.HttpApplication

{

    protected void Application_Start(object sender,

                                     EventArgs e)

    {

        BundleTable.Bundles.Add(                            // define a bundle

            new StyleBundle("~/BundledStyleSheets")         // virtual name (use ScriptBundle to bundle JavaScript)

            .IncludeDirectory("~/Styles", "*.css")          // embed these files

        );

    }

}


<webopt:bundlereference runat = "server"                    <!-- add a link to the bundled CSS files         -->

                        path  = "~/BundledStyleSheets" />   <!-- reference the bundle with its virtual name  -->
