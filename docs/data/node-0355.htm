transform:        rotate(0.5turn);                  /* specify the angle in in turns (0° == 360° == 1turn)            */

transform:        scale(1.5, 2.0);                  /* scale with and height with different factors                   */

transform:        scaleX(1.5);                      /* scale only the width                                           */

transform:        scaleY(1.5);                      /* scale only the height                                          */

transform:        skewY(3.1415rad);                 /* shear the image vertically                                     */

3D projection area


.container {                                        /* the 3D space must be defined on the surrounding container      */


    perspective:         500px;                     /* distance between the viewer and the screen                     */

                                                    /* this influences how pronouned the transformations appear       */


    perspective-origin:  50% 50%;                   /* the viewer's eyes are centered at this position within the     */

    perspective-origin:  {left|center|right }       /* the wrapping container                                         */

                         {top |center|bottom};      

                                                   

    transform-style:     preserve-3d;               /* use this property to inherit the 3D space (by deafult only the */

}                                                   /* the direct children can be transformed in 3D)                  */


.elementToTransform {                               /* the actual transformations are applied to the children         */


    transform:           perspective(500px);        /* the parent's perspective can be overridden via a function      */


    backface-visibility: hidden;                    /* hide the back side (e.g. if rotating 180° in 3D)               */

}



3D transformations


transform: translateZ(100px);                       /* move towards the viewer (negative values to move away)         */

transform: translate3d(10px,50px,100px);            /* move to the right, to the bottom and away from the viewer      */


transform: rotateX(45deg);                          /* rotate falling backwards (nagetive values to fall forward)     */

transform: rotateY(45deg);                          /* rotate like opening a jar (nagetive values for closing)        */

transform: rotateZ(45deg);                          /* rotate clockwise [2D] (nagetive values for counterclockwise)   */

transform: rotate3d(45deg, 45deg, 45deg);           /* rotate X, Y and Z at the same time                             */


transform: scaleZ(1.5);                             /* scale (similar to translateZ) but without perspective shift    */

transform: scale3d(1.5, 1.5, 1.5);                  /* scale x, y and z at the same time                              */
