public class MyRazorViewEngine : RazorViewEngine    // derive a customised Razor view engine

{

    public MyRazorViewEngine()

    {

        ViewLocationFormats = new string[]          // these properties are used as patterns to search for views

        {                                           // {0} = view name

            "~/Views/{1}/{0}.cshtml",               // {1} = controller name

            "~/Views/Common/{0}.cshtml"             // {2} = area name

        };

        MasterLocationFormats

        PartialViewLocationFormats

        AreaViewLocationFormats

        AreaMasterLocationFormats

        AreaPartialViewLocationFormats

    }

}


public class MvcApplication : HttpApplication       // register the customised view engine in Global.asax

{

    protected void Application_Start()

    {

        ViewEngines.Engines.Clear();                // remove the system's default view engine

        ViewEngines.Engines.Add(                    // ... and add the customised one

                    new MyRazorViewEngine());

    }

}
