Implementing an HTTP handler


public class MyHttpHandler : IHttpHandler                           // file extension: ashx

{

    public void ProcessRequest(HttpContext context)                 // serve a reqest

    {

        context.Response.ContentType = "text/plain";                // set the content type

        context.Response.Write("Hello World");                      // flush the actual content

    }


    public bool IsReusable { get {return false; } }

}

public class MyHttpHandler : HttpTaskAsyncHandler                    // asyncronous version

{

    public async override Task ProcessRequestAsync(HttpContext ctx)

    {

        int result = await Task.Run<int>(() => { return 0; });       // wait for asynchronous activity

        ctx.Response.Write("...");                                   // prepare the result

    }

}



Associating an HTTP handler with a file extension


<configuration>

  <system.webServer>

    <handlers>                                                      

      <add name = "MyHttpHandler"

           verb = "*"                                                <!-- allow all request types (GET, POST, ...)              -->

           path = "*.ping"                                           <!-- path and name (or file extension) to match            -->

           type = "PlaygroundASP.MyHttpHandler, PlaygroundASP" />    <!-- namespace.class, assembly (or APP_CODE if internal)   -->

    </handlers>

  </system.webServer>

  <system.web>

    <httpHandlers>                                                   <!-- this section is only required for IIS 6 and above     -->

      <add verb = "*"

           path = "*.ping"

           type = "PlaygroundASP.MyHttpHandler, PlaygroundASP" />

    </httpHandlers>



Implementing an HTTP module


public class AppendMessage : IHttpModule                             // regular .cs class file

{

    private HttpApplication _application = null;


    public void Dispose() { }


    public void Init(HttpApplication context)

    {

        _application = context;                                      // store the application for later use


        context.EndRequest += context_EndRequest;                    // register a synchronous handler


        EventHandlerTaskAsyncHelper helper =                         // register an asynchronous handler

            new EventHandlerTaskAsyncHelper(context_EndRequestAsync);

        context.AddOnEndRequestAsync(                              

                helper.BeginEventHandler, helper.EndEventHandler);

    }


    void context_EndRequest(object sender, EventArgs e)              // synchronous handler

    {

        _application.Context.Response.Write("test");

    }


    public async Task context_EndRequestAsync(object caller,         // asynchronous handler

                                              EventArgs e)

    {

        int result = await Task.Run<int>(() => { return 0; });

        _application.Context.Response.Write("test");

    }

}



Registering an HTTP module


<configuration>

  <system.webServer>

    <validation validateIntegratedModeConfiguration="false"/>        <!-- only for IIS 6 and above                              -->

    <modules>

      <add name = "AppendMessage"

           type = "PlaygroundASP.AppendMessage, PlaygroundASP" />    <!-- namespace.class, assembly (or APP_CODE if internal)   -->

    </modules>

  </system.webServer>

  <system.web>

    <httpModules>                                                    <!-- only for IIS 6 and above                              -->

      <add name = "AppendMessage"

           type = "PlaygroundASP.AppendMessage, PlaygroundASP" />

    </httpModules>
