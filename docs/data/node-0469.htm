Navigation and history


window.location.href                                // http://www.domain.com:80/path/file.html?id=1#overview

window.location.protocol                            // http:

window.location.host                                //        www.domain.com:80

window.location.hostname                            //        www.domain.com

window.location.port                                //                       80

window.location.pathname                            //                         /path/file.html

window.location.search                              //                                        ?id=1

window.location.hash                                //                                             #overview


window.location.reload(true);                       // true=load from server, false=load from cache (if possible)

window.location.replace("www.google.com.au/");      // navigate to another page WITHOUT a new browser history entry

window.location.assign("www.google.com.au/");       // navigate to another page ADDING a new browser history entry

window.location      = "www.google.com.au/";

window.location.href = "www.google.com.au/";

window.location.hash = "#section1";                 // every location component can be modified


window.history.back();                              // navigate back (the number of steps can be passed as well)

window.history.forward();                           // navigate forward (the number of steps can be passed as well)

window.history.go(-2);                              // navigate back two pages

window.history.go("google");                        // navigate to the oldest history site containing the term "google"


window.history.length == 0                          // page was opened directly (or via a bookmark)



Manual history management


window.history.pushState(                           // HTML5: save the current page state to return later

    { myProperty: "value" },                        // optional: the current page state's data (set to null if not applicable)

    "title",                                        // optional the title is not used (set to null if not applicable)

    "./index.html?...");                            // the URL to show in the browser (must be the same protocol/domain)

                                                    // the browser will not reload the page, but it might be bookmarked

                                                    // ... and therefore must still be a valid URL backed by the server


window.history.replaceState(                        // update the current state

    { myProperty: "newValue" }, "title");


window.onpopstate = function (event) {              // triggered by the back-button

    event.state.myProperty                          // restore the state from the event's state data (might be null)

}
