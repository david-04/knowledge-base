onshow.bs.dropdown                                    // before the dropdown is being opened

onshown.bs.dropdown                                   // after the dropdown has been opened

onhide.bs.dropdown                                    // before the dropdown is being closed

onhidden.bs.dropdown                                  // after the dropdown has been closed
