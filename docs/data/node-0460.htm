Pop-ups


var popup = window.open(                            // pop-ups are only allowed on user interaction and not during the page load

    "http://google.com.au/",                        // the URL to be loaded in the window

    "_top",                                         // target window/frame (a pop-up is created if it doesn't exist yet)

      "fullscreen=no,"                              // IE only: maximise window

    + "top=10,left=50,height=100,width=100,"        // size and position

    + "resizable=yes,scrollbars=yes,"               // make the window resizeable and show scrollbars

    + "location=yes,menubar=yes,toolbar=yes,"       // show the address, menu or toolbar

    + "status=yes",                                 // show the status bar

    true                                            // when loading in the same window (no pop-up): add new page to browser's history

);


popup.{resize|move}{By|To}                          // resizing and moving of popus is usually allowed

popup.opener                                        // reference to the window that opened the pop-up

popup.close();                                      // pop-ups can be closed as well
