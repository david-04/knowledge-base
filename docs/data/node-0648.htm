Chaining constructors


Public Sub New(name As String)

    Me.name = name

End Sub


Public Sub New(name As String, address As String)

    Me.New(name)                                    ' call another constructor of the same class

    Me.address = address                            ' process additional parameters

End Sub



Initialising the base class


Public Sub New(name As String, address As String)

    MyBase.New()                                    ' initialise the base class

    Me.address = address                            ' process additional parameters

End Sub

    Public name As String

    Public city As String


    Public Sub New()

    End Sub

End Class
