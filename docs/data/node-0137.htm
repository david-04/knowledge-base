JavaScript API


$('#myTab a[href="#profile"]').tab('show');           // select a tab specified by its target

$("#myTab a:first").tab('show');                      // select the first tab

$("#myTab a:last").tab('show');                       // select the last tab

$("#myTab li:eq(2) a").tab('show');                   // select the thir tab (zero-based indexing)


onshow.bs.tab                                         // before a tab is selected

onshown.bs.tab                                        // after a tab has been selected

onhide.bs.tab                                         // before a tab is deselected

onhidden.bs.tab                                       // after a tab has been deselected


$("#myTab a:first").on("shown.bs.tab", function (e) {

    e.target                                          // the newly selected tab

    e.relatedTarget                                   // the previously selected tab

});
