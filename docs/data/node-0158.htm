Plugins


$.fn.extend({                                       // register one or more custom plugin functions

    myPlugin: function (color) {                    // each function can receive custom parameters

        this.css({                                  // the current selection is passed in as the "this" jQuery object

            'color': color,                         // use jQuery methods like .css() to manipulate the selection

            'font-weight': 'bold'

        });

        return this;                                // always return this

    },

    myOtherPlugin: function () { ... }              // add as many functions as required

});


$("div").myPlugin("red").show();                    // chain custom plugin functions along with the standard jQuery methods
