Functions


@function myFunction($n) {                          /* define a custom function with parameters                             */

    $result: $n * ($n+1) / 2;                       /* perform calculations                                                 */

    @return $result;                                /* return the result                                                    */

}


#myElement {

    width: myFunction(5);                           /* invoke the function                                                  */

    height: myFunction($n:10);                      /* can optionally use named paramters                                   */

}
