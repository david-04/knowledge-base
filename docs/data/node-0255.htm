Restricting page access

aspacl aspaccesscontrol asppermissions asprestrictions asppagerestrictions asppageaccess aspprotectpages


<configuration>

  <location path="Admin">                            <!-- optional element: applies to the Admin subfolder only     -->

    <system.web>

      <authorization>                                <!-- the first matching rule applies                           -->

        <allow roles="Admins" />                     <!-- grant access to all users with the Admins role            -->

        <allow users="David,Admin" />                <!-- grant access to particular users                          -->

        <allow verbs="GET" />                        <!-- HTTP method (GET, POST, HEAD, DEBUG)                      -->

        <deny users="*" />                           <!-- deny access for everyone                                  -->

        <deny users="?" />                           <!-- deny access for unauthenticated users                     -->

      </authorization>                               <!-- if no rule matches, access is granted                     -->

    </system.web>

  </location>

</configuration>

<configuration>

  <system.web>

    <anonymousIdentification enabled       = "true"              <!-- track anonymous users through cookies                     -->

                             cookieless    = "UseCookies"        <!-- use cookies (or set to UseUri)                            -->

                             cookieTimeout = "1440"              <!-- let cookies expire after one day                          -->

                             cookieName    = "MyWebApp" />       <!-- application specific cookie name                          -->

    <profile defaultProvider="DefaultProfileProvider">

      <properties>                                               <!-- define the profile's properties here                      -->

        <add name           = "FullName"                         <!-- property name                                             -->

             allowAnonymous = "true"                             <!-- is maintained for anonymous users as well                 -->

             defaultValue   = "Anonymous User"                   <!-- default value                                             -->

             readOnly       = "false"                            <!-- grant read or read/write access                           -->

             type           = "System.String"/>                  <!-- optional: System.String is the default and can be omitted -->

             type           = "System.Collections.Generic.List`1[System.String]"/>

        <group name="Identification">                            <!-- properties can be grouped (nested)                        -->

          <add name="PassportNumber"/>

          <add name="DateOfBirth" type="System.DateTime"/>       <!-- custom classes are allowed (if serailisable)              -->

        </group>

Manual profiles (Web Application projects)


<configuration>

  <system.web>

    <anonymousIdentification enabled    = "true"                 <!-- optional: track anonymous users through cookies           -->

                             cookieless = "UseCookies"

                             cookieName = "MyWebApp"/>           <!-- optional: set the cookie name                             -->

    <profile defaultProvider = "DefaultProfileProvider"

             inherits        = "MyProject.MyPackage.MyProfile">  <!-- link the profile to a custom profile class                -->



public class MyProfile : ProfileBase                             // derived from ProfileBase

{

    public static MyProfile GetUserProfile(string username)      // get a selected user's profile

    {

        return Create(username) as MyProfile;

    }


    public static MyProfile GetUserProfile()                     // get the current user's profile

    {

        if (null != Membership.GetUser())                        // authenticated user

        {

            return Create(Membership.GetUser().UserName) as MyProfile;

        }

        else                                                     // anonymous user

        {

            return Create("anonymous", false) as MyProfile;

        }

    }


    [SettingsAllowAnonymous(true)]                               // supported profile properties

    public string FullName

    {

        get { return base["FullName"] as string; }

        set { base["FullName"] = value; }

    }


    [SettingsAllowAnonymous(true)]                               // generics and collections can be used as well

    public List<string> Likes

    {

        get { return base["Likes"] as List<string>; }

        set { base["Likes"] = value; }

    }

}



MyProfile profile = MyProfile.GetUserProfile();                  // retrieve the current user's profile

if (null != profile)

{

    this.UserNameLabel.Text = profile.FullName;                  // access data

    profile.Likes = new List<string>{ "money", "summer"};        // assign new values

    profile.Save();                                              // important: must save explicitely

}



Profile migration and maintenance


public class Global : System.Web.HttpApplication

{

    public void Profile_OnMigrateAnonymous(object sender,       // this event is fired when a previously anonymous users logs in

                           ProfileMigrateEventArgs args)

    {

        ProfileCommon oldProfile =

                 Profile.GetProfile(args.AnonymousID);          // retrieve the anonymous profile

        Profile.MyProperty = oldProfile.MyProperty;             // copy the relevant data to the current user's profile

        ProfileManager.DeleteProfile(args.AnonymousID);         // delete the anonymous profile

        AnonymousIdentificationModule.ClearAnonymousIdentifier(); 

        Membership.DeleteUser(args.AnonymousID, true);

    }

}


ProfileManager.DeleteInactiveProfiles(                          // cleanup profiles

    ProfileAuthenticationOption.Anonymous,                      // ... only for anonymous users

    DateTime.Now.AddDays(-30)                                   // ... which are at least 30 days old

);
