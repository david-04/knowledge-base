Triggers


trigger T1 on Account(before delete,                // triggers are not embedded into classes

                      after delete,                 // supports: before+after insert/update/delete and after undelete

                      after undelete) {             // (merge, upsert, undelete map to correponding insert/update/delete statements)



    if(Trigger.isBefore && Trigger.isDelete){

        Trigger.old[0].addError('Error');           // add errors to selected records to prevent the DML from completing

        MyClass.myStaticMethod(Trigger.new);        // call a static method and pass an array of Accounts (Trigger.new is an array)

    }

}



Properties


Trigger.isExecuting                                 // was this code invoked by a trigger?

Trigger.isInsert                                    // is an insert trigger?

Trigger.isUpdate                                    // is an update trigger?

Trigger.isDelete                                    // is a delete trigger?

Trigger.isBefore                                    // is a before trigger?

Trigger.isAfter                                     // is an after trigger?

Trigger.isUndelete                                  // is an undelete trigger?

Trigger.new                                         // list of the new versions (insert/update/undelete only)

Trigger.newMap                                      // a map of IDs to the new versions

Trigger.old                                         // list of the old versions (update/delete only)

Trigger.oldMap                                      // a map of IDs to the old versions

Trigger.size                                        // total number of records (sum of old and new)
