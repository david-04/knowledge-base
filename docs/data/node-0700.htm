XSD without namespace


<?xml version="1.0" encoding="utf-8"?>

<root

    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

    xmlns:noNamespaceSchemaLocation="schema.xsd"    <!-- reference the schema file without declaring a namespace       -->

>



XSD with namespace


<?xml version="1.0" encoding="utf-8"?>

<myns:data                                          <!-- must bind the XML element itself to the namespace prefix      -->

    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

    xsi:schemaLocation="http://me.com/ schema.xsd"  <!-- link the namespace to the specific schmema file (location)    -->

    xmlns:myns="http://me.com/"                     <!-- bind the namespace to a prefix (or use it as the default)     -->

>



DTD from external file


<?xml version="1.0" encoding="utf-8"?>

<!DOCTYPE myRootElement SYSTEM "my.dtd">            <!-- import my.dtd and only allow myRootElement to be the root     -->



DTD inline


<?xml version="1.0" encoding="utf-8"?>

<!DOCTYPE myRootElement [                           <!-- define the DTD inline within the XML file                     -->

  <!ELEMENT myRootElement   (myChildElement)>

  <!ELEMENT myChildElement     (#PCDATA)>

]>



DTD inline + external file


<?xml version="1.0" encoding="utf-8"?>

<!DOCTYPE myRootElement SYSTEM "external.dtd" [     <!-- load an external DTD file                                     -->

  <!ELEMENT myAddonElement (#PCDATA)>               <!-- ... and define additional elements locally                    -->

]>
