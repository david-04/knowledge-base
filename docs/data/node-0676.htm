Single property animation


Dim doubleAnimation As New DoubleAnimation With {

    .From = 1.0,

    .To = 0.0,

    .Duration = New Duration(TimeSpan.FromSeconds(4))

}

AddHandler doubleAnimation.Completed, Sub(o, s) ...                    ' callback when animation has finished

Me.myButton.BeginAnimation(Button.OpacityProperty, doubleAnimation)



Animated transformations


Dim doubleAnimation As New DoubleAnimation() With {.From = 0, .To = 360}

Dim rotateTransform As New RotateTransform()

Me.myButton.RenderTransform = rotateTransform

rotateTransform.BeginAnimation(RotateTransform.AngleProperty, doubleAnimation)
