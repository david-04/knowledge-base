Framesets


<!DOCTYPE HTML PUBLIC                               <!-- document type for HTML 4.01 framesets (HTML5 does not support framesets)  -->

    "-//W3C//DTD HTML 4.01 Frameset//EN"

    "http://www.w3.org/TR/html4/frameset.dtd">


<frameset {rows|cols}="250,25%,*">                  <!-- can be split by rows or columns sized in pixels, percent or grab-all      -->

    <frame src          = "frame_a.htm">

    <frame src          = "frame_b.htm">

    <frame src          = "frame_c.htm"             <!-- the file to show within the frame                                         -->

           frameborder  = "{0|1}"                   <!-- show (or hide) the frame's borders                                        -->

           longdesc     = "./frame_desc.html"       <!-- a page with a descriptio of the frame's "src" content                     -->

           marginheight = "10"                      <!-- top and bottom margin of the frame                                        -->

           marginwidth  = "15"                      <!-- left and right margin of the frame                                        -->

           name         = "myFrame"                 <!-- name of the frame (referenced by links via the "target" attribute)        -->

           scrolling    = "{yes|no|auto}"           <!-- enable or disable the scrollbars                                          -->

           noresize     >                           <!-- disable the resizing of the frames                                        -->

    <noframes>Sorry ...</noframes>                  <!-- fallback content to display if the browser doesn't support frames         -->

</frameset>



iFrames


<iframe src     = "my-page.html"                    <!-- show the given page within the iFrame                                     -->

        srcdoc  = "<b>hello world</b>"              <!-- show this content in the iFrame (overrides the src attribute)             -->

        width   = "400"                             <!-- width and height of the iFrame (in pixels)                                -->

        height  = "200"

        name    = "myIFrame"                        <!-- name of the iFrame (referenced by links via the "target" attribute)       -->

        sandbox                                     <!-- run in a sandbox and apply all restrictions                               -->

        sandbox = "allow-forms"                     <!-- enable form submission                                                    -->

        sandbox = "allow-pointer-lock"              <!-- enable APIs                                                               -->

        sandbox = "allow-popups"                    <!-- enable popups                                                             -->

        sandbox = "allow-same-origin"               <!-- allow the iframe's content to be treated as being from the same origin    -->

        sandbox = "allow-scripts"                   <!-- enable scripts                                                            -->

        sandbox = "allow-top-navigation" >          <!-- allow the iframe's content to navigate its top-level browsing context     -->

    Sorry...                                        <!-- fallback content for browsers that don't support iFrames                  -->

</iframe>
