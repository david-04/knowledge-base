Generating links


@Html.ActionLink(                                   // create an application-internal link

    "Click here",                                   // link text

    "Index",                                        // name of the action method to target

    "Home",                                         // optional: name of the controller to target

    new {                                           // optional: segment variables (unknown variables become query strings)

        username = "David"

    },

    new                                             // optional: add HTML attributes to the generated <a> element

    {

        id       = "myLinkId",

        @class   = "myCSSClass"

    }

)


@Html.ActionLink(                                   // create a fully customised link

    "Click here",

    "Index", "Home",                                // controller and action method

    "https", "mydomain.com", "myFragment",          // protocol, domain, fragment (...#myFragment)

    new { username = "David" },                     // route segment values (optional, can be set to null)

    new { id = "myLinkId"    }                      // HTML attributes (optional)

)          


@Html.ActionLink(                                   // create a link to another area

    "Click here",

    "Index",

    new { area = "Admin" }                          // name of the area (set to "" in order to reference the global scope)

)


@Html.RouteLink(                                    // create a link for a specified route name

    "Click me",

    "MyRouteName",

    "Index", "Customer"

)



Generating URLs


@Url.Action("Index", "Home");                        // generate an URL for the selected action method of the specified controller

@Url.Action("Index", new {username="David"});        // generate an URL for the selected action method

@Url.Content("~/Content/Site.css");                  // application-relative path to static content

@Url.RouteUrl(new{controller="Home",action="List"}); // generate an URL for the selected controller and action method
