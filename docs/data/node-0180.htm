Overview


                                                                                  +-store-+

actionCreator --[creates]--> action --[dispatched]--> reducer <--[manipulates]--> | state |--[selector]--> reactProps

      |                        |                         |                        +-------+      |

      |                        |                         |                          |            |

      |                        |                         |                          |            +--> transfers parts of the state to React props

      |                        |                         |                          |

      |                        |                         |                          +--> a simple/nested JavaScript object encapsulated in a store

      |                        |                         |                               (there is exactly one store in each application)

      |                        |                         |

      |                        |                         +--> a function that receives an action and incorporates the contained data into the state

      |                        |

      |                        +--> a simple JavaScript object with a "type" property (name of the action as string) and payload data

      |

      +--> a function that creates and returns an action object (but does not dispatch it)




Setup


> npm install --save redux                          // Redux core components


> npm install --save react-redux                    // optional components

> npm install --save-dev redux-devtools



Usage


var myInitialState = {};                            // set the application's initial state


function myActionCreator(name) {                    // define an action creator

    return {

        type: 'MY_ACTION',                          // must have a property named "type" which classifies the action type

        payloadDAta: {id: 1, message: 'Hi'}

    }

}


function myReducer(                                 // define a reducer

    state = myInitialState, action

) {

    switch (action.type) {

        case 'MY_ACTION':

            return Object.assign({}, state,         // incorporate the action payload into the state

                {user: action.user });

        default:

            return state                            // always return the current state for unknown actions

    }

}


var myStore = createStore(                          // attache the reducer to a new store

    myReducer, myInitialState

);


myStore.dispatch(myActionCreator('David'));         // dispatch actions to the store (reducer)
