Forms


<form #myForm="ngForm"                              <!-- give the form a symbolic name that can be referenced later on  -->

      (ngSubmit)="onSubmit()" >                     <!-- event handler to intercept the form's submission               -->


    <input [(ngModel)]="person.name"                <!-- two-way data binding                                           -->

           name="fullName" />                       <!-- the "name" attribute is mandatory                              -->


    <select id="nationality" required>                                             

        <option *ngFor="let c of countries"         <!-- generate list-box items from data collections                  -->

                [value]="c.id">

            {{c.name}}

    </option>

    <button type="submit"

            [disabled]="!myForm.form.valid">        <!-- enable the submit button only if all form data is valid        -->

        Submit

    </button>

</form>



Validation


                                                    /* these CSS classes are automatically added to [(ng-model)] fields  */

.ng-touched | .ng-untouched                         /* the control has been visited (has had the focus)                  */

.ng-dirty   | .ng-pristine                          /* the control has been modified                                     */

.ng-valid   | .ng-invalid                           /* the control's value is valid                                      */


.ng-valid[required], .ng-valid.required  {          /* use the classes in combination                                    */

      border-left: 5px solid green;

}

.ng-invalid:not(form)  {                            /* all invalid fields (excluding the form itself)                    */

      border-left: 5px solid red; /* red */

}
