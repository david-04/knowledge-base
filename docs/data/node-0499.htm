Email


var nodemailer = require('nodemailer');             // npm install nodemailer


var mailTransport = nodemailer.createTransport(     // open a connection to the mailserver

    require('nodemailer-smtp-transport')({

        service: 'Gmail',                           // use a predefined configuration (Gmail, Hotmail, iCloud, Yahoo!, ...)

        host: '127.0.0.1',                          // or set the endpoint manually

        port: 25,

        secureConnection: false,                    // enable TLS

        auth: {                                     // authentication (leave out if not required by the server)

            user: 'david@mydomain.com',

            pass: 'password'

        }

    })

);


mailTransport.sendMail(                             // send an email

    {

        from:     '"David" <david@mydomain.com>',   // sender

        to      :   'david@home.com,'               // multiltiple recipients can be separated by commas

                  + 'David <david@work.com>',

        subject : 'Test email',                     // email subject

        html    : '<h1>Hello world!</h1>',          // email body in HTML format (if required)

        text    : 'Hello world!',                   // email body in text format (will be sent along with the HTML version)

        generateTextFromHtml: true,                 // autogenerate the text version from the HTML code

        attachments: [                              // array of files to attach

            {

                path     : './data.txt',            // path and name of the file to be sent

                filename : 'your-invoice.txt'       // label name of the attachment within the email

            }

        ]

    },

    function (err) { }                              // called when the send has succeeded or failed (handle errors here)

);
