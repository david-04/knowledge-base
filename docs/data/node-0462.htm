Forms


document.forms                                      // collection of all the page's forms

document.forms["myForm"]                            // contains <form name="myForm">


form.action                                         // the URL to send the data to

form.method                                         // "get" or "post"

form.target                                         // like in <a href="..." target="...">

form.name                                           // the form's name

form.elements                                       // a collection all controls within the form

form.elements["myField"]                            // matches <input name="myField"> (might return a collection in case of radio buttons)

form.length                                         // the number of controls contained in the form

form.acceptCharset                                  // the character sets that the server can process

form.enctype                                        // the request encoding type


form.noValidate                                     // HTML5: set to false to disable validation


form.reset()                                        // reset to the default values (does fire the reset event)

form.submit()                                       // send the data to the server (does not fire the submit event)

form.checkValidity()                                // check if all fields meet all HTML5 constraints (e.g. "required")


[event] submit                                      // the form is about to be submitted (call event.preventDefault() to cancel it)

[event] reset                                       // the form is about to be reset (call event.preventDefault() to cancel it)


event.target                                        // the form



Fields


field.name                                          // the name attribute's value

field.value

field.type                                          // contains an <input>'s type attribute ("radio" or "checkbox") or one of the following:

                                                    // "select-one", "select-multiple", "submit", "button", "reset", "submit"

field.disabled                                     

field.readOnly

field.form                                         

field.tabIndex


field.focus()                                       // set the focus (not applicable to hidden fields)

field.blur()                                        // remove the focus


field.onblur                                        // a field loses its focus

field.onchange                                      // a field loses its focus after its value has been changed

field.onfocus                                       // a field receives the focus


event.target                                        // the field



Text fields


textField.selection{Start|End}                      // zero-based start/end of selected text (can be used for value.substring)

textField.required                                  // HTML5: field is required

textField.type                                      // "email" or "url"

textField.pattern                                   // regular expression to validate the input


textField.validity.customError                      // HTML5: true if setCustomValidity() was called

textField.validity.patternMismatch                  // input does not match the pattern

textField.validity.rangeOverflow                    // value too large

textField.validity.rangeUnderflow                   // value too small

textField.validity.stepMisMatch                     // value is in between the steps

textField.validity.tooLong                          // text is too long

textField.validity.typeMismatch                     // value is not of the correct type (email or url)

textField.validity.valid                            // boolean indicating if the field is valid

textField.validity.valueMissing                     // required field is empty


textField.select()                                  // select all text in a text box

textField.setSelectionRange(startIndex, endIndex)   // select the specified portion of text in a text box (set the focus to see the effect)

textField.checkValidity()                           // check if all HTML5 constraints (e.g. "required") are met

textField.step{Up|Down}()                           // HTML5: in-/decrement number value by 1

textField.step{Up|Down}(10)                         // HTML5: in-/decrement by 10


textField.onselect                                  // text has been selected in a text box

textField.onbeforecopy                              // not reliable: fired before a copy operation takes place

textField.oncopy                                    // fired during a copy operation

textField.onbeforecut

textField.oncut

textField.onbeforepaste

textField.onpaste

textField.oninput


event.clipboardData.getData("text")                 // copy/paste data, parameter is always "text"! (in IE: window.clipboardData)

                   .setData("text", "value")        // first paramter is always "text", the second is the text to copy

                   .clearData()



Select boxes (options)


selectBox.multiple                                  // set to true to allow multi-selection

selectBox.options                                   // collection of the contained options

selectBox.selectedIndex                             // currently selected item's index (or -1)

selectBox.size                                      // the number of visible items

selectBox.type                                      // "select-one" or "select-multiple"

selectBox.value                                     // no option selected: ""

                                                    // option with value attribute selected: value of the value attribute

                                                    // option without value attribute selected: the text displayed to the user

                                                    // multiple options selected: only the first option's value


selectBoxOption.index                               // the option's index within the select box

selectBoxOption.label                               // the HTML label attribute's value

selectBoxOption.selected                            // boolean: is the option currently selected?

selectBoxOption.text                                // the text displayed to the user

selectBoxOption.value                               // the HTML value attribute's value


selectBox.add(newOption, relOption)                 // add newOption (document.createElement("option")) before relOption (set to null to append at the end)

selectBox.remove(index)                             // remove the option at the given index



Buttons


buttonField.formNoValidate                          // set to true for a submit button to disable form validation



HTML5 custom validators


<{input|textarea} oninput="myValidator(this)" />    <!-- attach an event handler to be called whenever the value changes                                -->


function myValidator(input) {

    if (input == "...") {

        input.setCustomValidity("That's  wrong!");  // mark the element as invalid and set an error message (displayed by the browser)

    } else {

        input.setCustomValidity("");                // clear any error message if the input is (now) correct

    }

}
