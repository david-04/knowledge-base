JavaScript


<script type="text/javascript">                    <!-- inline JavaScript code, the type is optional in HTML5                         -->

    //<![CDATA[                                    <!-- prevent the HTML parser from accidentally picking up on JavaScript code       -->

        alert("Hello world!");

    // ]]>

</script>


<script src     = "my-script.js"                   <!-- load JavaScript from an external source                                       -->

        charset = "utf-8"                          <!-- character encoding of the external file                                       -->

        type    = "text/javascript"                <!-- the type is optional in HTML5                                                 -->

        async                                      <!-- run in parallel while loading/displaying the page                             -->

        defer   >                                  <!-- run after loading/displaying the page                                         -->

</script>                                          <!-- the <script> tag must have a closing tag                                      -->


<noscript>                                         <!-- show this block only if JavaScript is not available or disabled               -->



Plugins


<object data   = "my-plugin.swf"                   <!-- preferred method: embed any object (plugin, audio, HTML, PDF, applets, ...)   -->

        type   = "application/vnd.adobe.flash-movie"

        width  = "400"                             <!-- width and height of the container (in pixels)                                 -->

        height = "400"

        form   = "myForm"                          <!-- make this element part of a form                                              -->

        name   = "myPlugin"

        usemap = "#myImageMap" >                   <!-- use an image map when the user clicks on the object                           -->

    Sorry...                                       <!-- fallback content to show if the plugin is not supported or can't be loaded    -->

    <param name="..." value="..." />               <!-- define parameters that should be passed on to the plugin                      -->

</object>


<embed src    = "my-plugin.swf"                    <!-- old method: embed a plugin                                                    -->

       type   = "application/vnd.adobe.flash-movie"<!-- the plugin's MIME type                                                        -->

       width  = "200"                              <!-- width and height of the container (in pixels)                                 -->

       height = "100">

    Sorry...                                       <!-- fallback content to show if the plugin is not supported or can't be loaded    -->

</embed>
