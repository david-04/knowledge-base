Friendly URLs


PM>Install-Package Microsoft.AspNet.FriendlyUrls.Core


public class RouteConfig                                        // implement a new class in App_Code

{

    public static void RegisterRoutes(RouteCollection routes)

    {

        var settings = new FriendlyUrlSettings();

        settings.AutoRedirectMode = RedirectMode.Permanent;

        routes.EnableFriendlyUrls(settings);

    }

}


public class Global : System.Web.HttpApplication                // Create a "Global Application Class" Global.asax

{


    protected void Application_Start(object sender, EventArgs e)

    {

        RouteConfig.RegisterRoutes(System.Web.Routing.RouteTable.Routes);

    }


Request for .../page.aspx are now redirected to .../page (the extension is not shown in the browser url)



Asynchronous processing


<%@ Page ... Async        = "true"                              <!-- enable asynchronous processing on page level -->

             AsyncTimeout = "1"     %>                          <!-- set the timeout to 1 second -->


protected void Page_Load(object sender, EventArgs e)            // not recommended as it blocks the page life cycle

{

    RegisterAsyncTask(new PageAsyncTask(GetCustomersAsync));    // register tasks for asynchronous execution

}


private async Task GetCustomersAsync()                          // the PreRender event will wait until this task is completed

{

    int result = await Task.Run<int>(() => { return 0; });      // execute the asynchronous call via await

    this.TexBox = "...";                                        // access the controls like in Page_Load (when running synchronously)

}
