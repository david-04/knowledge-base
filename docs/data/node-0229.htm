Razor


@using MyNamespace.MyModels                         <!-- import a namespace (not a single class!)                                 -->


@model MyNamespace.MyModels.Product                 <!-- an object of this type is passed to the view                             -->

@model MyNamespace.MyModels.Product[]               <!-- the view can also recieve a whole array of those objects                 -->

@model int                                          <!-- simple value types can be passed as well (int? for nullable)             -->


@{                                                  <!-- a code block can contain any C# statements                               -->

    Layout = null;                                  <!-- this page does not require a layout (template / master page)             -->

}


<!DOCTYPE html>                                     <!-- regular HTML code is used                                                -->

<html>

<head>

    <title>@Model.MyProperty</title>                <!-- insert a model property ("Model" is passed into a strongly typed views)  -->

</head>

<body>

    It's @ViewBag.MyProperty ...                    <!-- insert a dynamic property (available in untyped views ass well)          -->


    <input type="checkbox"

           checked="@ViewBag.checked" />            <!-- will either omit the "checked" attribute or generate checked="checked"   -->

</body>

</html>



Iterations


@foreach (Order o in Model)                         <!-- iterate over arrays                                                      -->

{

    <li>@o.date</li>                                <!-- does not require "@:" because <b> is recognised as HTML code literal     -->

}



If-conditions


@if (0 == ViewBag.stock)                            <!-- condiditional IF processing                                              -->

{

    @:Out of Stock                                  <!-- literal text must be prefixed with "@:" within code blocks"              -->

}

else if (1 == ViewBag.stock)

{

    <b>Last one</b>                                 <!-- does not require "@:" because <b> is recognised as HTML code literal     -->

}

else

{

    @ViewBag.stock

}



Switch-case


@switch ((int)ViewBag.stock)                        <!-- switch-case                                                              -->

{

    case 0:

        @: Out of Stock                             <!-- literal text must be prefixed with "@:" within code blocks"              -->

        break;

    case 1:

        <b>Last one</b>                             <!-- does not require "@:" because <b> is recognised as HTML code literal     -->

        break;

    default:

        @ViewBag.stock

}
