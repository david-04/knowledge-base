Simple enums


enum Season {                                       // simple enumeration

    Spring,

    Summer,

    Autumn,

    Winter

}


Season season     = Season.Spring                   // reference an enumeration value directly

Season season     = Season.valueOf("Spring")        // get an enumeration value by its name

int    position   = Season.Spring.ordinal()         // get the index of the enumeration item amongst all items

Season seasons[]  = Season.values()                 // get an array of all available enumeration values



if (season == Season.Spring) {...}                  // compare enumeration values

if (season.compareTo(Season.Summer)) {...}          // compares the enumeration items' index and resturns -1/0/1



Complex enums


enum City implements MyInterface {                  // enums can implement interfaces (but not inherit from classes)


    Melbourne("Melbourne", "Australia"),            // the enum values are written as constructor calls

    Berlin   ("Berlin",    "Germany"  );


    private String city;                            // enums can have properties, methods and constructors

    private String country;


    private City(String city, String country) {

        this.city = city;

        this.country = country;

    }


    public String toString() {

        return this.city + " " + this.country;

    }

}
