Event handlers


.on ("click",             eventHandler)             // register an event handler

.one("click",             eventHandler)             // register an event handler that's executed only once

.off("click",             eventHandler)             // remove a specific event handler (or all) from the chain


.one("click", "selector", eventHandler)             // add an event handler to all (current and future) childen matching the selector


.one("click.myNamespace", eventHandler)             // register and remove event handlers within a dedicated namespace

.off("click.myNamespace", eventHandler)          


.one("myEvent",           eventHandler)             // register and remove event handlers for custom events

.off("myEvent",           eventHandler)


.trigger("click")                                   // fire standard events

.trigger("myEvent", {parameter:"value"})            // fire a custom event (the optional object is passed as a second parameter to the event handler function)

Event properties


event.originalEvent                                 // the original event fired by the system (jQuery event handlers received wrapped jQuery events)

event.data                                          // optional object of data passed to an event method

event.type

event.timeStamp

event.namespace

event.result                                        // the last value returned by an event handler that was triggered by this event, unless the value was undefined


event.target                                        // the DOM element that initiated the event

event.currentTarget                                 // the current DOM element within the event bubbling phase

event.delegateTarget                                // the element that has the event handler attached

event.relatedTarget                                 // the other DOM element involved in the event, if any


event.isDefaultPrevented()                          // propagation

event.isImmediatePropagationStopped()

event.isPropagationStopped()

event.preventDefault()

event.stopImmediatePropagation()

event.stopPropagation()


event.{alt|ctrl|shift|meta}Key                      // indicates whether any of the modifiers was (or is beeing) pressed (boolean)

event.pageX                                         // mouse position relative to the document

event.pageY

event.which                                         // for key or mouse events, this property indicates the specific key or button that was pressed
