Attributes and styles


.attr("href")                                       // retrieve an attribute's value

.attr("href", "google.com.au")                      // set an attribute's value

.attr({href:'google.com.au', target:'_top'})        // set multiple attributes at once

.removeAttr("href target")                          // remove attributes (takes multiple attributes since 1.7)


.css("color")                                       // retrieve a single CSS style attribute

.css(["color", "width", "border-width"]);           // retrieve a hashtable with all specified CSS style attributes

.css("color", "red")                                // set a single CSS style attribute

.css({"color":"red", "font-weight":"bold"})         // set multiple CSS style attributes at once


.hasClass("class1 class2")                          // check if an element contains all of the specified CSS classes

.addClass("class1 class2")                          // add CSS classes

.removeClass("class1 class2")                       // remove CSS classes

.toggleClass("class1 class2")                       // toggle CSS classes


.hide()                                             // make the selected elements invisible (display:none)

.show()                                             // make the selected elements visible (display:block)

.toggle()                                           // toggle the display between none and block



Data and content


.html()                                             // get the innerHTML (including HTML tags)

.text()                                             // get the innerTEXT (HTML tags are removed automatically)

.val()                                              // retrieve the value of a form's input field

.data("key")                                        // get a custom attribute (e.g. data-key="value") or a value set via jQuery's .data() function


.html("<b>Hello world</b>")                         // set the innerHTML (may contain HTML code)

.text("Hello world")                                // set the innerHTML (special HTML characters are masked automatically)

.val("Hello world")                                 // set a form's input element's value

.data("key",{"value"|array|object})                 // add custom data to an object (stored internally, not added as a custom attribute)


.empty()                                            // remove all text and child nodes (same as innerHTML="")



Nodes and hierarchies


$("<a/>")                                           // create a new DOM element


.clone(mustCloneEventHandlers)                      // clone the selection so that it can be inserted into the DOM a second time


.append      ({"htmlcode"|selection|element})       // add as the last child (accepts multiple parameters at once)

.prepend     ({"htmlcode"|selection|element})       // add as the first child


.before      ({"htmlcode"|selection|element})       // insert as the previous sibling (brother node)

.after       ({"htmlcode"|selection|element})       // insert as the next sibling (brother node)


.insertBefore({"htmlcode"|selection|element})       // insert the whole selection before every element matching the selector

.insertAfter ({"htmlcode"|selection|element})       // insert the whole selection after every element matching the selector


.replaceWith ({"htmlcode"|selection|element})       // replace the selected elements with the given elements

.replaceAll  ({"htmlcode"|selection|element})       // replace the given elements with the current selection


.wrap        ({"htmlcode"|selection|element})       // surround each selection's root element by a new <div> node

.wrapAll     ({"htmlcode"|selection|element})       // surround the first element by a <div> and move the remaining selection in there as well

.wrapInner   ({"htmlcode"|selection|element})       // surround all children of each selected element by a <div> container

.unwrap()                                           // remove the parent and move all of its childen one step up the DOM hierarchy


.remove()                                           // delete the element from its parent container (like setting outerHTML="")



Scrolling


.scrollTop()                                        // get the horizontal and vertical scrollbar offset (0/0 for the top-left position)

.scrollLeft()                 


.scrollTop(offset)                                  // set the scrollbar offset

.scrollLeft(offset)
