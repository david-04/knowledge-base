Attributes


<xsl:template          match  = "myElement"     >   <!-- attributes are ignored (and not processed) by default           -->

  <xsl:value-of        select = "@myAttribute" />   <!-- retrieve the value of a particular attribute                    -->

  <xsl:apply-templates select = "@myAttribute" />   <!-- process (recurse into) a particular attribute                   -->

  <xsl:apply-templates select = "@*"           />   <!-- process (recurse into) every attribute                          -->

</xsl:template>                                     <!-- if the xs:apply-templates finds no matching rule, then it       -->

                                                    <!-- will copy the attribute's value as the default action           -->





<xsl:template match="@myAttribute">                 <!-- attributes can be targetted by a template rule                  -->

  <xsl:value-of select="."/>                        <!-- retrieve the value of the matched attribute                     -->

</xsl:template>

</xsl:template>                                     <!-- insert its content into the output attribute                    -->

    use-attribute-sets     = "myOtherSet">          <!-- whitespace-separated list of sets to include (derive from)      -->

  </xsl:attribute>

                                                    <!-- copy including attributes and children                          -->





<xsl:copy                                           <!-- copy the current node excluding attributes and children         -->

          use-attribute-sets="myAttributeSet">      <!-- optional: add the given attribute sets                          -->

  ...                                               <!-- this content goes inside the copied node                        -->

</xsl:copy>





<xsl:element  name      = "myElement"               <!-- create a new <myElement> XML node                               -->

              namespace = "my-domain.com"           <!-- bind the element to this namespace                              -->

              use-attribute-sets="myAttributeSet">  <!-- add the given attribute sets (whitespace-separated list)        -->


  <xsl:attribute name="myAdditionalAttribute">      <!-- add an attribute to the surrounding XML element                 -->

    myAttributeValue                                <!-- the attribute value (cuation: whitespace is preserved)          -->

  </xsl:attribute>


  ...                                               <!-- this content goes inside the new <myElement> node               -->


</xsl:element>
