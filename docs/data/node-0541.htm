Exception handling


class MyException < RuntimeError            # custom exceptions are derived from Exception (or a child class)

end                                         # no need to implement a constructor


begin                                       # exception handling requires a begin-end block

  raise MyException, "Ooops"                # raise an exception

rescue MyException => exception             # catch exception

  puts "MyException: #{exception}"

  retry                                     # run the code again

rescue => exception

  puts "Another exception #{exception}"

end


1 / 0 rescue nil                            # inline "ignore" error handler
