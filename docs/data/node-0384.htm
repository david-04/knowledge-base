Document structure


<!DOCTYPE html>                                     <!-- simplified HTML5 structure: <html>, <head> and <body> are optional           -->

<title>My document</title>                

<p>Content goes here ...</p>



<!DOCTYPE html>                                     <!-- recommended HTML5 structure                                                  -->

<html lang="en">                                    <!-- specify the natural language (can be applied to any HTML element)            -->

<head manifest="offline.appcache"                   <!-- the manifest is optional and only required for offline applications          -->

      xmlns="http://www.w3.org/1999/xhtml">         <!-- required for XHTML but not allowed for regular HTML                          -->

    <meta charset="utf-8">                          <!-- the character enconding must be specified as soon as possible                -->

    <title>...</title>                              <!-- each page must have a title                                                  -->

</head>

<body>

    ...

</body>

</html>



Header declarations


<base href       = "http://mydomain.com/my-app/"    <!-- default base url for all relative links/references on this page              -->

      target     = "{_blank|_parent|_self|_top}" /> <!-- or pass the name of a frame                                                  -->


<meta name       = "viewport"                       <!-- mobile browsers should NOT zoom out to show the whole page at once           -->

      content    = "width=device-width,

                    initial-scale=1,

                    shrink-to-fit=no" />               


<meta http-equiv = "X-UA-Compatible"                <!-- request IE to use the latest rendering engine (no compatibility mode)        -->

      content    = "IE=edge">


<meta name       = "{application-name|author|       <!-- set metadata for search engines                                              -->

                     description|generator|

                     keywords}"

      content    = "..." />


<meta http-equiv = "content-type"                   <!-- explicitely specify the MIME type and encoding                               -->

      content    = "text/html; charset=UTF-8" />


<meta http-equiv = "default-style"                  <!-- select one of the <link> style sheets as the default one                     -->

      content    = "stylesheet-link-title" />       <!-- this must match the "title" attribute of the <link> element                  -->


<meta http-equiv = "refresh"                        <!-- refresh (reload) the page after 60 seconds                                   -->

      content    = "60" />

External links


<link href        = "my-style.css"                  <!-- load an external file (style sheet, favicon, ...)                            -->

      hreflang    = "en"                            <!-- natural language of the linked resource                                      -->

      type        = "text/html"                     <!-- the MIME type of the linked resource                                         -->

      media       = "print and (resolution:300dpi)" <!-- media-query indicating which media the link is optimised for                 -->

      type        = "text/css"                      <!-- the MIME type                                                                -->

      sizes       = "{32x32 16x16|any}"             <!-- image sizes (for .ico files) or "any" for scalable vector graphics           -->

      crossorigin = "{anonymous|use-credentials}"   <!-- how to handle cross-origin requests                                          -->

      rel         = "alternate"                     <!-- alternate presentation of the document (e.g. a translation)                  -->

      rel         = "author"                        <!-- link to information about the document's author                              -->

      rel         = "dns-prefetch"                  <!-- request a preempzive DNS lookup                                              -->

      rel         = "help"                          <!-- link to a help document                                                      -->

      rel         = "icon"                          <!-- the page's favicon                                                           -->

      rel         = "license"                       <!-- link to copyright information                                                -->

      rel         = "next"                          <!-- next document in the series (i.e. next page)                                 -->

      rel         = "pingback"                      <!-- address of the pingback server handling pingbacks to the current document    -->

      rel         = "preconnect"                    <!-- request to preemptively establish a connection                               -->

      rel         = "prefetch"                      <!-- the browser should fetch and cache the document in advance                   -->

      rel         = "preload"                       <!-- the browser must fetch and cache the document in advance                     -->

      rel         = "prev"                          <!-- the previous document in the series (i.e. the previous page)                 -->

      rel         = "search"                        <!-- link to a resource used to search through the current document               -->

      rel         = "stylesheet" />                 <!-- the <link> references a CSS style sheet                                      -->



CSS


<link href  = "my-styles.css"                       <!-- include an external style sheet                                              -->

      rel   = "stylesheet"                          <!-- type of the link must be "stylesheet"                                        -->

      type  = "text/css"                            <!-- optional: the MIME type (can safely be emitted for rel="stylesheet")         -->

      media = "screen and (max-width: 568px)" />    <!-- full-blown media queries are not supported in IE<=8 (use fallback below)     -->


<!--[if lt IE 9]>                                   <!-- fallback style sheet for IE<=8 that ignores <links> with media queries       -->

<link rel="stylesheet" href="my-ie-styles.css">

<![endif]-->


<style type  = "text/css"                           <!-- inline style definitions (the MIME type is optional and can be omitted)      -->

       media = "screen and (max-width: 568px)" >    <!-- optional: the media type or a media query                                    -->

    p { color : red; }

</style>
