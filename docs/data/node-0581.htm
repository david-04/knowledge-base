Nesting


#myElement {


    line-height: 1.5em;                             /* global properties are applied to all nested rules                    */


    em, b {                                         /* compiles to "#myElement em, #myElement b"                            */

        color: red;

    }


    &:hover {                                       /* compiles to "#myElement:hover" ("&" represents the parent selector   */

        font-weight: bold;                          /* and can be used freely, e.g. ".myClass &" or "&-highlighted")        */

    }


    @media screen and (orientation: landscape) {    /* the media query covers (wraps) all parent selectors                  */

        width: 500px;

    }

}



Nested global rules


@media print {

    #myElement {


        width: 8px;                                 /* local properties are not added to @at-root rules                     */


        @at-root .myClass1 {                        /* pretend that .myClass1 is defined outside of all selectors (i.e. not */

            color:red;                              /* nested into #myElement), but still inside the media query)           */

        }


        @at-root {                                  /* same as above (but containing multiple rules)                        */

            .myClass2 {

                font-size:2em;

            }

            .myClass3 {

                font-size:5em;

            }

        }


        @at-root (without: media) {                 /* specify which boundaries to cross or not to cross:                   */

            .myClass4 {                             /* without media ... keep selectors but strip media queries             */

                color: blue;                        /* without rule .... keep media queries but strip selectors             */

            }                                       /* without all ..... strip both selectors and media queries             */

        }                                           /* can also use "with ..." instead of "without ..."                     */

    }

}
