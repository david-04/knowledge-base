Unit testing


@RunWith(SpringRunner.class)                        // JUnit will invoke this class to run the test (instead of running it itself)

@SpringBootTest                                     // request the whole appliction context (use @WebMvcTest for web layers only)

@AutoConfigureMockMvc                               // enable auto-configuration (applicable to test classes only)

public class HelloControllerTest {


    @Autowired                                      // request dependency injection

    private MockMvc mvc;


    @Test                                           // mark it as a unit test method

    public void getHello() throws Exception {

        mvc.perform(

            MockMvcRequestBuilders.get("/")

                .accept(MediaType.APPLICATION_JSON))

                .andExpect(status().isOk())

                .andExpect(content().string(equalTo("Greetings from Spring Boot!")));

    }

}
