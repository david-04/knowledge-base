.menu {                                             /* style the menu list                                                     */

    list-style: none;

    font-weight: bold;

    margin-bottom: 10px;

    float: left;                                    /* clear floats                                                            */

    width: 100%;

    padding-left:0;

    /* position: relative; */                       /* bring the .menu above everything else - uncomment if needed             */

    /* z-index: 5; */

}

.menu li {

    float: left;

    margin-right: 10px;

    position: relative;

}

.menu a {

    display: block;

    padding: 5px;

    color: #fff;

    background-color: #333;

    text-decoration: none;

}

.menu a:hover {

    color: #fff;

    background-color: #6b0c36;

    text-decoration: underline;

}


.menu ul {

    background-color: #fff;                         /* backward compatibility: must be the same as the page's background       */

    background: rgba(255,255,255,0);                /* modern browsers: make the background fully transparent                  */

    list-style: none;

    position: absolute;

    left: -9999px;                                  /* hide off-screen while not hovering                                      */

}

.menu ul li {

    padding-top: 1px;

    float: none;

}

.menu ul a {

    white-space: nowrap;                            /* stop text wrapping and creating multi-line dropdown items               */

}

.menu li:hover ul {                                 /* display the dropdown on hover                                           */

    left: 0;                                        /* bring back on-screen when needed                                        */

    padding-left:0;

}

.menu li:hover a {                                  /* keep the submenu visible, even when the cursor has moved down the list  */

    background-color: #6b0c36;

    text-decoration: underline;

}

.menu li:hover ul a {

    text-decoration: none;

}

.menu li:hover ul li a:hover {                      /* what happens when you hover each individual link                         */

    background-color: #333;

}
