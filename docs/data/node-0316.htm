Exception handling


try

{

    Exception exception = new MyException("Error message");

    exception.HelpLink = "URL with additional information";     // link to further help

    exception.Data.Add("Argument count", args.Length);          // custom key-value-pairs

    throw exception;

}

catch (MyException exception)                                   // catch a specific type

{

    Console.WriteLine(exception);

    foreach (DictionaryEntry data in exception.Data)

        Console.WriteLine("{0} = {1}", data.Key, data.Value);

}

catch (Exception exception)

when (exception.Message.Contains("DEBUG"))                      // catch only if criteria is met

{

    throw;                                                      // using "throw exception" instead disturbes the stack trace!

}

catch                                                           // catch any exception

{

}

finally                                                         // processed after all catch blocks have been processed

{

}



Defining exceptions


Generate the template by typing "exception" and pressing the TAB key twice:


[Serializable]

public class MyException : Exception

{

    public MyException() { }

    public MyException(string message) : base(message) { }

    public MyException(string message, Exception inner) : base(message, inner) { }

    protected MyException(

        System.Runtime.Serialization.SerializationInfo info,

        System.Runtime.Serialization.StreamingContext context) : base(info, context)

    { }

}
