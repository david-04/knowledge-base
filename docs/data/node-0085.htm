Iteration


<li *ngFor="let person of people">                  <!-- iterate over the elements of a collection                      -->

    {{person.fullName}}

</li>


<div *ngFor="let person of people; let i=index">    <!-- make the iteration index available as variable "i"             -->

    {{i + 1}} - {{person.name}}

</div>


<div *ngFor="let person of people; trackBy:getId">  <!-- performance tuning: don't re-create (but update) the DOM if    -->

    {{person.name}}                                 <!-- the getId function returns returns the same (unchanged) value  -->

</div>


getId(index: number, person: Person) {              // this function returns the key to match identical objects

    return person.id;

}
