Cascading includes


<!ENTITY % externalDTD SYSTEM "file.dtd">           <!-- load content from external file                    -->

%externalDTD;                                       <!-- inline the external file's content                 -->



Override named entities


<!ENTITY % children "(child1|child2)+">             <!-- DTD defines default for named entity               -->

<!ELEMENT element %children;>           


<!DOCTYPE rootElementName [                         <!-- XML can override the DTD's named entity            -->

  <!ELEMENT % children "child1*">             

]>



Selective in- and excludes


<!ENTITY % elementRequired "INCLUDE">

<![%elementRequired;[                               <!-- elementRequired can be INCLUDE or EXCLUDE         -->

  <!ELEMENT element (#PCDATA)>

]]>


<!DOCTYPE rootElementName [                         <!-- XML can override the DTD's preset                 -->

  <!ENTITY % elementRequired "EXCLUDE">           

]>
