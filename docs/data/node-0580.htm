Mixins


@mixin myAddonMixin {                               /* mixins can target other rules/selectors                              */

    color: #ff0000;                                 /* they can add additional properties to their parent rule              */

    #myContainer &:hover {                          /* ... and/or define additional rules (possibly based on the parent)    */

        color: red;

    }

    @include myOtherMixin;                          /* mixins may also include other mixins                                 */

}


.myClass {

    @include myAddonMixin;                          /* a rule can include/apply a mixin                                     */

    @include myOtherMixin;                          /* ... or possibly multiple at the same time                            */

    font-size: 2em;                                 /* ... while also specifying its own properties                         */

}





@mixin myGlobalMixin {

    a {                                             /* global mixins may only contain rules (but no root-level properties)  */

        color: red;                                 /* the use of "&" as parent reference is prohibited as well             */

    }

}


@include myGlobalMixin;                             /* global mixins are imported on root-level (i.e. not within a rule)    */



Parameters


@mixin myMixin($color, $width: 50%) {               /* mixins can have parameters with (optionally) default values          */

    background-color: $color;

    max-width:        $width;

}


@include myMixin(blue, 50%);                        /* parameters can be passed in the order defined by the mxin            */

@include myMixin($width: 50%, $color: red);         /* ... or in any order if the parameters are named                      */

@include myMixin(red, $width:50%);                  /* ... or in a mixed form with only some parameters being named         */



Variable arguments


@mixin myMixin($parameters...) {                    /* mixins also support variable argument lists                          */

    box-shadow: $parameters;                        /* they can be used to just insert all values space-separatedly         */

    @include myOtherMixin($parameters...);          /* ... or they can be passed on to other mixins                         */

    $myKeyValueMap: keywords($parameters);          /* retrieve a key-value-map of all named parameters                     */

}


$myList: #ff0000, #00ff00, #0000ff;                 /* define a list to pass in                                             */

.primary {

    @include myMixin($myList...);                   /* expand the list when calling the mixin                               */

    @include myMixin(red green blue);               /* ... or pass the list as a literal                                    */

    @include myMixin(red, green, blue);             /* ... or format them like single paramters                             */

}



Content


@mixin myMixin {

    .myClass {

        @content;                                   /* use @content to insert body data passed by the caller                */

    }

}


@include myMixin {                                  /* add a {...} body when including the mixin                            */

  color: red;

}
