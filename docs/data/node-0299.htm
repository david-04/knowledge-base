Chaining constructors


public Person(string name)

{

    this.Name = name;

}


public Person(string name, string city)

    : this(name)                          // call another constructor first

{

    this.City = city;                     // process only the remaining parameters

}



Initialising the base class


internal class Employee : Person

{

    public int Id;


    public Employee(int id, string name)

        : base(name)                      // call the base class' constructor first

    {

        this.Id = id;                     // process the parameters specific to the derived class

    }

}



Instantiation


internal class Person

{

    public string Name;

    public string City;

    public Person() { }

    public Person(string name)

    {

        this.Name = name;

    }

}


Person person = new Person("David")       // call one of the constructors first

{

    City = "Melbourne"                    // assign additional properties thereafter

};


Person person = new Person                // call the default constructors

{

    Name = "David",                       // set all properties manually

    City = "Melbourne"

};
