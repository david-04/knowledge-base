Defining sequences


CREATE SEQUENCE my_sequence AS INT                  -- create a sequence

       START WITH 1 INCREMENT BY 1                  -- start value and increment

       MINVALUE 1 MAXVALUE 100                      -- range of valid values

       CYCLE;                                       -- start over if the upper boundary has been hit (default: NO CYCLE)


ALTER SEQUENCE my_sequence                          -- reconfigure an existing sequence

      RESTART WITH 50 INCREMENT BY 2

      MINVALUE 1 MAXVALUE 2

      NO CYCLE;



Using sequences


SELECT NEXT VALUE FOR my_sequence;                  -- generate and retrieven the next sequence number


SELECT current_value FROM sys.sequences             -- read the current sequence number without incrementing it

                     WHERE name='my_sequence'
