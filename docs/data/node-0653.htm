Modules


Module MyModule                  ' modules are like a static (shared) class


    Friend field As Integer      ' all fields are static (shared) by default


    Sub New()                    ' the constructor can't be public and is executed

        ' ...                    ' when the module is invoked for the first time

    End Sub


    Sub MySubroutine()

        ' ...

    End Sub


    Public Class SomeClass       ' modules can contain classes as well

        ' ...

    End Class


End Module


Module MyProgram


    Sub assign()

        field = 0               ' field references and subroutine/function calls

        MySubroutine()          ' don't need to include the the source module name

    End Sub


End Module
