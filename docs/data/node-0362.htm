Multiple columns


column-count:      3;                               /* let the conten flow across 3 columns (like in a newspaper)       */


column-width:      {100px|auto};                    /* set the column width (or let the browser deside)                 */


column-gap:        10px;                            /* spacing between the columns                                      */


column-rule:       1px solid black;                 /* shorttuc: column-rule-width column-rule-style column-rule-color  */

column-rule-width: 1px;                             /* width of the vertical lines                                      */

column-rule-color: red;                             /* color of the vertical lines                                      */

column-rule-style: {none|hidden|dotted|dashed       /* show (or hide) vertical lines between the columns                */

                   |solid|double|groove|ridge

                   |inset|outset};

                                                   


column-span:       {1|2|...|all};                   /* span this element across the given number of columns             */
