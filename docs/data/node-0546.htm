Require


require 'net/http'                          # require from GEM_PATH

require './lib.rb'                          # local files require a relative path

require_relative 'lib.rb'                   # look for local files only


load 'lib.rb'                               # like require, but file can be loaded/executed multiple times


%w{file1 file2}.each { |f| require f }      # can be invoked dynamically


$:                                          # contains the ruby-internal folders from where "require" loads

$:.push    '/home/david/gems'               # add more directories at the end

$:.unshift '/home/david/gems'               # add more directories at the beginning
