Embedded Ruby (ERB)


require 'erb'                               # standard library


array = %w{ Melbourne Victoria }


template = %q{

  <% array.each do |item| %>                # <% ...code... %> to embed Ruby commands

    <p><%= item %></p>                      # <%= ...exp... %> to embed the expression's value

  <% end %>

}


result = ERB.new(template).result(binding)  # render the template and return the result as a string

ERB.new(template).run(binding)              # render the template and directly print it to the standard output
