Exception handling


Try


    Dim exception As Exception = New MyException("Error message")

    exception.HelpLink = "URL with additional information"          ' link To further help

    exception.Data.Add("Line number", 10)                           ' custom key-value-pairs

    Throw exception


Catch exception As MyException                                      ' catch a specific type


    Console.WriteLine(exception)

    For Each Data As DictionaryEntry In exception.Data

        Console.WriteLine("{0} = {1}", Data.Key, Data.Value)

    Next


Catch exception As Exception _

When exception.Message.Contains("DEBUG")                            ' catch only if criteria is met

    ' ...

    If condition Then Exit Try                                      ' exit the error handler

    Throw                                                           ' using "Throw exception" instead disturbes the stack trace

    ' ...

Catch                                                               ' catch any exception

    ' ...

Finally                                                             ' processed after all catch blocks have been processed

    ' ...

End Try



Defining exceptions


<Serializable>

Class MyException

    Inherits Exception


    Public Sub New()

    End Sub


    Public Sub New(message As String)

        MyBase.New(message)

    End Sub


    Public Sub New(message As String, inner As Exception)

        MyBase.New(message, inner)

    End Sub


    Public Sub New(info As System.Runtime.Serialization.SerializationInfo,

                    context As System.Runtime.Serialization.StreamingContext)

        MyBase.New(info, context)

    End Sub

End Class
