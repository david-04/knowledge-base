Current hierarchy path


<asp:SiteMapPath ...    PathSeparator         = " >> "                          <!-- text separator between path elements         -->

                        ParentLevelsDisplayed = "3" />                          <!-- show only the 3 innermost elements           -->

    <CurrentNodeStyle   CssClass              = "CurrentPathStyle" />           <!-- the last node in the path (the current page) -->

    <NodeStyle          CssClass              = "NodeStyle" />                  <!-- parent nodes in the path                     -->

    <PathSeparatorStyle CssClass              = "SeparatorStyle" />             <!-- style of the separator between path elements -->


    <PathSeparatorTemplate>HTML</PathSeparatorTemplate>                         <!-- use custom HTML code as path separator       -->

</asp:SiteMapPath>
