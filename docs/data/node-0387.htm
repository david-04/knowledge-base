Lists


<ol type="{1|A|a|I|i}"                              <!-- type of enumeration (numeric, alphabetic, roman-numeric)                           -->

    start="5"                                       <!-- the start index for the first list item                                            -->

    reversed>                                       <!-- enumerate backwards (does not change the order of the list items themselves)       -->


<ul>                                                <!-- unordered lists to not support any attributes                                      -->


<li value="7">...</li>                              <!-- for <ol> only: set the number of the current list item                             -->



Description lists


<dl>                                                <!-- description list                                                                   -->

    <dt>Coffee</dt>                                 <!-- description term                                                                   -->

    <dd>Black hot drink</dd>                        <!-- description definition                                                             -->

   

    <dt>Milk</dt>

    <dd>White cold drink</dd>

</dl>



Tables


<table sortable>                                    <!-- no attributes other than sortable allowed in HTML5                                 -->

   

    <caption>...</caption>                          <!-- the table's caption                                                                -->


    <colgroup style="background:blue;"  span="2" >  <!-- style columns (<colgroup> must come before <thead>, <tfoot> and <tbody>)           -->

        <col  style="background:red;"           />  <!-- use <col> to add more styles to single columns within the <colgroup>               -->

        <col  style="background:green;" span="2"/>  <!-- use "span" to style multiple columns at once                                       -->

    </colgroup>


    <thead>                                         <!-- the <thead> is shown above all data rows                                           -->

        <tr>                                        <!-- may contain multiple table rows                                                    -->

            <th>Month</th>                          <!-- <th> usually has another formatting than <td> (e.g. centred + bold)                -->

            <th>Savings</th>

        </tr>

    </thead>


    <tfoot>                                         <!-- the <tfoot> is shown below all data rows                                           -->

        <tr>                                        <!-- may contain multiple table rows                                                    -->

            <td>Sum</td>

            <td>$180</td>

        </tr>

    </tfoot>


    <tbody>                                         <!-- the <tbody> must coma after <colgroup>, <thead> and <tbody>                        -->

        <tr>                                        <!-- <tr> does not support any attributes in HTML5                                      -->

            <td>January</td>

            <td>$100</td>

        </tr>

        <tr>

            <td>February</td>

            <td>$80</td>

        </tr>

    </tbody>


</table>


<th abbr    = "Sum"                                 <!-- abbreviated version of the header cell's content                                   -->

    colspan = "2"                                   <!-- span the cell across multiple rows or columns                                      -->

    rowspan = "3"

    scope   = "{col|colgroup|row|rowgroup}"         <!-- specify if it is the header of a column, colgroup, row or rowgroup                 -->

    sorted  = "{reversed|number|reversed number}"   <!-- define the sort direction of the column                                            -->

    headers = "name,author" >                       <!-- id value of one or more <th> elements that this cell belongs to                    -->


<td colspan = "2"                                   <!-- span the cell across multiple rows or columns                                      -->

    rowspan = "3"

    headers = "name,author" >                       <!-- id value of one or more <th> elements that this cell belongs to                    -->
