Buffers


buffer = new Buffer(1024)                           // create an empty buffer with a capacity of 1024 bytes (contains random data)

buffer = new Buffer('data')                         // create a buffer from a string value (assumed to be UTF-8 encoded)

buffer = new Buffer('data', '{utf8|ascii|base64}')  // create a buffer from a string value with the given encoding


buffer[0]                                           // access single bytes in the buffer

buffer.length                                       // the buffer's capacity


buffer.toString()                                   // convert the buffer to an UTF-8 encoded string

buffer.toString('{ascii|utf8|base64}')              // convert the buffer to a string using the specified encoding


buffer.slice(start, end)                            // does not clone: create a new Buffer object that points to the same memory

                                                    // location but only references buffer[start] (inclusive) ... buffer[end] (exclusive)


sourceBuffer.copy(                                  // copy binary data from one buffer to another

    targetBuffer,                                   // the target buffer to copy to

    targetStartIndex,                               // start position in the target buffer where the past starts

    sourceStartIndex,                               // start position in the source buffer where the copy starts

    sourceEndIndex                                  // the position of the first byte NOT to be copied (endIndex+1)

);
