Serving custom error pages


<configuration>

  <system.web>

    <customErrors mode            = "On"                             <!-- RemoteOnly to show exceptions to clients on the same machine only -->

                  defaultRedirect = "~/Errors/Default.aspx"

                  redirectMode    = "ResponseRewrite">               <!-- or ResponseRedirect                                               -->

      <error statusCode="404" redirect="~/Errors/Error404.aspx" />   <!-- can't access the user profile in these pages                      -->

      <error statusCode="500" redirect="~/Errors/Error500.aspx" />   <!-- all other errors go to defaultRedirect                            -->

    </customErrors>



public partial class Error404 : System.Web.UI.Page

{

    protected void Page_Load(object sender, EventArgs e)

    {

        Response.StatusCode             = 404;                       // set the HTTP error code

        Response.Status                 = "404 Not Found";

        Response.TrySkipIisCustomErrors = true;                      // optional on IIS Express but mandatory for production

    }

}



Intercepting exceptions globally


public class Global : System.Web.HttpApplication                     // central error handling in the global application file

{

    protected void Application_Error(object sender, EventArgs e)     // fired whenever an unhandled exception occurred

    {

        Exception myException = HttpContext.Current.Server           // retrieve the exception (can be done in any custom code file!)

                                           .GetLastError()

                                           .GetBaseException();

        // log/save/email the exception ...

        Response.Write("generic error message");

        Response.StatusCode = 200;

        Server.ClearError();

        Response.End();

    }

}

Enabling tracing


<%@ Page ... Trace="true" TraceMode="SortByCategory"%>               <!-- enable tracing for one page (the sort parameter is optional       -->


<configuration>

  <system.web>

    <trace mostRecent   = "true"                                     <!-- enable tracing for all pages                                      -->

           enabled      = "true"

           requestLimit = "100"                                      <!-- keep only the 100 most recent requests                            -->

           pageOutput   = "false"                                    <!-- don't show tracing on the ASPX pages themselves                   -->

           localOnly    = "true" />                                  <!-- restricted to local clients (on the same machine as the server)   -->


http://localhost:63068/Trace.axd                                     // access the accumulated traces vai the virtual page Trace.axd



Creating trace messages


Trace.Write("Trace message");                                        // add a custom message (or a warning) to the trace stream

Trace.Warn("aspx.page", "Trace message", exception);                 // optional: add a category (for sorting) and an exception

HttpContext.Current.Trace.Write/Warn                                 // access the ASP.NET tracing from outside of web forms



Redirecting trace streams


<trace writeToDiagnosticsTrace="true"                                <!-- redirect web specific tracing to the standard trace framework    -->

       pageOutput="true" enabled="true"/>


<configuration>                                                      <!-- redirect standard tracing to the web specific ASP.NET tracing    -->

  <system.diagnostics>

    <trace autoflush="true">

      <listeners>

         <add name="webListener"

              type="System.Web.WebPageTraceListener, System.Web, Version=4.0.0.0, Culture=neutral, PublicKeyToken=b03f5f7f11d50a3a"/>

      </listeners>

    </trace>

  </system.diagnostics>
