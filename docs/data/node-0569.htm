Install gems


gem list                                    # display all locally installed gems

gem list --remote                           # list gems from remote server - or browse https://rubygems.org/


gem install chronic                         # install a gem

gem install -v 0.10.2 chronic               # install a specific version

gem update                                  # upgrade to latest version(s)

gem uninstall chronic                       # remove a gem



Manage dependencies with bundler


Create file called "Gemfile":

+-------------------------------+

| source 'https://rubygems.org' |           # the repository

| gem 'nokogiri'                |           # allow any version

| gem 'rack', '~>1.5'           |           # require 1.5 or above (but below 2)

+-------------------------------+


> bundle install                            # creates a "Gemfile.lock"





require 'rubygems'                          # ???

require 'bundler/setup'                     # adds the gems from the Gemfile.lock to the load path

require 'nokogiri'                          # include the preselected gem versions

Bundler.require(:default)                   # ...or auto-require all gems from the Gemfile.lock



Creating gems manually


Create a folder structure:


my_gem

+- lib                                      # sources

+- test                                     # unit tests, can also be called spec

+- doc                                      # usually generated via rdoc

+- bin                                      # start scripts

+- my_gem.gemspec                           # manifest


Populate the my_gem.gemspec:


+----------------------------------------+

| Gem::Specification.new do |s|          |

|   s.name = 'my_gem'                    |

|   s.version = '0.0.1'                  |

|   s.summary = 'Description'            |

|   s.platform = Gem::Platform::RUBY     |

|   s.files = Dir.glob("**/**/**")       |  # include all files from all folders

|   s.test_files = Dir.glob("test/*.rb") |  # or test/*_test.rb or spec/*_spec.rb

|   s.authors = ["My name"]              |

|   s.email = "me@email.com"             |

|   s.required_ruby_version = '>= 2.0.0' |  # optional

| end                                    |

+----------------------------------------+


> gem build my_gem.gemspec                  # creates my_gem-0.0.1.gem

> gem push my_gem-0.0.1.gem                 # push to rubygems.org (requires an account)


> bundle gem my_gem                         # or use bundler

> rake release                              # ...and rake
