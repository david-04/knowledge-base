Creating a base page class


public class BasePage : System.Web.UI.Page                      // inherit from the ASP page base class

{

    public BasePage()

    {

        this.PreRender += Page_PreRender;                       // register the applicable event handlers

    }


    private void Page_PreRender(object sender, EventArgs e)     // invoked when the corresponding event fires

    {

        this.Title = "Playground.APS v2";

    }


}
