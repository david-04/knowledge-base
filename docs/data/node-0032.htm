Messaging


PUBLISH      myChannel "Hello world!"               // publish a message to a channel


SUBSCRIBE    myChannel                              // subscribe for messages on the given channel

PSUBSCRIBE   myChannel*                             // subscribe with a wildcard channel name

UNSUBSCRIBE  myChannel                              // unsubscribe from a channel

PUNSUBSCRIBE myChannel*                             // unsubscribe with a wildcard channel name


PUBSUB       CHANNELS                               // list all active channels (with at least one subscriber)

PUBSUB       CHANNELS my*                           // ... optionally filtered by a channel name mask

PUBSUB       NUMSUB                                 // count the number of currently active subscribers

PUBSUB       NUMSUB my*                             // ... optionally filtered by a channel name mask

PUBSUB       NUMPAT                                 // count the number of clients with active PSUBSCRIBEs

client.publish("myChannel", "Hello world!");        // publish a message to a channel


client.on("message", function(channel, message) {   // attach an event handler for received messages

});

client.subscribe("global", "myChannel");            // subscribe to a channel
