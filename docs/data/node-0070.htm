Defining custom filters


myModule.filter("brace", function () {              // register the required filter functions

             return myBraceFunction;                 

         }).filter("skip", function () {           

             return mySkipFunction

         }).filter("take", function ($filter) {     // supports depencency injection

             return myTakeFunction

         });


function myBraceFunction(value, option1, option2) { // value filter (e.g. {{user.name | brace:'<':'>' }})

    if (angular.isString(value)) {                  // verify that a scalar value was passed in

        option2 = option2 || option1                // use a default if the user did not provide a parameter value

        return option1 + value + option2;           // transform the value as required and return the result

    }

    return value;                                   // fallback: at least return the (unmodified) input

}


function mySkipFunction(array, count) {             // array filter (e.g. ng-repeat="users | skip:10")

    if (   angular.isArray(array)                   // ensure that a collection was passed in

        && angular.isNumber(count)) {               // ... and that the options are valid as well

        return array.slice(count);                  // return a new array with the selected/modified elements

    }

    return array;                                   // fallback: at least return the (unmodified) input

}


function myTakeFunction(array, skip, take) {        // composite filter (requires the injection of $filter, see above)

    var tmp = $filter("skip")(array, skip);         // use $filter("name") to invoke another (custom) filter

    return $filter("limitTo")(tmp, take);           // built-in filters can be utilised as well

}

Built-in collection filters


limitTo:5                                           // take only the first five elements

limitTo:-5                                          // take only the last five elements


filter:{property:"value"}                           // take only elements whose with given property value

filter:myFilterFunction                             // take all elements for wich $scope.myFilterFunction returns true

                                                    // example: $scope.myFilterFunction = function(user) {return 1000 < user.id;}


orderBy:"price"                                     // order by attribute price ascending (property name must be put in quotes)

orderBy:"-price"                                    // order by attribute price descending

orderBy:getSortKey                                  // use a controller function to retrieve the sort key (no quotes here!)

                                                    // example: $scope.getSortKey = function(product) { return product.name || "?"; }

orderBy:[getSortKey, '-price']                      // sort by the generated sort key and within this group sort by price descending

Built-in number filters


currency                                            // prepend a currency symbol ($) and round to 2 decimal digits

currency:"€"                                        // use the given currency symbol (can't pass HTML code / will be masked)

number:0                                            // round to an integer value

number:2                                            // round to two decimal places

number:-1                                           // round to multiples of ±10
