Anatomy


module                                              // a module can be an application (containing controllers) or a libary

|                                                  // (used to share functionaly accross multiple applications

+-- controller                                     // the controller provides data and behaviours (functions) to the view

|    +-- $scope                                    // each controller establishes its own scope (context)

+-- directives                                     // directives usually manipulate the DOM (e.g. adding or hiding elements)

+-- filters                                        // filters are used by the view to filter arrays or format sclar values (e.g. currencies)

+-- services                                       // services expose general purpose functions that are not tied to a particular view

|    +-- decorator                                 // services can be encapsulated by applying decorators

+-- contant/value                                  // contants and values are used to configure a module (easy to adjust during unit tests)



Model and controller


var todoApp = angular.module("todoApp", []);        // create the application module (second parameter must be an array)


todoApp.controller("ToDoCtrl", function ($scope) {  // create a controller (should always end with Ctrl)


    $scope.todo = {                                 // add data to the controller (usually loaded via Ajax)

        user: "David",

        items: [

            { action: "Do this", done: false },

            { action: "Do that", done: false }

        ]

    };                          


    $scope.countItems = function () {               // functions (behaviours) to retrieve or manipulate the model

        return item.length;

    };


    $scope.addNewItem = function (task) {

        $scope.todo.items.push({

            action: task, done: false

        });

    }

});



View


<!DOCTYPE html>

<html ng-app="todoApp">                             <!-- link the application module (can be attached to any HTML element)      -->

<head>

    <script src="lib/angular.js"></script>

    <script src="script.js"></script>

</head>

<body ng-controller="ToDoCtrl">                     <!-- attach the controller to a view (can be attached to any HTML element)  -->

    <h1>{{todo.user}}'s {{countTasks()}} Tasks</h1> <!-- one-way data binding and function calls via {{ }}                      -->

    <input ng-model="newTask" />                    <!-- two-way data binding (task is added to the $scope automatically)       -->

    <button ng-click="addTask(task)">Add</button>   <!-- add event handlers (calling the controller's functions)                -->

    <table ng-hide="countTasks() == 0">             <!-- do not whot this element if the condition is fulfilled                 -->

        <tr ng-repeat="item in todo.items | filter:{done: false}  | orderBy:'action'">

            <td>{{item.action}}</td>                <!-- inline one-way data bidning                                            -->

            <td>

                <input type="checkbox"

                       ng-model="item.done" />      <!-- two-way binding against a pre-existing $scope property                 -->

            </td>

        </tr>

    </table>

</body>

</html>
