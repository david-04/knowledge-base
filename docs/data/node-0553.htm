Date and time


Time.now                                    # get the current date/time

Time.local(2021, 12, 24, 18, 30, 00, 000)   # create a fixed date/time

Time.local(2021, 12, 24)                    # everything after the year is optional

Time.utc(2021, 12, 24)                      # UTC time

Time.gm(2021, 12, 24)                       # same as above (UTC)


tomorrow = Time.now + 24 * 60 * 60          # can add/substract seconds

seconds = tomorrow - Time.now               # returns seconds between the dates

epochSeconds = tomorrow.to_i                # convert a Time to epoch seconds

tmorrow= Time.at(epochSeconds)              # convert epoch seconds back to a Time object


time = Time.local(2021,05, 26, 16, 45, 59, 123456)


                                            # 2021-05-26 16:45:59,123456

time.year                                   # 2021

time.month                                  #      05

time.day                                    #         26

time.mday                                   #         26                    synonym for day

time.hour                                   #            16

time.min                                    #               45

time.sec                                    #                  59

time.usec                                   #                      123456   microseconds

time.wday                                   #                               0=Sunday ... 6=Saturday

time.yday                                   #                               146 (day of the year)

time.utc?                                   #                               indicates if timezone is UTC

time.gmt?                                   #                               same as utc?



chronic


require 'chronic'                           # gem install chronic


Chronic.parse('last tuesday 5am')           # returns a Time object

Chronic.parse('june 29th at 1am')

Chronic.parse('in 3 years')

Chronic.parse('sep 23 2033')
