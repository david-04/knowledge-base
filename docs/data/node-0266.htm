Implementing the data model


public class Customer                                                // data class (data row)

{

    public int Id { get; set; }

    public string Name { get; set; }

}


public class CustomerRepository                                      // container class

{

    private List<Customer> Customers = new List<Customer>();


    public List<Customer> Select(int customerId)                     // select all records (for container views)

    {

        return (from c in this.Customers

                where c.Id == customerId

                select c).ToList<Customer>();

    }


    public List<Customer> Select()                                   // select one record (for details views)

    {

        return (from c in this.Customers

                select c).ToList<Customer>();

    }


    public void Insert(Customer c)                                   // add a new record

    {

        // ...

    }

}



Configurating the ObjectDataSource


<asp:ObjectDataSource ...

    TypeName           = "PlaygroundASP.App_Code.CustomerRepository" <!-- the container class                  -->

    DataObjectTypeName = "PlaygroundASP.App_Code.Customer"           <!-- the data class                       -->

    SelectMethod       = "Select">                                   <!-- the container class' UPDATE method   -->

    <SelectParameters> 

        <asp:Parameter Name="Id" Type="Int32" />

    </SelectParameters>

</asp:ObjectDataSource>
