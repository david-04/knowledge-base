$("body").scrollspy({                                 // configure the ScrollSpy

    target: "#myNavbar",

    offset: 70

})


$('[data-spy="scroll"]').scrollspy('refresh');        // refresh the ScrollSpy after manipulating the DOM


onactivate.bs.scrollspy                               // when a new item becomes active
