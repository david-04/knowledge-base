Geolocation


navigator.geolocation.getCurrentPosition(

    function onSuccess(position) {                  // onSuccess

        position.coords.latitude

        position.coords.longitude

        position.coords.accuracy                    // precision in metres

        position.coords.altitude                    // these properties have only limited support

        position.coords.altitudeAccuracy

        position.coords.heading                     // 0=north, 90=east?, ... or NaN

        position.coords.speed

        position.timestamp                          // timestamp when the coordinates were retrieved

    },

    function onError(event) {                       // onError

        event.message

        event.code                                  // 1 => permission denied, 2 => position unavailable, 3 => timeout

    },

    {                                               // options

        enableHighAccuracy : false,                 // only activate when absolutely necessary (slows things down)

        timeout            : 1000,                  // wait no more than 1 second for the result (set to 0 to wait for as long as it takes)

        maximumAge         : 60000                  // cache the result for 60 seconds (set to 0 to disable caching completely)

    }

);


var id = navigator.geolocation.watchPosition(...);  // same parameters as getCurrentPosition but is called automatically whenever the position changes

navigaror.geolocation.clearWatch(id);               // stop the watch timer
