Namespaces


<root>

  <md:element xmlns:md="http://my-domain.com/"   >  <!-- the element itself must be prefixed                           -->

    <   child                                   />  <!-- children don't inherit the parent's namespace                 -->

    <md:child                                   />  <!-- ... but they can use the prefixes defined by their ancestors  -->

    <md:child    attribute="not in namespace"   />  <!-- the element's prefix does not apply to attributes             -->

    <md:child md:attribute="    in namespace"   />  <!-- ... they have to be prefixed manually                         -->

    <md:child xmlns:md="http://my-domain.com/2" />  <!-- prefixes can be overridden                                    -->

  </dm:element>

</root>

<element xmlns="http://my-domain.com/">             <!-- define the default namespace without a prefix                 -->

  <child                                            <!-- child nodes are automatically in the default namespace, but:  -->

         attribe="not in default namespace"/>       <!-- the default namespace is never applied to attributes          -->
