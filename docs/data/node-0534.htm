HTML parsing


require 'nokogiri'                          # gem install nokogiri


document = Nokogiri::HTML(html_code)

document = Nokogiri::HTML(URI.open(url))    # requires 'open-uri'


array = document.css("h1")                  # get all elements

element = document.at("ul")                 # get the first match only


element.inner_html                          # get the innerHTML

element.at("li")                            # can continue querying in fragment via at() or css()


doc.at("ul").css("li").each do |li|         # iterate over all matches

  puts li.inner_html

end



XML parsing


require 'nokogiri'                          # gem install nokogiri


document = Nokogiri::XML(xml)               # parse the XML content


array = document.css("paragraph")           # get all elements with the given tag name

array[0].children                           # get the text content
