Namespace prefixes


                                                    <!-- the DTD can only enforce a prefix (not a namespace)    -->


<!ELEMENT element                      (#PCDATA)>   <!-- matches: <element> or <element xmlns="anyNamespace">   -->

<!ELEMENT myns:element                 (#PCDATA)>   <!-- matches: <myns:element>                                -->


<!ATTLIST element attr           CDATA #REQUIRED>   <!-- matches: <element attr="1">                            -->

<!ATTLIST element myns:attr      CDATA #REQUIRED>   <!-- matches: <element myns:attr="1">                       -->

<!ATTLIST myns:element attr      CDATA #REQUIRED>   <!-- matches: <myns:element attr="1">                       -->

<!ATTLIST myns:element myns:attr CDATA #REQUIRED>   <!-- matches: <myns:element myns:attr="1">                  -->



Configurable prefixes


<!ENTITY % ns-prefix   "myns:">                     <!-- define the prefix and the column as                    -->

<!ENTITY % ns-element  "%ns-prefix;element">        <!-- define myns:element name as a parameterised entity     -->

<!ELEMENT %ns-element; (#PCDATA)>                   <!-- define the element itself parameterised by the entity  -->


<?xml version="1.0" encoding="utf-8" ?>

<!DOCTYPE ???:element SYSTEM "external.dtd" [       <!-- dinclude the external DTD file                         -->

  <!ENTITY % ns-prefix "ns1">                       <!-- override the prefix value                              -->

]>

<ns1:element xmlns:ns1="DefaultNamespace">          <!-- at least in theory this should now be valid            -->

</ns1:element>



Implied namespaces


<!ATTLIST myElement xmlns CDATA #FIXED              <!-- implicitely attach a default names to all elements     -->
