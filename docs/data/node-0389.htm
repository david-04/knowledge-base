Document structure


<header>                                            <!-- global page header                                                                 -->

    <h1>-<h6>                                       <!-- should always contain a traditional headline                                       -->

    <nav>                                           <!-- may contain navigation                                                             -->


<footer>                                            <!-- global page footer should be short information (no sitemap, no social media icons) -->

    <header>                                        <!-- a header for the footer                                                            -->

    <nav>                                           <!-- may contain navigation                                                             -->


<nav>                                               <!-- standalone navigation                                                              -->

    <header> or <h1>-<h6>

    <aside>                                         <!-- can contain an aside                                                               -->


<aside>                                             <!-- standalone additional information, not related to the document's main content      -->


<article>                                           <!-- a "self-contained composition" (newspaper article, blog post, user comment, ...)   -->

    <header>                                        <!-- header related to this article                                                     -->

    <aside>                                         <!-- additional information tangentially related to the article                         -->

    <section>

    <footer>                                        <!-- footer related to this article                                                     -->


<section>                                           <!-- off-topic data that's still outline-relevant (e.g. list of newspaper articles)     -->

    <header>                                        <!-- should contain a header (otherwise use a <div> rather than a <section>)            -->

    <aside>                                         <!-- additional information tangentially related to the section                         -->

    <article>

    <footer>                                        <!-- footer related to this section                                                     -->


<main>                                              <!-- main content (use only once but not inside article, aside, footer, header, nav)    -->


<details open>                                      <!-- not in IE: accordion (omit the "open" attribute to show it initially collapsed)    -->

    <summary>User details</summary>                 <!-- clickable section header                                                           -->

    ...                                             <!-- content is toogled between collapse and expand on each header-click                -->

    <details>...</details>                          <!-- details can be nested                                                              -->

</details>


<h6> ... <h6>                                       <!-- does not support any attributes in HTML5                                           -->



Backwards compatibility


<style>                                             /* for older browsers only: style the semantic tags as blocks (not inline)               */

    article, aside, figure, figcaption,                

    footer, header, main, nav, section, summary {

      display: block;

    }

</style>


<script>                                            // IE < 8 does not allow styling unknown tags, workaround:

    document.createElement('article');              // after this unrelated "createElement" the CSS styles are applied

</script>

<![endif]-->



Inline markup


<time datetime="2016-12-24"                         <!-- datetime can be omitted if innerHTML has the right format                          -->

      datetime="2016-12"

      datetime="2016"

      datetime=     "12-24"                         <!-- 24th December (in any year)                                                        -->

      datetime=           "19:30"

      datetime=           "19:30+08:00"             <!-- local time with the given timezone offset                                          -->

      datetime="2016-12-24 19:30+08:00"

      datetime="2016-12-24T19:30"                   <!-- the "T" is still supported but deprecated                                          -->

      datetime="2016-12-24 19:30"

      datetime="2016-12-24 19:30:45"

      datetime="2016-12-24 19:30:45.687"

      datetime="PT5D9H2M7S"                         <!-- a duration of 5 days + 9 hours + 2 minutes + 7 seconds                             -->

      pubdate>                                      <!-- the publishing date (only applicable inside <article>)                             -->

    24<sup>th</sup> December '16 (7.30pm)           <!-- can have any format if teh datetime attribute is set                               -->

</time>


<address>                                           <!-- site owner's/operator's contact details                                            -->



Metadata


<p itemscope itemtype="http://schema.org/Person">   <!-- Microdata - standardised and understand by search engines                          -->

    <h3 itemprop="name">David</h3>                  <!-- itemscope creates a wrapper for itemtype-typed data                                -->

    For more details visit:                         <!-- properties are marked with the itemprop attribute                                  -->

    <a itemprop="url" href="mydomain.com">...</a>   <!-- itemscopes can be nested                                                           -->

</p>                                                <!-- catalogue of standarised types and properties:: https://schema.org/docs/full.html  -->



<header role="banner">                              <!-- ARIA (Accessible Rich Internet Applications) - accessibility for sceen readers     -->

                                                    <!-- elements are ehanced with a role attribute that uses standardised values           -->



<span property="dc:title">..............</span>     <!-- RDFa (Resource Description Framework                                               -->

<span rel="foaf:interest">..............</span>     <!-- a rather verbose (complicated syntax)                                              -->

<span resource="urn:ISBN:0752820907">...</span>     <!-- defines a set of additional attributes                                             -->



<div class="vcard">                                 <!-- Microformats                                                                       -->

    <h3 class="fn">Mike Rowe</h3>                   <!-- class names (that don't resolve to styles) are used to add semantics               -->

</div>                                              <!-- predominatly used for contact details (hCard) and event listings (hCalendar)       -->
