Floating


float:    {left|right|none};                        /* float the element to the left, to the right or not at all                 */


clear:    {left|right|both|none};                   /* the element should have no floats to its left, right, ...                 */


<div class="outerDiv">                              <!-- this outer DIV might be smaller than its content                       -->

    <div style="height:100px;float:left;"></div>    <!-- ... if it contains floating elements                                   -->

    <div style="clear:both;"></div>                 <!-- add a dummy DIV with clear to force the outer DIV to the right height  -->

</div>


.outerDiv:before,                                   /* alternative approach: instead of adding a dummy DIV inside, define        */

.outerDiv:after { content: "";                      /* these styles for the outer DIV element ("micro clearfix hack")            */

                  display: table; }                   

.outerDiv:after { clear:   both;  }

.outerDiv       { zoom:    1;     }                 /* this rule is only required for IE 6+7                                     */
