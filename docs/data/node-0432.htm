Event handlers (AWT)


public class MyApplet extends Applet

                      implements MouseListener {    // implement the listener interface


    public void init() {

        this.addMouseListener(this);                // register the event handler

    }


    public void mouseClicked(MouseEvent e) {        // implement the required callback methods

        this.getGraphics().drawString(

            e.getX() + " " + e.getY(), 150, 150

        );

    }

}





class MyEventListener implements MouseListener {    // when implementing the lister interface, all callback methods must be implemented

    public void mouseClicked (MouseEvent e) {...}

    public void mouseEntered (MouseEvent e) {...}

    public void mouseExited  (MouseEvent e) {...}

    public void mousePressed (MouseEvent e) {...}

    public void mouseReleased(MouseEvent e) {...}

}





class MyEventListener extends MouseAdapter {        // when extending an adapter, only the actually required callback methods are needed

    public void mouseClicked (MouseEvent e) {...}

}





this.addMouseListener(new MouseAdapter() {          // the event handler can also be defined as anonymous class on the fly

    public void mouseClicked(MouseEvent e) { }

});





myButton.addActionListener(e -> {...});             // if the event listener has a functional interface, lambda expressions can be used



Events (AWT)


EventObject                                         // base class for all events

+- AWTEvent                                         // AWT specific events

   +- ActionEvent                                   // a button is pressed, an item is clicked or a menu is selected

   +- AdjustmentEvent                               // scroll event

   +- ComponentEvent                                // a component is shown, hidden, removed or resized

   +- ContainerEvent                                // a component is added to or removed from a container

   +- FocusEvent                                    // a component obtains or loses the focus

   +- InputEvent                                    // a component received some form of input

   |  +- KeyEvent                                   // input was received from the keyboard

   |  +- MouseEvent                                 // mouse movements and clicks

   +- ItemEvent                                     // a checkbox or list item is clicked or a selection or menu item is chosen

   +- MouseWheelEvent                               // mouse scrolling

   +- TextEvent                                     // the content of a text field or text area changes

   +- WindowEvent                                   // a window is opened, closed, activated, ...



Listeners (AWT)


ActionListener                                      // a button is pressed, an item is clicked or a menu is selected

AdjustmentListener                                  // scroll event

ComponentListener          ComponentAdapter         // a component is shown, hidden, removed or resized

ContainerListener          ContainerAdapter         // a component is added to or removed from a container

FocusListener              FocusAdapter             // a component obtains or loses the focus

ItemListener                                        // a checkbox or list item is clicked or a selection or menu item is chosen

KeyListener                KeyAdapter               // input was received from the keyboard

MouseListener              MouseAdapter             // mouse movements and clicks

MouseMotionListener        MouseMotionAdapter       // mouse was moved

MouseWheelListener                                  // mouse scrolling

TextListener                                        // the content of a text field or text area changes

WindowFocusListener                                 // the window obtains or loses the focus

WindowListener             WindowAdapter            // a window is opened, closed, activated, ...
