Alignment


vertical-align: {                                   /* works only for inline elements and table cells                    */

                |baseline                           /* baseline ...... regular alignment on the same text-line           */

                |middle                             /* middle ........ align vertically centered with the text           */

                |sub                                /* sub ........... align to the same bottom line as a subscript      */

                |super                              /* super ......... align to the same top line as a superscript       */

                |text-top                           /* text-top ...... align to the same top line as the text            */

                |text-bottom                        /* text-bottom ... align to the same bottom line as the text         */

                |top                                /* top ........... align to the top of the parent's content area     */

                |bottom};                           /* bottom ........ align to the bottom of the parent's content area  */


                                                   


position:      relative;                            /* alternative centering technique:                                  */

top:           50%;                                 /* leave the upper 50% empty                                         */

margin-top:    -0.5em;                              /* and shift back half of the element's actual size                  */

margin:        0 auto;                              /* centre horizontally                                               */
