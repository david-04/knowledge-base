Flexbox container


display:         {flex|inline-flex};                    /* mark as flexbox (block or line)                          */


flex-flow:       {row|row-reverse|column|column-reverse}/* shortcut: flex-direction                                 */

                 {nowrap|wrap|wrap-reverse};            /*           flex-wrap                                      */


flex-direction:  {row                                   /* horizontal left-to-right                                 */

                 |row-reverse                           /*            right-to-left                                 */

                 |column                                /* vertical   top-to-bottom                                 */

                 |column-reverse};                      /*            bottom-to-top                                 */


                                                       


flex-wrap:       {nowrap                                /* squeeze all children into one line                       */

                 |wrap                                  /* wrap across multiple lines if necessary                  */

                 |wrap-reverse};                        /* wrap lines but reverse the direction (ltr <-> rtl)       */


                                                       


justify-content: {flex-start                            /* align to the left                    [xxx    ]           */

                 |flex-end                              /* align to the right                   [    xxx]           */

                 |center                                /* centre in the middle                 [  xxx  ]           */

                 |space-between                         /* justified                            [x  x  x]           */

                 |space-around};                        /* justified with padding on both sides [ x x x ]           */


                                                       


align-items:     {flex-start|flex-end                   /* horizontal layout: align at top/bottom                   */

                 |centre                                /*                    align in the middle                   */

                 |stretch                               /*                    stretch from top to bottom            */

                 |baseline};                            /*                    align that text has the same baseline */


                                                       


align-content:   {flex-start|flex-end                   /* horizontal layout: align at top/bottom                   */

                 |center                                /*                    align in the middle                   */

                 |stretch                               /*                    scale to fill the whole height        */

                 |space-between                         /*                    justify with no padding/margin        */

                 |space-around};                        /*                    justify with padding/margin           */


                                                       



Flexbox children


order:       1;                                         /* sort order                                               */


                                                       


flex-grow:   0;                                         /* indicator of how much this element can be stretched      */


                                                       


flex-shrink: 1;                                         /* indicator of how much this element can be shrunk         */


flex-basis:  {100px|10%                                 /* set the preferred size of the element                    */

             |auto                                      /* redirect to the width (or height) property               */

             |content|{max|min|fit}-content};           /* not well supported: determine size base on the content   */


flex:        0                                          /* flex-grow                                                */

             0                                          /* flex-shrink                                              */

             auto;                                      /* flex-basis                                               */
