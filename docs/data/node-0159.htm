Selections


$(document)                                         // the whole document

$(this)                                             // the JavaScript context (usually the element that raised an event)

$("css_Selector")                                   // a CSS style selector (see below)


.add({"selector"|elementOrSelection|"htmlCode"})    // add another selection or a fragment of HTML code to the current selection

.not({"selecor"|selection|function)                 // remove all selection's root elements that meet the criteria

.eq(index|-indexFromEnd)                            // select only the root element with the given index

.slice(startIndex,endIndex)                         // select all elements between [startIndex;endIndex[


.parent("selector")                                 // get the direct parent (that must also match the selector if provided)

.parents("selector")                                // get all direct and indirect parents (that match the selector if provided)

.closest({"selector"|selection|element})            // get the closest parent that matches the specified criteria


.siblings("selector")                               // get all brother elements on the same DOM level (excludes the elements of the source selection)

.prev("selector")                                   // get the directly preceding brother element (that must also match the selector if provided)

.next("selector")                                   // get the directly following brother element (that must also match the selector if provided)

.prevAll("selector")                                // get all previous brother elements (that match the selector if provided)

.nextAll("selector")                                // get all following brother element (that match the selector if provided)


.children("selector")                               // get all direct children (that match match the selector if provided)

.find({"selector"|elementOrSelction})               // select direct and indirect children



Selectors


"*"                                                 // all elements

".class"                                            // elements with a specific CSS class

"h1"                                                // elements with a specific element/tag name

"#ID"                                               // elements with a specific ID

":visible"                                          // elements that satsify a spcific filter


"h1.class1.class2#ID:visible"                       // elements must match all of the conditions


"selector1, selector2"                              // elements must match any of the selectors

"parentSelector   indirectChildSelector"            // indirectChildSelector must match anywhere within the parentSelector

"parentSelector > directChildSelector"              // directChildSelector must match a direct child of parentSelector

"previousSelector + targetSelector"                 // targetSelector must have a directly preceeding brother matching previousSelector

"previousSelector ~ targetSelector"                 // targetSelector must have any preceeding brother matching previousSelector



Attributes


"[attribute]"                                       // must have the given attribute

"[attribute=value]"                                 // the attribute must have the given value

"[attribute!=value]"                                // the attribute is either not set at all or does otherwise have another value

"[attribute^=prefix]"                               // the attribute's value must start with the given prefix

"[attribute|=prefix]"                               // the attribute's value must either be "prefix" or start with "prefix-"

"[attribute*=substring]"                            // the attribute's value must contain the given substring

"[attribute~=word]"                                 // the attribute's value must contain the given word (separated by whitespace)

"[attribute$=suffix]"                               // the attribute's value must end with the given suffix

"[attribute1=...][attribute2=...]"                  // element must match all the specified attribute rules



Parents


":not('selector')"                                  // must not match the given selector

":empty"                                            // must not have any children (neither nodes nor text)

":parent"                                           // must be a parent node that contains child nodes and/or text

":has({'selector'|element})"                        // must have a direct or indirect child that match the selector

":contains('text')"                                 // must contain the given text in its innerHTML



Children


":first"                                            // the first element in the selection

":last"                                            

":even"                                             // elements with an even index

":odd"                                             

":eq(index)"                                        // the selection's index'th element

":gt(index)"                                        // all elements with an index greater than the given one

":lt(index)"                                        // all elements with an index smaller than the given one


":first-child"                                      // the first child within its parent container

":last-child"                                      

":first-of-type"                                    // like first-child but applied to each element type/tag separately

":last-of-type"                                    

":nth-child(index)"                                 // the index'th child within its parent container

":nth-last-child(index)"                            // like nth-child but couting backwards from the end

":nth-of-type(index)"                               // like nth-child but applied to to each element type/tag separately

":nth-last-of-type(index)"

":only-child"                                       // element must be its parent's only child

":only-of-type"                                     // like only-child but applied to each element type/tag separately



Properties

":enabled"                                          // must be enabled
