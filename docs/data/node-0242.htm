Using the UpdatePanel


<asp:TextBox ... />                                              <!-- data outside the panel is not reloaded but still posted back -->

<asp:UpdatePanel ... RenderMode = "Block"                        <!-- the content as shown as block (rather than inline)           -->

                     UpdateMode = "Conditional" />               <!-- only update when the specified trigger occurs                -->

    <ContentTemplate>

        <asp:Label ... />                                        <!-- this label is updated by the AddGreeting event handler       -->

        <asp:Button ... OnClick="Button_Click" />                <!-- the button reloads the panel and must have an OnClick method -->

    </ContentTemplate>

    <Triggers>

        <asp:AsyncPostBackTrigger ControlID = "Button2"          <!-- make elements outside of the panel trigger Ajax calls        -->

                                  EventName = "Click" />

    </Triggers>

</asp:UpdatePanel>

<asp:Button ... OnClick="Button_Click"/>                         <!-- the button must have an OnClick method as well               -->

Enabling the browser's back-navigation


<asp:ScriptManager ... EnableHistory = "true"                    <!-- enable back-navigation for Ajax requests                    -->

                       OnNavigate    = "OnNavigate" />           <!-- register the server-side navigation handler                 -->



protected void Button1_Click(object sender, EventArgs e)

{

    this.UpdateDateAndTime(DateTime.Now.ToString());             // redirect Ajax event handlers to a separate method

}


public void UpdateDateAndTime(string dateAndTime)                // handles Ajax and back-navigation

{

    if (dateAndTime == null)

    {

        this.Label1.Text = "Please push the button.";            // initial state before any Ajax request

    }

    else

    {

        this.Label1.Text = dateAndTime;                          // use only the data passed as parameter to update the page

    }

    if (   ScriptManager1.IsInAsyncPostBack

        && !ScriptManager1.IsNavigating)

    {

        ScriptManager1.AddHistoryPoint(                          // add a history entry that the user can return back to

            "myIndexPoint", dateAndTime, "Page title");          // data can be passed as NameValueCollection object as well

    }

    else

    {

        Page.Title = "Page title";                               // update the page title (for display beneath the back-button)

    }

}


protected void OnNavigate(object sender, HistoryEventArgs e)     // this method during backward navigation

{

    this.UpdateDateAndTime(e.State["myIndexPoint"]);

}
