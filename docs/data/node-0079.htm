angular.module("myApp", ["ngRoute", "ngResource"])  // require the ngRoute module (angular-route.js must be installed)

    .config(myConfigFunction)

    .controller("myCtrl", myCtrlFunction);


function myConfigFunction($locationProvider,        // request the location/routing services to be injected

                          $routeProvider,

                          $route,

                          $routeParams) {


    $locationProvider.html5Mode(true);              // change the browser's URL without actually reloading the file


    $routeProvider.when("/list", {                  // define a route (should start with "/", otherwise $location.path will be prepended)

        caseInsensitiveMatch : true,                // match the route case sensitive or insensitive

        templateUrl          : "/list.html",        // the file with the HTML fragment (can also be a function that returns the filename)

        template             : "<b>{{id}}</b>",     // the HTML fragment to show (can also be a function that returns the fragment)

        redirectTo           : "/list-new",         // redirect to another route (can also be a function that returns the new path)

        reloadOnSearch       : true,                // the route will reload when $location.search and $location.hash change

        controller           : "myCtrl",            // set the controller to be associated with the view

        controllerAs         : "myCtrlAlias",       // an alias to be used for the controller

        resolve              : {                    // define a set of dependencies for the controller

            data: function ($resource) {            // the controller can request the "data" property to be injected (see below)

                resource = $resource(...);          // example only: create an access object to perform RESTful service calls

                return resource.query();            // this returns a promise that triggers (delays) the controller's instanciation

            }

        },

    });


    $routeProvider.when("/edit/:id/:data*", {       // routes can use placeholders (:data* can match multiple segments at once)

        templateUrl: "/edit.html"

    });


    $routeProvider.otherwise({                      // default/fallback rule

        templateUrl: "/tableView.html"

    });

}

   

function myCtrlFunction ($scope, $location, data) { // can request the injection of "data" as defined in the "resolve" property above


    $scope.data = data;                             // transfer the data provided by the route into the controller's $scope


    $scope.$on("$routeChangeSuccess", function () { // route-changes trigger events that can be handled

        if(0==$location.path().indexOf("/edit/")) { // check if a given path/route has been activated

            var id = $routeParams["id"];            // retrieve named parameters from the route

        }

    });


    $scope.$on("$routeChangeStart", function(){ }); // triggered before the route is changed

    $scope.$on("$routeUpdate",      function(){ }); //  triggered when the route is refreshed

    $scope.$on("$routeChangeError", function(){ }); // triggered if the route cannot be changed


    $route.reload();                                // reload the current route

    $location.path("/list");                        // manually change the current route


    $route.routes                                   // collections of the routes defined through the $routeProvider

    $route.current.controller                       // provides information about the active route

    $route.current.locals.$scope

}
