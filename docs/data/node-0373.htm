Date and time


secondsSince1970 = systime()                          # current time in seconds since 01.01.1970 (local time)


secondsSince1970 = mktime("2015 12 24 18 30 00")      # the specified time in seconds since 01.01.1970 (local time)


secondsSince1970 = mktime("2015 12 24 18 30 00 -1")   # DST flag: =0 ... standard time (time is not modified)

                                                      #           >0 ... DST (adds one hour to the given string)

                                                      #           <0 ... autodetect DST (this is the default value)


string = strftime()                                   # format the current time with PROCINFO["strftime"]

string = strftime("%H:%M:%S", secondsSince1970, utc)  # utc can be 0 (local time) or 1 (convert to UTC)



Time formats


    # 06.01.2015      24.12.2015                       # Monday   Tuesday      ...   Saturday      Sunday    

    # -----------------------------------              # ---------------------------------------------------------------------

%d   "06"            "24"                          %u   "1"       "2"               "6"           "7"

%e   " 6"            "24"                          %w   "1"       "2"               "6"           "0"

%m      "01"            "12"                       %a   "Mon"     "Tue"             "Sat"         "Sun"

%b      "Jan"           "Dec"                      %A   "Monday"  "Tuesday"         "Saturday"    "Sunday"

%B      "January"       "December"

%G         "2015"          "2016"                      # Other 

%Y         "2015"          "2016"                      # ---------------------------------------------------------------------

%C         "20"            "20"                    %j    day of the year (001–366)

%y           "15"            "16"                  %V    week of the year (01-53) with the 1. Monday as the 1. day of week one

%F   "2016-01-06"    "2015-12-24"                  %W    week of the year (00-53) with the 1. Monday as the 1. day of week one

%D   "01/06/2015"    "12/24/2015"                  %U    week of the year (00-53) with the 1. Sunday as the 1. day of week one



    #  08:01:03          17:33:59                      # Timezone

    # -----------------------------------              # ---------------------------------------------------------------------

%H    "08"              "17"                       %z   "+1100"

%k    " 8"              "17"                       %Z   "AUSEDT"

%I    "08"              "05"

%l    " 8"              " 5"                      

%M      "01"              "33"                     

%S        "03"              "59"                      # Literals

%p             "AM"               "PM"                # ---------------------------------------------------------------------

%r    "08:01:03 AM"     "05:33:59 PM"              %%    literal ‘%’.

%T    "08:01:03"        "17:33:59"                 %t    tabulator

%R    "08:01"           "17:33"                    %n    newline
