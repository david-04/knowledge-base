Compression with zlib


require 'zlib'                              # standard library


comressed = Zlib::Deflate.deflate('text')   # compress and uncompress data in memory

uncompressed = Zlib::Inflate.inflate(data)


Zlib::GzipWriter.open('abc.gz') do |stream| # compress and uncompress files while reading/writing

  stream.write 'data'

end

Zlib::GzipReader.open('abc.gz') do |stream|

  puts stream.read

end
