    protected void Button_Click(object sender,              // callback function

Directives


<%@ Page             %>                                     <!-- describes a content page                                               -->

<%@ Master           %>                                     <!-- describes a master page                                                -->

<%@ Control          %>                                     <!-- describes a user control                                               -->

<%@ WebHandler       %>                                     <!-- Enables a page to be used as an HttpHandler                            -->


<%@ Implements       %>                                     <!-- specify base class when no code-behind file is used                    -->

<%@ Import           %>                                     <!-- like the C# using command                                              -->


<%@ Register         %>                                     <!-- similar to @Import, but imports exactly one user control               -->

<%@ PreviousPageType %>                                     <!-- declare the source of a cross-page posting                             -->

<%@ MasterType       %>                                     <!-- include a master page into a content page                              -->


<%@ Assembly         %>                                     <!-- the sames as adding references throught he solution explorer           -->

<%@ OutputCache      %>                                     <!-- enable output caching                                                  -->

<%@ Reference        %>                                     <!-- compile the specified page or user control along with the current page -->
