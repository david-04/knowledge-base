Validator syntax


<asp:RequiredFieldValidator ControlToValidate    = "TextBox1"          <!-- bind to an input control                                     -->

                            Text                 = "*"                 <!-- overrides ErrorMessage, can contain HTML code (like <img>)   -->

                            ErrorMessage         = "Name is required"  <!-- shown in the ValidationSummary                               -->

                            Display              = "Dynamic"           <!-- only visible on error (Static: fixed size placeholder)       -->

                            EnableClientScript   = "true"              <!-- validation through JavaScript (enabled by default)           -->

                            SetFocusOnError      = "false"             <!-- focus on this field in case of an error (disable by default) -->

                            ValidationGroup      = "Group1" />         <!-- group all items that belong to one submit button             -->


<asp:Button                 ValidationGroup      = "Group1" />         <!-- trigger all validators in this group (only) upon submit      -->


<asp:ValidationSummary      ValidationGroup      = "Group1"            <!-- show all error messages (or only the specified group)        -->

                            HeaderText           = "Your mistakes:"    <!-- headline for the report                                      -->

                            DisplayMode          = "BulletList"        <!-- show as BulletList, List or SingleParagraph                  -->

                            ShowSummary          = "true"              <!-- show the summary on the page                                 -->

                            ShowMessageBox       = "true"              <!-- show the summary as a JavaScript alert                       -->

                            ShowModelStateErrors = "true" />           <!-- include errors raised by the model binding                   -->



Standard validators


<asp:RequiredFieldValidator     ... InitialValue      = "<name>" />    <!-- file must not be empty (or different from the InitalValue)   -->


<asp:CompareValidator           ... ControlToValidate = "Mail1"        <!-- compare tow input controls (equal, less than, ...)           -->

                                    ControlToCompare  = "Mail2"        <!-- or use ValueToCompare to compare against a constant value    -->

                                    Operator          = "Equal" />


<asp:CompareValidator           ... Type              = "Integer"      <!-- require input of a specific data type                        -->

                                    Operator          = "DataTypeCheck"<!-- check if the type matches ...                                -->

                                    ValueToCompare    = "18"/>         <!-- ... or compare it against a constant value                   -->


<asp:RangeValidator             ... MinimumValue      = "0"            <!-- number within a given range                                  -->

                                    MaximumValue      = "120"

                                    Type              = "Integer" />   <!-- enforce integer values (defaults to string)                  -->


<asp:RegularExpressionValidator ... ValidationExpression="[a-z]+" />   <!-- string must match the regular expression                     -->

Custom validators


<asp:CustomValidator ... OnServerValidate         = "OnValidate"       <!-- reference server-side validation in the code-behind file     -->

                         ClientValidationFunction = "ValidateInput" /> <!-- reference embedded JavaScript for client-side validation     -->


<script>                                                               // embed (or link to) JavaScript code for client-side validation

    function ValidateInput(source, args) {

        var email = args.Value;                                        // retrieve the value to be validated

        args.IsValid = true;                                           // set args.IsValid to indicate the validation outcome

    }

</script>


protected void OnValidate(object source,                               // custom server-side validator in the code-behind file

                          ServerValidateEventArgs args)

{

    string email = args.Value;                                         // retrieve the value to be validated

    args.IsValid = false;                                              // set args.IsValid to indicate the validation outcome

}



Disabling unobtrusive validation


<configuration>

  <appSettings>

    <add key   = "ValidationSettings:UnobtrusiveValidationMode"

         value = "None" />                                             <!-- disable unobtrusive validation in web.config                 -->

  </appSettings>


Page.UnobtrusiveValidationMode = UnobtrusiveValidationMode.None;       // set it programatically in Page_Load or Global.Application_Start



Enabling unobtrusive validation


<configuration>

  <appSettings>

    <add key   = "ValidationSettings:UnobtrusiveValidationMode"

         value = "WebForms" />                                        <!-- enable unobtrusive validation in web.config                   -->


Page.UnobtrusiveValidationMode = UnobtrusiveValidationMode.WebForms;  // set it programatically in Page_Load or Global.Application_Start


public class Global : System.Web.HttpApplication

{

    protected void Application_Start(object sender, EventArgs e)

    {

        ScriptManager.ScriptResourceMapping.AddDefinition(            // register jQuery in the global application file

            "jquery",                                                 // symbolic name (alias)

            new ScriptResourceDefinition

            {

                Path = "~/Scripts/jquery-3.1.0.min.js"                // file reference to the actual JavaScript file

            }

        );

    }

}
