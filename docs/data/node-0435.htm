Context menus


myFrame.addMouseListener(new MouseAdapter() {       // the pop-up menu is activated via a right-click

    public void mousePressed(MouseEvent me) {

        if (me.getButton() == MouseEvent.BUTTON3) {

            JPopupMenu menu = new JPopupMenu();     // create and populate the pop-up menu

            menu.add(new JMenuItem("action 1"));

            menu.add(new JMenuItem("action 2"));

            menu.setInvoker(myFrame);               // attach the menu to the parent container

            menu.setLocation(                       // position the menu relative to the click location

                  me.getX()

                + menu.getPreferredSize().width,

                me.getY()

            );

            menu.setVisible(true);                  // show the menu

           

        }

    }

});
