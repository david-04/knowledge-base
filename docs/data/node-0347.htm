Backgrounds


background-color     : red;                         /* uniform background color (can also be set to transparent)                      */


background-image     : url("my-image.jpg");         /* use an image as background                                                     */

background-image     : url("1.jpg"), url("2.jpg");  /* CSS3: multiple images (use multiple values for the other properties as well)   */


background-repeat    : repeat;                      /* (no-)repeat, repeat-{x|y}, round (repeat+stretch), space (repeat+distribute)   */


background-position  : 10px 50%;                    /* positioning can be relative or absolute                                        */

background-position  : right top;                   /* can also position to left, right, top, bottom, center                          */


background-attachment: fixed;                       /* fixed .... within the viewport (should be used on the <html> or <body> only)   */

                                                    /* local .... scrolls within the viewport and within the element itself           */

                                                    /* scroll ... scrolls within the viewport but is fixed within the element itself  */


background-clip      : border-box;                  /* select which parts of the element are covered by the background:               */

                                                    /* border-box .... the background covers the whole element (including the border) */

                                                    /* padding-box ... the background excludes the border (but includes padding)      */

                                                    /* content-box ... the background excludes the border and the padding             */


background-origin    : border-box;                  /* select the frame of reference for the background-position property:            */

                                                    /* border-box .... relative to the whole element (including the border)           */

                                                    /* padding-box ... relative to the client area (excluding only the border)        */

                                                    /* content-box ... relative to the content (excluding the border and padding)     */


background-size      : 50% auto;                    /* set the size of the background, can also be set to:                            */

                                                    /* contain ... scale the image to fully fit into the container                    */

                                                    /* cover ..... scale and crop the image to fully fill the container               */


background-blend-mode: {normal|multiply|screen      /* not IE: overlay mode for backgrounds (coolours, images, gradients)             */

                       |overlay|darken|lighten

                       |color-dodge|saturation

                       |color|luminosity};


background           : red                          /* shortcut: background-color                                                     */

                       url("my-image.jpg")          /*           background-image                                                     */

                       no-repeat                    /*           background-repeat                                                    */

                       fixed                        /*           background-attachment                                                */

                       10% 100px;                   /*           background-position                                                  */



Gradients


background-color: red;                              /* set a uniform background first (fallback for older browsers)                   */

background-image: linear-gradient(...);             /* gradients can be used instead of the url() function                            */


                                                    /* generator: http://www.colorzilla.com/gradient-editor                           */


background: linear-gradient(to bottom left,         /* can be to top, to bottom left, ... or 0deg...360deg                            */

                            #000000 0%,             /* as many colour stops as required                                               */

                            #ff00ff,                /* the percent-location can be omitted (defaults to the middle)                   */

                            #ffffff 100%);


background: repeating-linear-gradient(to top,       /* repeat the gradient multiple times                                             */

                                      yellow,       /* intermediate colour stops can be assigned as needed                            */

                                      green 10%);   /* the upper limit can be set in px as well (10% = repeat 10x)                    */


background: radial-gradient(circle at 10% 100px,    /* radial gradient with an optional centre point (default: at center)             */

                            red 0%,                 /* colour stops as required                                                       */

                            yellow 100px,           /* the stop's location can be omitted or specified as absolute value              */

                            green 100%);


background: radial-gradient(ellipse, red, yellow);  /* like circle, but stretched to the container's width and height                 */
