Margin and padding


margin:    3em;                                     /* padding outside of the border                                            */

margin:    0 auto;                                  /* top+bottom, left+right (set to auto to centre horizontally )             */

margin:    5px 10px 20px 5px;                       /* top, left, bottom, right                                                 */

margin-{top|bottom|left|right}: 10%;                /* set a margin for only one side                                           */


padding:    3em;                                    /* padding inside of the border                                             */

padding:    0 2%;                                   /* top+bottom, left+right (does not support auto)                           */

padding:    5px 10px 20px 5px;                      /* top, left, bottom, right                                                 */

padding-{top|bottom|left|right}: 10%;               /* set a margin for only one side                                           */


box-sizing: content-box;                            /* width and height meassure the content (excluding padding/border/margin)  */

box-sizing: border-box;                             /* width and height meassure content+padding+border (excluding the margin)  */
