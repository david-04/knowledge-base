Show and hide


show                                               <!-- visible in every context (display:block)                          -->

visible{xs|sm|md|lg}-{inline|block|inline-block}   <!-- visible only for the selected screen size                         -->

visible-print-{block|inline|inline-block}          <!-- visible only when printing                                        -->

sr-only                                            <!-- visible only for screen readers                                   -->


hidden                                             <!-- hidden (display:none) in every context                            -->

hidden-{xs|sm|md|lg}                               <!-- hidden only for the selected screen size                          -->

hidden-print                                       <!-- hidden only when printing                                         -->


invisible                                          <!-- invisible (but still occupying empty space)                       -->

text-hide                                          <!-- opacity:0 + font-size: 0px                                        -->
