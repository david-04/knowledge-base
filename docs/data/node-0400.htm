import org.junit.*;                                 // important: Assert must be used from the org.junit package

import junit.framework.TestCase;                    // the TestCase class seems to have remained in the junit.framework package


public class MyCalculatorTest extends TestCase {    // the class name should end with ...Test


    private MyCalculator myCalculator;


    public void setUp() {                           // initialisation before the test cases are executed

        this.myCalculator = new MyCalculator();

    }


    public void tearDown() {                        // cleanup after the test has finised

        this.myCalculator = null;

    }


    @Test

    public void testAdd() {                         // test cases with assertions (the name should start with test...)

        Assert.assertEquals(

                1+2, this.myCalculator.add(1, 2)

        );

    }

}
