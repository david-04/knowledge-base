Links


<a routerLink="/my-account">                        <!-- link to a static route                                               -->


<a [routerLink]="['/user',userId]">                 <!-- link to a dynamcally created route                                   -->


<a [routerLink]="['/user',{mode:'edit',id:5}]">     <!-- append optional parameters (/user;mode=edit,id=5 (no query string!)  -->


<a routerLink="/people"                             <!-- create a link that matches the "people" route                        -->

   routerLinkActive="active highlight"              <!-- attacht the given CSS classes if this route is the currently active  -->

   [routerLinkActiveOptions]="{ exact : true }"     <!-- don't apply the above CSS to active parent routes (default: false)   -->

   [queryParams]="{ id : '1', log : 'false' }"      <!-- append the query string ...?id=1&log=false                           -->

    fragment="anchor" >                              <!-- link to an anchor tag (http://.../route#anchor)                      -->


<button routerLink="people">                        <!-- applicable to buttons as well                                        -->

import { Router,

         ActivatedRoute,

         Params          } from "@angular/router";

import { Location        } from "@angular/common";


constructor(                                        // inject the rquired services into the component's constrcutor

    private router  : Router,

    private route   : ActivatedRoute,

    private location: Location

) { }


this.location.back();                               // go back to the previous page (like with the browser's back button)


this.router.navigate([                              // navigate to the specified URL

    "/person",                                      // /person

    this.person.id,                                 //        /3

    { mode : "view", readonly : "yes"}              //          ;mode=view;readonly=yes

]);                                                 //          +-> these are optional parameters, they are not declared

                                                    //              in the routing table, but can be retrieved by the

                                                    //              receiving component like any other route parameter

this.router.navigate(

    [this.person.id], { relativeTo: this.route }    // navigate to ./3 (relative to the current route)

);



let navExtras: NavigationExtras = {

    queryParams        : { "session_id" : "123" },  // set query string parameters

    fragment           : "anchor",                  // ... and optionally an anchor tag / fragment

    preserveQueryParams: true,                      // optionally: preserve these values when generating subsequent links

    preserveFragment   : true

};


this.router.navigate(["/login"], navExtras);        // navigate to the specified route and add query string and fragment
