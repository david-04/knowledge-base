XLinks


                                                    <!-- the processor implementation is not standardised, they browser    -->

                                                    <!-- may follow the links, but might as well ignore them completely    -->

<myElement

    xmlns:xlink   = "http://www.w3.org/1999/xlink"  <!-- must reference the correct namespace                              -->

    xlink:href    = "http://..."                    <!-- the link can be a URL or any URI (e.g. "urn:isbn:2354352332)      -->

    xlink:type    = "simple"                        <!-- simple ...... simple link like a href in HTML                     -->

                                                    <!-- extended .... see below                                           -->

                                                    <!-- locator ..... see below                                           -->

                                                    <!-- arc ......... see below                                           -->

                                                    <!-- title ....... see below                                           -->

                                                    <!-- resource .... see below                                           -->

    xlink:show    = "new"                           <!-- new ......... open the link in a new window/tab                   -->

                                                    <!-- replace ..... open the link in the current window/tab             -->

                                                    <!-- embed ....... embed the content (e.g. an image)                   -->

                                                    <!-- other ....... application specific instructions                   -->

                                                    <!-- none ........ do not process/follow the link at all               -->

    xlink:actuate = "onRequest"                     <!-- onLoad ...... load the link straight away                         -->

                                                    <!-- onRequest ... load the link when requested (e.g on click)         -->

                                                    <!-- other ....... application specific instructions                   -->

                                                    <!-- none ........ do nothing (useful for non-URL links, e.g. ISBNs)   -->

    xlink:title   = "More information"              <!-- free-form text description, used for example as tooltip           -->

    xlink:role    = "http://..."                    <!-- URI pointing to something that describes the link [type]          -->

>



Extended links


<book xlink:type="extended">                        <!-- this section contains extended links                              -->


  <title>The strange world of XLink</title>


  <author xlink:type   = "resource"                 <!-- a local resource (all content is included in this XML node)       -->

          xlink:label  = "dh"                       <!-- every extended link needs a label (must be a valid XML name)      -->

          xlink:title  = "D. Hofmann"               <!-- free-form text description, used for example as tooltip           -->

          xlink:role   = "http://..." >             <!-- URI pointing to something that describes the resource [type]      -->

    David Hofmann

  </author>


  <edition xlink:type  = "locator"                  <!-- a locator is an external link that point to an external object    -->

           xlink:href  = "urn:isbn:0688069444"      <!-- the link itself                                                   -->

           xlink:label = "edition1"                 <!-- the label is required so that arcs (links) can be defined         -->

           xlink:title = "1. Edition"               <!-- title and role are optional                                       -->

           xlink:role  = "http://tswox.com/1" />

  <edition xlink:type  = "locator"

           xlink:href  = "urn:isbn:0192839306"

           xlink:label = "edition2"

           xlink:role  = "http://tswox.com/2" />

    <details xlink:type= "title" >                  <!-- the title can be embedded as a rich-text child node               -->

      The <b>fully revised</b> 2nd edition

    </details>

  </edition>


  <edition xlink:type  = "locator"

           xlink:href  =  http://amazon.com"

           xlink:label = "shop"                     <!-- the label can be re-used if there multiple locators or resources  -->

           xlink:title = "Amazon" />                <!-- that should act as a link source or target group                  -->

  <edition xlink:type  = "locator"

           xlink:href  = "http://ebooks.com"

           xlink:label = "shop"              

           xlink:title = "eBooks.com" />


  <next xlink:type    = "arc"                       <!-- an arc is a link that connects                                    -->

        xlink:from    = "edition1"                  <!-- ... a source element (locator or [local] resource)                -->

        xlink:to      = "edition2"                  <!-- ... to a target element                                           -->

        xlink:title   = "next edition"

        xlink:arcrole = "http://..."  />            <!-- URI pointing to a description of the relationship [type]          -->

   

  <author xlink:type = "arc"

          xlink:from = "edition1"                   <!-- can link from/to external locators as well as internal resources  -->

          xlink:to   = "dh"       />


  <purchase xlink:type = "arc"

            xlink:from = "edition1"

            xlink:to   = "shop" />                  <!-- can link from/to a group of elements that share the same label    -->


  <seeAlso xlink:type="arc"                         <!-- if from or to is missing, all resources+locators within the       -->

            xlink:to="dh"  />                       <!-- extended node are targetted (here: link everything to the author) -->


</book>
