INSERT


INSERT INTO tab DEFAULT VALUES;                     -- insert a single record with only default values


INSERT INTO tab (col1,col2) VALUES (val1,DEFAULT);  -- insert a single record (columns are optional)

INSERT INTO tab (col1,col2) VALUES (v1,v2),(v3,v4); -- insert multiple records without sub-select

INSERT INTO tab (col1,col2) SELECT ...;             -- insert the results of another select


SELECT * INTO new_table FROM old_table;             -- create a new table as copy of an existing one



UPDATE


UPDATE tab set col1=DEFAULT, col2=val2 WHERE ...;   -- columns can be reset to the default value


UPDATE tab set col=val                              -- the where condition can be replaced by a sub-select

    FROM tab, tab2                                  -- the JOIN syntax does not seem to be supporeted here

    where tab1.col = tab2.col                       -- all involved tables must be joined

    and tab2.col2 = '...';



DELETE


DELETE from tab WHERE ...;                          -- delete selected records


DELETE tab FROM tab, tab2 WHERE ...;                -- works the same as for UPDATE (see above)



TRUNCATE


TRUNCATE TABLE tab;                                 -- a faster version of DELETE FROM tab



MERGE


MERGE INTO my_table mt                              -- merge data from a delta-table into a main table

USING staging_table st ON mt.col1 = st.col1         -- join the delta table against the target table

WHEN MATCHED THEN                                   -- for records that exist in both tables:

    UPDATE SET mt.col2 = st.col2                    -- update selected fields

WHEN NOT MATCHED THEN                               -- for new (delta) records:

    INSERT (col1, col2) VALUES (st.col1, st.col2)   -- insert them into the target table
