Nillable


<xs:element name     = "myElement"                  <!-- the element itself is mandatory (becaue minOccurs and          -->

            type     = "xs:int"                     <!-- maxOccurs both default to one); however:                       -->

            nillable = "true"      />               <!-- the element can be marked as nillable (the default is false)   -->


<myElement  xsi:nil  = "true"      />               <!-- in the XML the element is present but marked as xsi:nil        -->

                                                    <!-- it is treated as if it was omitted, but it still validates the -->

                                                    <!-- the XSD (despite not containing an xs:int value)               -->



Alias names


<xs:element       name = "city"

                  type = "xs:string"/>

<xs:element       name = "stadt"

                  substitutionGroup = "city" />     <!-- this element is an alias name for the existing "city" element  -->


<xs:element       name = "user"               >

  <xs:complexType>

    <xs:sequence>

      <xs:element ref  = "city"              />     <!-- both <city> and <stadt> can be inserted here                   -->

    </xs:sequence>

  </xs:complexType>

</xs:element>


<xs:element       name = "benutzer"

                  substitutionGroup = "user" />     <!-- an alias can be defined for complex elements as well           -->



Unique values


<xs:element name="contacts"                         <!-- the name must be unique within this selection                  -->

            type="addr:contactsType"

            minOccurs="0">

  <xs:unique name="phoneNums">

    <xs:selector xpath="phone"/>                    <!-- XPath of all elements this restriction applies to              -->

    <xs:field xpath="@addr:number"/>                <!-- XPath to the unique values within the selected elements        -->

  </xs:unique>

</xs:element>


<xs:key name="ssnKey">                              <!-- like xsunique but allows keyref                                -->

  <xs:selector xpath="addr:address"/>

  <xs:field xpath="@addr:ssn"/>

</xs:key>
