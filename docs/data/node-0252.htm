Configuring the SMTP server in Web.config


<configuration>

  <system.net>

    <mailSettings>

      <smtp deliveryMethod="Network" from="David &lt;me@my-domain.com&gt;">

        <network host="127.0.0.1" userName="UserName" password="Password" port="587" enableSsl="true" />

      </smtp>

    </mailSettings>

  </system.net>



Creating files instead of sending messages


<configuration>

  <system.net>

    <mailSettings>

      <smtp deliveryMethod="SpecifiedPickupDirectory" from="David &lt;me@my-domain.com&gt;">

        <specifiedPickupDirectory pickupDirectoryLocation="R:\Temp" />

      </smtp>

    </mailSettings>

  </system.net>



Sending mails


MailMessage myMessage = new MailMessage();

myMessage.Subject = "Test Message";

myMessage.Body = "Hello world!";

myMessage.From = new MailAddress("admin@my-domain.com", "Admin");

myMessage.To.Add(new MailAddress("user@my-domain.com", "User"));

SmtpClient mySmtpClient = new SmtpClient();

mySmtpClient.Send(myMessage);
