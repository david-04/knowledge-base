Enumerations


Public Enum Season As Integer  ' Integer is also de default and shouldn't be changed

    Spring                     ' 0

    Summer                     ' 1

    Autumn = 100               ' 100

    Winter                     ' 101

End Enum


Dim season As Season = Season.Summer



Iterating over enum values


For Each season As Season In Season.Spring.GetType().GetEnumValues()

    Console.WriteLine(season.ToString())

Next



Parsing strings


Dim season As Season = CType(System.Enum.Parse(GetType(Season), "Winter", True), Season)
