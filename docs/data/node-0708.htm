Defining attributes


<xs:attribute    name            = "checksum"       <!-- name of the attribute                                           -->

                 type            = "xs:int"         <!-- data type                                                       -->

                 use             = "optional"       <!-- prohibited, optional, required                                  -->

                 default         = "123"            <!-- default value (when the attribute is not set)                   -->

                 form            = "qualified"      <!-- qualified or unqualified; overrides schema.attributeFormDefault -->

                 fixed           = "true"     />    <!-- may not be altered during inheritance                           -->





<xs:attribute    ref             = "checksum" />    <!-- mirror an attribute defined on the schema's top-level           -->





<xs:anyAttribute namespace = "http://my-domain.com" <!-- allow any attribute(s) from the given namespace                 -->

                 processContents = "skip"        /> <!-- skip ..... do not validate the attribute(s)                     -->

                                                    <!-- lax ...... validate known elements and ignore everything else   -->

                                                    <!-- strict ... validate everything                                  -->





<xs:attributeGroup name="myAttributeGroup">         <!-- define a re-usable group of attributes that are used together   -->

  <xs:attribute ... />

  <xs:attribute ... />

</xs:attributeGroup> 

 

<xs:complexType name="myType">

  <xs:simpleContent>

    <xs:extension base="xs:string">

      <xs:attributeGroup ref="myAttributeGroup"/>   <!-- reference the attribute group from the type definition          -->

    </xs:extension>

  </xs:simpleContent>

</xs:complexType>



Attaching attributes


<xs:complexType name="myType">                      <!-- simple types don't support attributes by themselves             -->

  <xs:simpleContent>

    <xs:extension base="xs:string">                 <!-- they must be extended (with nothing) first                      -->

      <xs:attribute      ... />

      <xs:attributeGroup ... />                     <!-- attributes can be combined (and repeated) as required           -->

      <xs:anyAttribute   ... />

    </xs:extension>

  </xs:simpleContent>

</xs:complexType>





<xs:complexType name="myType">                      <!-- complex types                                                   -->

  <xs:sequence>

    ...

  </xs:sequence>

  <xs:attribute      ... />                         <!-- attributes can be combined (and repeated) as required           -->

  <xs:attributeGroup ... />

  <xs:anyAttribute   ... />

</xs:complexType>
