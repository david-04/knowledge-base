Compilation


> lessc input.less                                  // compile to stdout

> lessc input.less output.css                       // compile into the specified output directory

> lessc --clean-css input.less output.min.css       // create a minified version (requires the clean-css plugin to be installed)





<link rel="stylesheet/less"                         <!-- include uncompiled less source files like regular CSS styles         -->

      type="text/css" href="styles.less" />

<script src="less.js" type="text/javascript">       <!-- and add a runtime-compiler provided as JavaScript                    -->

</script>

Imports


@import "style-1.css";                              // will be left untouched by Less (the final CSS contains the @include)

@import "style-2.less";                             // embeds the content (like #include in C)

@import "style-2";                                  // same as above (defaults to .less if the file extension is missing)

@import "style-3.php";                              // same as above (extensions other than .css are treted as Less files)


@import (reference) "style.less";                   // import only the styles that are referenced (inherited or mixed in)

@import (inline   ) "style.less";                   // inline the file content but don't parse/process it (simple text replacement)

@import (less     ) "style.less";                   // treat the input file as Less, no matter what the extension is

@import (css      ) "style.less";                   // treat the input file as CSS, no matter what the extension is

@import (once     ) "style.less";                   // this is the default: import the file only once (ignore subsequent @imports)

@import (multiple ) "style.less";                   // import the file multiple times, as often as the @import occurs

@import (optional ) "style.less";                   // if the referenced file does not exist, then silently skip this import
