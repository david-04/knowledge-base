Inheritance


<xs:complexType     name      = "myBaseType"    >   <!-- adding constraints to a base type                            -->

  <xs:sequence>

    <xs:element     name      = "name"

                    type      = "xs:string"

                    maxOccurs = "3"            />

    <xs:element     name      = "age"

                    type      = "xs:int"       />

  </xs:sequence>

</xs:complexType>


<xs:complexType     name      = "myDerivedType" >

  <xs:complexContent>

    <xs:restriction base      = "myBaseType"    >

      <xs:sequence>

        <xs:element name      = "name"              <!-- name and type can't be modified                              -->

                    type      = "xs:string"

                    maxOccurs = "1"            />   <!-- only minOccurs and maxOccurs can be overridden               -->

        <xs:element name      = "age"               <!-- all of the base type's elements must be repeated here,       -->

                    type      = "xs:int"       />   <!-- even if they don't receive any additional restrictions       -->

      </xs:sequence>

    </xs:restriction>

  </xs:complexContent>

</xs:complexType>





<xs:complexType     name      = "myBaseType"    >   <!-- adding more children to all, choice or sequence              -->

  <xs:sequence>

    <xs:element     name      = "name"

                    type      = "xs:string"    />

    <xs:element     name      = "age"

                    type      = "xs:int"/>

  </xs:sequence>

</xs:complexType>

 

<xs:complexType     name      = "myDerivedType" >

  <xs:complexContent>

    <xs:extension   base      = "myBaseType"    >

      <xs:sequence>

        <xs:element name      = "city"              <!-- these elements are appended to the base type's child list    -->

                    type      = "xs:string"     />  <!-- children defined by the base type can't be altered           -->

      </xs:sequence>

    </xs:extension>

  </xs:complexContent>

</xs:complexType>

Restricting inheritance


<xs:simpleType    name  = "myType"    >

  <xs:restriction base  = "xs:string" >

    <xs:length    value = "5"

                  fixed = "true"     />             <!-- derived children are not allowed to override this setting    -->

  </xs:restriction>

</xs:simpleType>
