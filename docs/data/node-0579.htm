Inheritance


#myElement .myNewClass {                           /* inheritance works only within the same media query                   */


  @extend .oldClass;                               /* the current selector inherits all styles applied to .oldClass, even  */

                                                   /* if they are defined in a complex selector (like #myElement.oldClass) */

  @extend .myClass, a:hover;                       /* can inherit multiple rules and target complex base selectors as well */


  @extend .experimentalClass;                      /* don't throw warnings if no selector matches this specification       */



  color: red;

}





%virtualSelector {                                 /* this selector will not be compiled into the CSS, it can only be used */

  color: red;                                      /* as a base class for other (derived) rules                            */

}
