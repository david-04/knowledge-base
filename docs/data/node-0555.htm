Ranges


range = ('A'..'Z')                          # "A", "B", "C", ..., "Z"


1..5                                        # 1-5

1...5                                       # 1-4


range.include?('B')                         # check if the range contains the given element

range.to_a                                  # expand into an array

range.each { |item| ... }                   # iterate over all range elements
