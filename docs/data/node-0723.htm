Templates


<xsl:template match    = "myElement"       >        <!-- a rule to apply to everything matching the given XPath            -->


<xsl:template match    = "myElement"                <!-- a rule applied to everything matching the given XPath             -->

              mode     = "toc"                      <!-- only execute this rule when running in "toc" mode                 -->

              name     = "myTemplate"               <!-- public name to be used with xsl:call-template                     -->

              priority = "100"             >        <!-- only the matching template with the hightes priority is executed  -->


  <xsl:param  name     = "myParameter"              <!-- this template expects a parameter to be passed                    -->

              select   = "myChildElement"  >        <!-- optional: use this XPath's result as the default value            -->

    default value                                   <!-- ... or specify the default value as a template                    -->

  </xsl:param>


  <xsl:value-of select = "$myParameter1"  />        <!-- access the actual parameter value through an XPath expression     -->


</xsl:template>



Template invocation


<xsl:apply-templates />                             <!-- recurse into (and process) all children of the current node       -->





<xsl:apply-templates select = "myElement"           <!-- recurse into (and process) all direct children matching the XPath -->

                     mode   = "toc"    />           <!-- only apply templates that match the given mode                    -->





<xsl:apply-templates ... />


  <xsl:sort   select     = "author"                 <!-- optionally: sort by a given (XPath) child node content            -->

              data-type  = "{text|number}"

              lang       = "en-au"

              case-order = "{lower|upper}-first"

              order      = "{ascending|descending}"/>


  <xsl:with-param name   = "myParameter"            <!-- pass a parameter to the applied templates                         -->

                  select = "@myAttribute"/>         <!-- retrieve the parameter value from this XPath                      -->


</xsl:apply-templates>





<xsl:call-template name="myTemplate">               <!-- call a named template on the current node (not its children!)     -->


  <xsl:with-param name   = "myParameter"            <!-- pass a parameter to the template                                  -->

                  select = "@myAttribute"/>         <!-- retrieve the parameter value from this XPath                      -->


</xsl:call-template>
