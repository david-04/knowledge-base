Component styling


All controls contain different Style elements thatt can be used to modify styles and attach CSS classes, for example


<RowStyle />                                                   <!-- all rows (including header and pager)                           -->

<AlternatingRowStyle />                                        <!-- even rows                                                       -->

<SelectedRowStyle />                                           <!-- the selected row                                                -->

<EditRowStyle />                                               <!-- rows that are currently being edited                            -->

<EmptyDataRowStyle />                                          <!-- no data found (EmptyDataText or EmptyDataTemplate is displayed) -->

<HeaderStyle />                                                <!-- table header                                                    -->

<FooterStyle />                                                <!-- table footer                                                    -->

<PagerStyle />                                                 <!-- GridView's pagination section                                   -->


<SortedAscendingCellStyle />                                   <!-- sorted columns and header cells                                 -->

<SortedAscendingHeaderStyle />

<SortedDescendingCellStyle />

<SortedDescendingHeaderStyle />


All these style elements can be targeted by skins


<asp:RowStyle CssClass="MyRowStyle" runat="server" />



Data styling


<asp:GridView ... OnRowDataBound = "OnRowDataBound" >          <!-- register an OnLoad event handler                                -->


protected void OnRowDataBound(object sender,

                             GridViewRowEventArgs e)

{

    if (e.Row.RowType == DataControlRowType.DataRow)           // only apply to table rows (not headers, pagers, ...)

    {

        DataRowView myDataRowView =

            (DataRowView)e.Row.DataItem;                       // retrieve the current row to process

        if ((decimal)100.0 <=

                    (decimal)myDataRowView["Price"])           // check if a condition is met

        {

            e.Row.ForeColor = System.Drawing.Color.Red;        // apply styles to the individual row


            LinkButton deleteLink =                            // retrieve controls from the current row

                e.Row.FindControl("DeleteLink") as LinkButton;

            if (deleteLink != null)

                deleteLink.Enabled = false;                    // disable features based on the current data

        }

    }

}
