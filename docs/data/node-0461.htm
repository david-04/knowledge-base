Size and position (BOM)


window.screen                                       // contains the screen's pixel/DPI resolution, colour depth, ...

window.screen{Left|Top}                             // window position in all browsers except Firefox (some measure the window, others the viewport)

window.screen{X|Y}                                  // Firefox equivalent of the above

window.{inner|outer}{Width|Height}                  // browser window size (in Opera: only the viewport's size)

window.orientation                                  // 0 (portrait) or +/-90 for rotated


window.moveTo(0, 0);                                // move the window to the upper-left coordinate (usually disabled)

window.moveBy(0, 100);                              // move the window down by 100 pixels (usually disabled)

window.resizeTo(100, 100);                          // resize to 100 x 100 (usually disabled)

window.resizeBy(100, 50);                           // resize to width+200 x height+150 (usually disabled)


element.client{Width|Height}                        // size (in pixels) including the padding only

element.offset{Widht|Height}                        // size (in pixels) including scrollbars, borders and padding (but probably not margin)

element.offset{Left|Top}                            // position relative to the surrounding element's client area

element.offsetParent                                // the surrounding element (e.g. the <table> rather than the <td>)

element.scroll{Width|Height}                        // total content size (including the portions scrolled out of the view)

element.scroll{Top|Left}                            // number of pixels hidden to the top/left of the current scroll view


element.getBoundingClientRect()                     // returns the elements {left|top|right|bottom} relative to the viewport (IE<9 start positioning at (2,2))


document.documentElement.client{Width|Height}       // viewport size

document.body.client{Width|Height}                  // IE 6 and below


function getViewportSize(){                         // browser-senstive calucation of the viewport's size

    if (document.compatMode == "BackCompat"){

        return {

            width: document.body.clientWidth,

            height: document.body.clientHeight

        };       

    } else {

        return {

            width: document.documentElement.clientWidth,

            height: document.documentElement.clientHeight

        };       

    }

}
