        <li data-target="#myCarousel"

            data-slide-to="0" />

        <li data-target="#myCarousel"

            data-slide-to="1"

            class="active" />

    </ol>  


    <div class="carousel-inner">

        <div class="item">

        </div>

    <a class="carousel-control left"

    </a>

    <a class="carousel-control right"

JavaScript API


$("#myCarousel").carousel();                          // start the carousel

$("#myCarousel").carousel("cycle");                   // start the carousel

$("#myCarousel").carousel("pause");                   // pause the carousel

$("#myCarousel").carousel("prev");                    // go to the previous slide

$("#myCarousel").carousel("next");                    // go to the next slide

$("#myCarousel").carousel(2);                         // go to the third slide


$("#myCarousel").carousel({

    interval : {5000|false},                          // auto-browse every 5 seconds (500) or not at all (false)

    pause    : {"hover"|false},                       // pause on hover (or not)

    wrap     : true,                                  // start over again at the end

    keyboard : true                                   // allow sliding via keyboard

});


onslide.bs.carousel                                   // before the slide changes

onslid.bs.carousel                                    // after the slide has been changed
