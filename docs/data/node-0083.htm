Animations


import { trigger,

         state,

         style,

         transition,

         animate,

         keyframes,

         group       } from "@angular/core";


@Component({

    animations: [

        trigger("personState", [                    // symbolic name of the animation

            state("default", style({                // when the trigger value is "default", apply these styles

                color   : "black",                  // styles can be hard-coded

                fontSize: "*"                       // ... or use the current value (for calculated styles like width/height)

            })),

            state("selected", style({               // when the trigger value is "selected", apply these styles

                color   : "red",

                fontSize: "2em"

            })),

            transition(                             // define transitions between the different states

                "default => selected",

                animate("100ms ease-in")

            ),

            transition(

                "selected => default",

                animate("100ms ease-out")

            )

        ])

    ]

})

States


transition(

    "default => selected",                         // single unidirectional transition

    animate("100ms ease-in")

)


"selected => default, default => selected"         // multiple unidirectional transitions in one rule

"selected <=> default"                             // bidirectional transition

"selected => *"                                    // transition from "selected" to any other state (* also matches void)

"void => *"                                        // applied when the is (re-) attached to the DOM and becomes visible

":enter"                                           // the element becomes visible (equals "void => *")

":leave"                                           // the element beomces invisible (equals "* => void)



Timing and easing


transition(

    "default => selected",

    animate("100ms ease-in")                       // timing and easing

)


"100ms ease-in"                                    // animate easing-in for a total of 100ms

"200ms 100ms ease-in"                              // wait 200ms before starting the animation


http://easings.net/                                // animated overview of all available easings



Keyframe animations


transition("void => *", [

    animate(300, keyframes([                       // define discrete keyframes

        style({ color: "red"   , offset: 0.0 }),   // initial state

        style({ color: "yellow", offset: 0.3 }),   // state 30% along the way

        style({ color: "green" , offset: 1.0 })    // end state

    ])

)



Animation groups


transition("void => *", [

    style({ width: 10, opacity: 0 }),               // initial element state

    group([                                         // run all contained animations in parallel

        animate("0.3s 0.1s ease", style({           // animation 1: change the width quickly but with a 300ms delay

            width: 120

        })),

        animate("0.3s ease", style({                // animation 2: change the opacity slowly without delay

            opacity: 1

        }))

    ])

])
