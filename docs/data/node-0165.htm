Event handlers


class App extends React.Component {


    constructor() {

        this.onClick = this.onClick.bind(this);      // the event handler must be wrapped into a bind() call (only required when

    }                                                // using object oriented ES6+ to prevent "this" from being set to null)


    onClick(event) {                                 // implement the event handler as a regular method

        event.preventDefault();                      // event handlers don't have return values (e.g. can't return false to prevent

        // ...                                       // navigation); call preventDefault() instead

    }


    onClick = (event) => {                           // experimental syntax (define and bind the handler in one step)

        event.preventDefault();

        // ...

    }


    render() {

        return (

            <div>

                <button onClick={this.onClick}>      // the event attribute must be camelCased (don't append parentheses!)

                    myButton

                </button>

                <button onClick={                    // not recommended: this syntax can invoke the handler without wrapping it

                            (e) => this.onClick(e)   // into a bind() call, but it might create multiple redundant instances

                        }>                           // of the event handler function (bad for performance)

                    myButton

                </button>

            </div>

        );

    }

}



Initialisers


<input ref={(i) => {this.textInput = i;}} />         // the ref function is executed as soon as the component has been mounted

                                                     // the input paramter "i" is the corresponding DOM node that can be stored

                                                     // for later (the ref function is also called upon dismounting with i==null)


<MyComponent ref={(c) => {this.myComp = c;}} />      // the ref function can be defined for custom components as well, but only

                                                     // if they are defined as classes and not as functions
