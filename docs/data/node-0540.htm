if / unless


if age < 18                                 # if-else

  puts 'child'

elsif age < 67

  puts 'adult'

else

  puts 'pensioner'

end





unless age < 18                             # unless

  puts "minor"

else

  puts "adult"

end





puts "child" if age <12                     # can be used inline

puts "adult" unless age <= 18



switch


case country

when "German"

  continent = "Europe"              # no break required

when "China"

  contient = "Asia"

else

  contient = "unknown"

end





continent = case country            # reult expression can be assigned

            when "German"

              "Europe"              # no break required

            when "China"

              "Asia"

            else

              "unknown"

            end





when 0                                      # exact match

when 0..9                                   # in range

when /[a-z]+/                               # matching regular expression

in { error: message, no: code}              # object with properties error and no (accessible as variables message and code in the when block)



for


for i in 1..10                              # regular counter loop

  redo  if ...                              # repeat without incrementing i

  break if ...                              # exit the loop

end


5.times { puts "test" }                     # does not pass any parameters to the code block


0.upto(5) { |index| puts index }            # from 0 to 5

5.downto(0) { |index| puts index }          # from 5 to 0

0.step(50, 5) { |index| puts index }        # from 0 to 50 in steps of 5



for element in [1, 2]                       # iterate over all elements in the collection

  redo | break                              # (for hashes it iterates over all keys and values separately)

end



loop


loop do                                     # infinite loop

  break if ...

end



while / until


while x < 100                               # while loop

  x = x * 2

  break

end


until x < 1                                 # until loop

  x = x / 2

  break

end


x = x * 2 while x < 1000                    # can be used inline as well

x = x / 2 until x < 100
