public class MyClass {


    public static void main(String[] args)

                  throws Exception {

       

        SwingUtilities.invokeLater(                 // must be run in the swing thread (otherwise events won't be dispatched)

            () -> MyClass.initialiseWindow()        // alternatively call invokeAndWait

        );

    }


    private static void initialiseWindow() {

       

        JFrame frame = new JFrame("My app");        // create a frame (main window)

        frame.setSize(500, 500);

        frame.setDefaultCloseOperation(

                         JFrame.EXIT_ON_CLOSE);

        frame.setLayout(new FlowLayout());

       

        JLabel label = new JLabel("My label");      // add controls

        frame.add(label);                           // JDK5 (in JDK<5 call frame.getcontentPane().add(label)


        JButton button = new JButton("1");

        frame.add(button);

       

        button.addActionListener(                   // event handling works the same as for AWT

            (ActionEvent ae)

            -> label.setText("Click!")

        );

       

        frame.setVisible(true);

    }

}



Windows (AWT)


Component                                           // base class for every GUI element (windows, buttons, checkboxes, ...)

+-- Container                                       // allows other components to be nested inside

|   +-- Window

|   |   +-- Frame                                   // a regular desktop application's top-level window

|   |   +-- Dialog                                  // a window that is shown as modal or non-modal dialog

|   +-- Panel                                       // used as the an applet's "window" (without menu and titlebar)

+-- Canvas                                          // a blank window that can be drawn upon





class MyWindow extends Frame {                      // top-level windows are usually derived from Frame


    public MyWindow(String title) {

        super(title);


        this.addWindowListener(new WindowAdapter(){ // add an event handler to close the window

            public void windowClosing(

                                   WindowEvent e) {

                setVisible(false);                  // ... either by just hiding it

                dispose();                          // ... or by deleting it and freeing all associated resources

            }

        });

    }

   

    public void paint(Graphics g) {                 // the screen representation is composed and updated in the paint method

        g.drawString("Hello world!", 10, 50);

        super.paint(g);                             // it is sometimes necessary to also invoke the base class' paint method

    }


}





MyWindow myWindow = new MyWindow("Title");          // in the main program: create a window instance and make it visible

myWindow.setSize(500, 300);

myWindow.setVisible(true);
