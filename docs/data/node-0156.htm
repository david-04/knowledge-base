Dialogs


$("div#myDialog").dialog({                          // jQuery UI: create and show a dialog from an element (which itself can be set to display:none)

    title     : "My dialog",                        // use this as the dialog's title (default: the element's title attribute)

    modal     : true,                               // make the dialog modal

    autoOpen  : true,                               // immediately display the dialog

    resizable : true,                               // make the dialog resizeable / draggable

    draggable : true,

    show      : 'explode',                          // use an animation effect when making the the dialog visibile

    open      : function(event, ui) { },            // called when the dialog is opened

    close     : function(event, ui) { }             // called when the dialog is closed ("this" is the source element from wich the dialog was created)

});


dialog = $("selector").dialog({autoOpen:false});    // create a dialog but do not show it (yet)

dialog.dialog("open");                              // show the pre-created dialog (use "close" to hide it)



Datepicker


$("input[name=date]").datepicker({                  // jQuery UI: add a datepicker control to a text input field

    dateFormat  : "dd/mm/yy",                       // set the date format used for the input (text) box

    firstDay    : 1,                                // start weeks on Monday (default: 0=Sunday)

    minDate     : Date.now(),                       // the smalles and largest date selectable

    maxDate     : new Date(2019, 12-1,31),

    changeMonth : true,                             // show a drop-down box with month names

    changeYear  : true,                             // show a drop-down box of years

    yearRange   : "2015:2019"                       // show only these years in the drop-down box

});                                                 // see https://api.jqueryui.com/datepicker/ for more options

    </div>

</div>


$("#myTabs").tabs({                                 // jQuery UI: convert a container into a tab environment

    active : 0,                                     // zero-based index of the tab to be selected/displayed initially

    show   : 'explode',                             // use animations when showing/hiding the tabs

    hide   : 'fade'

});



Accordion


<ul style="list-style: none;">                      <!-- jQuery UI: the accordion requires a collection like an unsorted list   -->

    <li>                                            <!-- each section is a list element                                         -->

        <a>Section 1</a>                            <!-- by default the first element (usually <a> is used as the header        -->

        <p>content ...</p>                          <!-- the content can consist of anything (e.g. multiple paragraphs)         -->

    </li>

    <li>

        <h4>Section2</h4>

        <p>content ...</p>

    </li>

</ul>


$('ul').accordion({                                 // jQuery UI: convert a container into an accordion

    active      : {index|false},                    // initially show the pane with the given index (zero-based) or collapse all (if set to false)

    collapsible : true,                             // allow all panes to be closed at the same time

    event       : 'mouseover',                      // don't wait for a click, switch panes right on mouseover

    header      : 'h4',                             // use the first <h4> element within the <li> as the header

    heightStyle : "content",                        // preventive measure to avoid overlapping of content if active==false

});



Drag and drop text


$("#dragSource").draggable({                        // jQuery UI: make an element draggable

    helper  : "clone",                              // keep the original element and drag only a copy

    opacity : 0.5                                   // drag the copy semi-transparently

});                                                 // see https://api.jqueryui.com/draggable/ for more options

       

$("#dropTarget").droppable({                        // jQuery UI: make an element a valid drop-zone

    accept     : "selector",                        // only accept elements that match the given selector

    hoverClass : "hoverClass",                      // add this class while the mouse is hovering the drop-zone

    drop       : function(event, ui) {              // method to handle the actual drop operation

                     ui.draggable[0]                // array of elements being dragged (which is HTML code)

                     ui.draggable[0].remove();      // remove elements from the DOM tree

    }                                               // see https://api.jqueryui.com/droppable/ for more options

});



Re-arrange list items (via drag and drop)


$('ul').sortable({                                  // jQuery UI: make list items in any <ul> container re-arrangable

    opacity     : 0.5,                              // make <li> element semi-transparent while being dragged

    cursor      : 'move',                           // show a cross-arrow cursors

    placeholder : "placeholderCssClass",            // how to style the placeholder (shown where the element would be dropped right now)

    connectWith : 'ul#secondList'                   // allow elements to be dropped onto any list that matches the selector

});


$("ul").sortable("serialize", {                     // jQuery UI: create a query string from the current list order

    attribute: "id",                                // the <li> element's attribute to be used

    expression : /^(.*)$/,                          // important: select the whole attribute value (otherwise only the trailing digits are used)

    key: "li[]"                                     // name of the query string's key

});                                                 // creates: li[]=myId4&li[]=myId1&li[]=myId3...



Draw rectangles to select


.ui-selectable-helper {                             /* Query UI: this class is applied automatically to the rectangle being drawn             */

    position: absolute;                             /* position must be set to absolute, otherwise it does not appear at the right location   */

    background-color: rgba(0, 161, 255, 0.24);      /* style the rectangle as desired                                                         */

}


$("div#container").selectable({                     // jQuery UI: enable drawing a rectangle to select items

    appendTo: 'div#container',                      // the container within which the (temporary) selection box will be drawn

    start: function (event, ui) { },                // called when the selection starts

    stop: function (event, ui) { },                 // called when the selection has finished

    selecting: function (event, ui) {               // called for every element that becomes part of the selection

        $(ui.selecting).css("color", "red");        // the affected elements are passed in as ui.selecting

    },

    unselecting: function (event, ui) {             // called for every element that falls out of the selection

        $(ui.unselecting).css("color", "black");    // the affected elements are passed in as ui.unselecting

    }

});
