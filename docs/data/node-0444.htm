public class MyFxApp extends Application {


    public static void main(String args[]) {

        launch(args);

    }

   

    public void init() {...}                        // override this method only if initialisation is required on start-up

                                                    // do not create any stages or scenes here (nor in the constructor)

   

    public void start(Stage stage)                  // called when the application is started; all stages and scenes must be

                        throws Exception {          // created here, as this method is executed by the "application thread"


        stage.setTitle("My app");                   // the "stage" passed in is the main application window

       

        FlowPane pane = new FlowPane();             // create a layout manager (each window/stage has 1 root pane)

       

        Label label = new Label("Hello world!");    // add components

        pane.getChildren().add(label);


        Button button = new Button("Click here");

        pane.getChildren().add(button);


        button.setOnAction(                         // add event handlers

            (ActionEvent ae)

            -> label.setText("clicked")

        );

       

        Scene scene = new Scene(pane, 500, 500);    // create a scene to wrap the layout manager

        stage.setScene(scene);                      // add the scene to the stage (main window)

        stage.show();                               // show the stage (main window)

    }

   

    public void stop() {...}                        // override this method only if cleanup is required on shut-down

}                                                   // this method runs in the "application thread" and can access the GUI
