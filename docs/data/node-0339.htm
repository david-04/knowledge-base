If


{if $name == 'David'}

    ...

{elseif $id == 170}

    ...

{else}

    ...

{/if}



Switch


{switch $numMarbles}

    {case 0}

        You have no marbles.

    {case 1, 2, 3}

        You have a few marbles.

    {default}

        You have many marbles.

{/switch}



For


{for $i in range(10)}       ... {/for}      // 0...9


{for $i in range(5, 10)}    ... {/for}      // 5...9


{for $i in range(5, 10, 2)} ... {/for}      // 5, 7, 9



Foreach


{foreach $item in $collection}

    {if isFirst($item)} ... {/if}           // only during the first iteration

    {if isLast($item)} ... {/if}            // only during the last iteration

    index($item)                            // zero-based iteration counter

{ifempty}                                   // use the below content if the collection is empty

    ...

{/foreach}
