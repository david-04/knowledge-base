Elements


<xs:element name="myElement" type="xs:string"/>     <!-- inline-definition of an element with a simple type      -->


<xs:element name="myElement" type="myType" />       <!-- element with a custom-defined type (see below)          -->


<xs:element ref="myElement" />                      <!-- mirror another element defined on the root level        -->



Syntax


<xs:element name="myElement">                       <!-- embed the type definition into the element declaration  -->

  <xs:simpleType>

    ...

  </xs:simpleType>

</xs:element>





<xs:simpleType name="myType">                       <!-- define a re-usable type with a dedicated name           -->

  ...

</xs:simpleType>


<xs:element name="myElement" type="myType"/>        <!-- ... and reference the type in the type definition       -->



Customising types


<xs:simpleType         name  = "myNumericType"  >   <!-- numeric constraints                                     -->

  <xs:restriction      base  = "xs:integer"     >

    <xs:maxInclusive   value = "999"           />

    <xs:maxExclusive   value = "1000"          />

    <xs:minInclusive   value = "100"           />

    <xs:minExclusive   value = "99"            />

    <xs:totalDigits    value = "10"            />

    <xs:fractionDigits value = "5"             />

  </xs:restriction>

</xs:simpleType>





<xs:simpleType         name  = "myStringType"   >   <!-- string constraints                                      -->

  <xs:restriction      base  = "xs:string"      >

    <xs:pattern        value = "^[0-9A-Z]+"    />

    <xs:whiteSpace     value = "preserve"      />   <!-- preserve ... retain any kind of whitespace              -->

    <xs:length         value = "20"            />   <!-- replace .... replace [\t\r\n] by blanks                 -->

    <xs:minLength      value = "10"            />   <!-- collapse ... replace and then condense repeated blanks  -->

    <xs:maxLength      value = "50"            />

  </xs:restriction>

</xs:simpleType>





<xs:simpleType         name  = "myEnumType"     >   <!-- enumeration (can contain any of the specified values)   -->

  <xs:restriction      base  = "xs:string"      >

    <xs:enumeration    value = "day"           />

    <xs:enumeration    value = "night"         />

  </xs:restriction>

</xs:simpleType>





<xs:complexType        name  = "myEmptyType"    >   <!-- empty type (may not contain any elements or character   -->

  <xs:attribute        name  = "myAttribute"        <!-- data, useful for attribute-only elements)               -->

                       type  = "xs:string"     />

</xs:complexType>





<xs:simpleType         name  = "myType"         >   <!-- list of values (e.g.: <myElement>1 2 3 4 5</myElement>  -->

  <xs:list             itemType = "xs:int"     />

</xs:simpleType>



Unions


<xs:simpleType         name  = "myType"         >   <!-- union type, can be any of the listed types              -->

  <xs:union memberTypes="xs:string xs:integer" />

</xs:simpleType>
