JavaScript API


$("[data-toggle='tooltip']").tooltip();               // activate tooltips (are disabled by default)

$("[data-toggle='popover']").popover();               // activate popovers (are disabled by default)


$("#myTooltip").tooltip("show");                      // show a tooltip

$("#myTooltip").tooltip("hide");                      // hide a tooltip

$("#myTooltip").tooltip("toggle");                    // switch between show and hide

$("#myTooltip").tooltip("destroy");                   // remove a tooltip


$("#myPopover").popover("show");                      // show a popover

$("#myPopover").popover("hide");                      // hide a popover

$("#myPopover").popover("toggle");                    // switch between show and hide

$("#myPopover").popover("destroy");                   // remove a popover


$(".myTooltipOrPopover").{tooltip|popover}({          // configure tooltips or popovers

    animation : true,                                 // fade in and out

    html      : false,                                // set to true to allow HTML content

    placement : "{top|left|right|bottom|auto}",       // aligment relative to the element

    selector  : ".myClass",                           // attach to the selected HTML elements

    title     : "default tooltip title",              // set the default title if the HTML title attribute is missing

    content   : "default popover content",            // set the default content if the HTML data-content attribute is missing

    trigger   : "{click|hover|focus|manual}",         // can pass multiple values separated by space

    delay     : {0 | {show: 0, hide: 500} },          // delay in milliseconds

    container : {false|"myDiv"},                      // append to the body (false) or a given element ("myDiv")

    viewport: {"#myDiv"                               // keep the tooltip or popover visible within the given element

              |{selector: "#myDiv",padding: 0â€‰}},


                                                      // set the layout template


    template  : "<div class='tooltip' role='tooltip'><div class='tooltip-arrow'></div><div class='tooltip-inner'></div></div>",

    template  : "<div class='popover'><div class='arrow'></div><h3 class='popover-title'></h3><div class='popover-content'></div></div>"

});


onshow.bs.tooltip                                     // before a tooltip is being made visible

oninserted.bs.tooltip                                 // when the tooltip template was added to the DOM (after show but before shown)

onshown.bs.tooltip                                    // after a tooltip has been made visible

onhide.bs.tooltip                                     // before a tooltip is hidden

onhidden.bs.tooltip                                   // after a tooltip has been hidden


onshow.bs.popover                                     // before a popover is being made visible

oninserted.bs.popover                                 // when the popover template was added to the DOM (after show but before shown)

onshown.bs.popover                                    // after a popover has been made visible

onhide.bs.popover                                     // before a popover is hidden

onhidden.bs.popover                                   // after a popover has been hidden
