Custom templates


<asp:TemplateField HeaderText="Full Name">                            <!-- freely configurable custom template           -->


    <ItemTemplate>                                                    <!-- template for displaying (not editable)        -->

        <asp:Label ID="FullName" runat="server"

            Text='<%# Eval("LastName")+","+Eval("FirstName") %>' />   <!-- use Eval() to retrieve attribute data values  -->

    </ItemTemplate>


    <EditItemTemplate>                                                <!-- template for editing (not inserting)          -->

        <asp:TextBox ID="LastNameEdit" runat="server"

            Text='<%# Bind("LastName") %>' />,                        <!-- use Bind for two-way-binding                  -->

        <asp:TextBox ID="FirstNameEdit" runat="server"

            Text='<%# Bind("FirstName") %>' />

    </EditItemTemplate>

   

    <ItemTemplate            ... />                                   <!-- regular style                                 -->

    <InsertItemTemplate      ... />                                   <!-- editable representation for INSERTs           -->

    <HeaderTemplate          ... />                                   <!-- table header cell for this column             -->

    <FooterTemplate          ... />                                   <!-- table footer cell                             -->
