  version="1.0"

  xmlns:xsl="http://www.w3.org/1999/XSL/Transform"

  xmlns:myns="http://my-namespace.com/">            <!-- declare the relevant namespaces on the xsl:stylesheet           -->


  <xsl:template match="myns:myElement" />           <!-- reference the namespace prefix in XPath expressions             -->





<xsl:namespace-alias                                <!-- replace template namespace prefixes with other ones             -->

  stylesheet-prefix = "myns"                        <!-- the prefix used within the current style sheet                  -->

  result-prefix     = "{#default|...}" />           <!-- put the elements in the default namespace or assign the         -->

Output configuration


<xsl:output method         = "{xml|html|text}"      <!-- the target output format                                        -->

            version        = "1.0"                  <!-- output format version (this setting is ignored completely       -->

            encoding       = "utf-8"                <!-- output enconding                                                -->

            standalone     = "{yes|no}"             <!-- create <?xml standalone="{yes|no}"?>                            -->

            indent         = "{yes|no}"             <!-- pretty-print the output (might be ignored by the processor)     -->

            media-type     = "text/html"            <!-- the result's MIME type (probably for browser transformations)   -->

            doctype-public = "-//W3C//DTD HTML 4.01//EN"                <!-- the doctype can't be used to create a       -->

            doctype-system = "http://www.w3.org/TR/html4/strict.dtd"    <!-- simple HTML5 declaration                    -->

            cdata-section-elements= "elem1 elem2.." <!-- output the content of these elements CDATA-wrapped              -->

            omit-xml-declaration  = "{yes|no}"   /> <!-- create (or suppress) the XML prologue                           -->



Fallback


<xsl:loop select="title">                           <!-- when using non-standard (parser-specific) extensions:           -->

  <xsl:fallback>                                    <!-- provide a fallback that's only executed if the XSLT processor   -->

    <xsl:for-each select="title">                   <!-- does not understand the parent element (xsl:loop)               -->

      <xsl:value-of select="."/>

    </xsl:for-each>

  </xsl:fallback>

</xsl:loop>
