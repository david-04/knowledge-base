Future methods


global class FutureClass {                          // https://developer.salesforce.com/docs/atlas.en-us.apexcode.meta/apexcode/apex_invoking_future_methods.htm

    @future                                         // run this method asynchronously in the background

    public static void myFutureMethod() { }         // must be static, can't return anything, can't have sObject as input

}                                                   // also useful when manipulating users (can't be done in one transaction with DML)



Queueable


public class MyClass implements Queueable {         // https://developer.salesforce.com/docs/atlas.en-us.apexcode.meta/apexcode/apex_queueing_jobs.htm

    public void execute(QueueableContext context) {

        Account a = new Account(...);               // perform operatins (DML, web service calls)

        insert a;

    }

}


ID jobID = System.enqueueJob(                       // queue the job

    new MyClass()

);

AsyncApexJob jobInfo = [                            // retrieve job information

        SELECT Status,NumberOfErrors

        FROM AsyncApexJob

        WHERE Id=:jobID

];



Apex scheduler


global class MyClass implements Schedulable {       // https://developer.salesforce.com/docs/atlas.en-us.apexcode.meta/apexcode/apex_scheduler.htm

    global void execute(SchedulableContext SC) {

        // ...

    }

}


String jobID = system.schedule(

        'My job',                                   // the job title

        '20 30 8 10 2 ?',                           // a cron time pattern

        new MyClass()

);


CronTrigger ct = [                                  // retrieve trigger status information

    SELECT TimesTriggered, NextFireTime

    FROM CronTrigger WHERE Id = :jobID

];
