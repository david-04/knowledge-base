                                                    <!-- in combination with XLinks                                       -->

Points and ranges


                                                    <!-- points are always located between to characters                  -->


start-point(//title)                                <!-- start point inside the title (between the end of <title> and the -->

                                                    <!-- first character inside the <title> element                       -->

start-point(//title/text())                         <!-- the same as above (a text node does not have an opening tag)     -->

end-point(//title)                                  <!-- point between the last character of the node's content and the   -->

                                                    <!-- the beginning of the closing </title> tag                        -->


range(//title)                                      <!-- the title-node from start to end (including the opening and      -->

                                                    <!-- closing tags); may return multiple ranges if the XPath has       -->

                                                    <!-- more than one match                                              -->

range-inside(//person)                              <!-- like range(), but excludes the openin and closing tags           -->

//title/range-to(year))                             <!-- works relative to the context node (here: selecte everything     -->

                                                    <!-- from the opening from <title> to the closing </year>)            -->

//title/range-to(text())                            <!-- select everything from <title> to the the end of the text that   -->

                                                    <!-- is contained in the title (i.e. stop before </title>)            -->

string-range(//title, "Wizard")                     <!-- select all <title>s, then strip away all tags, afterwards search -->

                                                    <!-- for "Wizard" and return all matches as ranges                    -->

string-range(//title, "Wizard", 7, 4)               <!-- like above, but select the first 4 characters after "Wizard"     -->

                                                    <!-- (which has 7 characters, the offset)                             -->

string-range(//title, "Wizard", -3, 4)              <!-- select the first 4 characters before "Wizard"                    -->
