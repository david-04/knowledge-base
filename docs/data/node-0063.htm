Animations


<script src="lib/angular-animate.js"></script>      <!-- must be downloaded separately -->


<style type="text/css">                             /* the base name (.ngMyFadeEffect) does not need to start with ng   */

    .ngMyFadeEffect.ng-enter {                      /* ng-enter:  content is shown to the user                          */

        transition: 0.5s linear all;                /* ng-leave:  content is hidden from the user                       */

        opacity: 0;                                 /* ng-move:   content is moved to another DOM location              */

    }                                               /* ng-add:    presumably like ng-enter                              */

    .ngMyFadeEffect.ng-enter-active {               /* ng-remove: presumably like ng-leave                              */

        opacity: 1;                                

    }

</style>


<button ng-click="visible=!visible">Toggle</button> <!-- button to display or hide the following paragraph             -->


<p ng-if="visible" class="ngMyFadeEffect">Hi!</p>   <!-- attaching the class automatically enables the animation       -->


angular.module("myApp", ["ngAnimate"]);             // require the ngAnimate module (angular-animate.js must be installed)



Supported directives


ng-repeat                                           // enter, leave, move

ng-view                                             // enter, leave

ng-include                                          // enter, leave

ng-switch                                           // enter, leave

ng-if                                               // enter, leave

ng-class                                            // add, remove

ng-show                                             // add, remove

ng-hide                                             // add, remove
